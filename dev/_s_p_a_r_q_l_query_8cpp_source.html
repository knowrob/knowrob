51 void SPARQLQuery::add(std::ostream &os, const std::shared_ptr<GraphTerm> &graphTerm) { // NOLINT
52  switch (graphTerm->termType()) {
53  case GraphTermType::Union: {
54  auto sequence = std::static_pointer_cast<GraphSequence>(graphTerm);
55  bool isFirst = true;
56  for (const auto &term: sequence->terms()) {
57  if (!isFirst) {
58  os << " UNION\n";
59  } else {
60  isFirst = false;
61  }
62  os << " {\n";
63  add(os, term);
64  os << " }\n";
65  }
66  break;
67 
68  }
69  case GraphTermType::Sequence: {
70  auto sequence = std::static_pointer_cast<GraphSequence>(graphTerm);
71  for (const auto &term: sequence->terms()) {
72  add(os, term);
73  }
74  break;
75  }
76  case GraphTermType::Pattern:
77  add(os, *std::static_pointer_cast<GraphPattern>(graphTerm)->value());
78  break;
79  case GraphTermType::Builtin:
80  add(os, *std::static_pointer_cast<GraphBuiltin>(graphTerm));
81  break;
82  }
83 }

85 void SPARQLQuery::add(std::ostream &os, const FramedTriplePattern &triplePattern) {
86  os << " ";
87  if (triplePattern.isNegated()) {
88  // Handle negated patterns. There is literature available for negation in SPARQL:
89  // https://ceur-ws.org/Vol-1644/paper11.pdf
90  // The authors list some approaches:
91  // - use OPTIONAL and then check with BOUND that it failed. But that would only do the trick
92  // if the negated pattern actually has some runtime variables, as far as I understand.
93  // - use FILTER NOT EXISTS: `FILTER NOT EXISTS { ?x ?y ?z }`.
94  // - use MINUS: `MINUS { ?x ?y ?z }`.
95  // NOTE: MINUS will not work without a positive statement preceding the negated one
96  // as far as I understand because both can only be used to "eliminate" solutions that were produced before.
97  // Which is actually fine as the KB does order positive/negative literals in the query,
98  // However queries with only negated patterns will not work with this code!
99  if (flags_ & SPARQLFlag::NOT_EXISTS_UNSUPPORTED) {
100  // NOTE: redland does not support NOT-EXISTS or MINUS.
101  negationViaOptional(os, triplePattern);
102  } else {
103  negationViaNotExists(os, triplePattern);
104  }
105  } else if (triplePattern.isOptional()) {
106  if (optional(os, triplePattern)) {
107  filter_optional(os, lastVar_, triplePattern.objectTerm(), triplePattern.objectOperator());
108  }
109  } else {
110  where_with_filter(os, triplePattern);
111  }
112  os << '\n';
113 }

115 void SPARQLQuery::comparison(std::ostream &os, const GraphBuiltin &builtin, const char *comparisonOperator) {
116  // Filter using a comparison operator, also succeed if one of the arguments is not bound.
117  // e.g. `FILTER (!BOUND(?begin) || !BOUND(?end) || ?begin < ?end)`
118  os << " FILTER ( ";
119  if (builtin.isOptional()) {
120  if (builtin.arguments()[0]->isVariable()) {
121  os << "!BOUND(?" << std::static_pointer_cast<Variable>(builtin.arguments()[0])->name() << ") || ";
122  }
123  if (builtin.arguments()[1]->isVariable()) {
124  os << "!BOUND(?" << std::static_pointer_cast<Variable>(builtin.arguments()[1])->name() << ") || ";
125  }
126  }
127  os << '(';
128  where(os, builtin.arguments()[0]);
129  os << comparisonOperator << ' ';
130  where(os, builtin.arguments()[1]);
131  os << "))\n";
132 }

134 void SPARQLQuery::bindOneOfIf(std::ostream &os, const GraphBuiltin &builtin, const char *comparisonOperator) {
135  // e.g. `BIND(IF(BOUND(?optional_val) && !(?begin < ?optional_val), ?optional_val, ?begin) AS ?begin)`
136  // NOTE: Here it is assumed that only the second argument could be undefined in the evaluation context.
137  os << " BIND ( IF( (";
138  if (builtin.arguments()[1]->isVariable()) {
139  os << "BOUND(?" << std::static_pointer_cast<Variable>(builtin.arguments()[1])->name() << ") && ";
140  }
141  os << "(";
142  {
143  where(os, builtin.arguments()[0]);
144  os << ' ' << comparisonOperator << ' ';
145  where(os, builtin.arguments()[1]);
146  }
147  os << ")), ";
148  where(os, builtin.arguments()[1]);
149  os << ", ";
150  where(os, builtin.arguments()[0]);
151  os << ") AS ?" << builtin.bindVar()->name() << ")\n";
152 }

154 void SPARQLQuery::add(std::ostream &os, const GraphBuiltin &builtin) {
155  switch (builtin.builtinType()) {
156  case GraphBuiltinType::Equal:
157  comparison(os, builtin, "=");
158  break;
159  case GraphBuiltinType::Less:
160  comparison(os, builtin, "<");
161  break;
162  case GraphBuiltinType::Greater:
163  comparison(os, builtin, ">");
164  break;
165  case GraphBuiltinType::LessOrEqual:
166  comparison(os, builtin, "<=");
167  break;
168  case GraphBuiltinType::GreaterOrEqual:
169  comparison(os, builtin, ">=");
170  break;
171  case GraphBuiltinType::Bind:
172  os << " BIND (";
173  where(os, builtin.arguments()[0]);
174  os << "AS ?" << builtin.bindVar()->name();
175  os << ")\n";
176  break;
177  case GraphBuiltinType::Min:
178  bindOneOfIf(os, builtin, ">");
179  break;
180  case GraphBuiltinType::Max:
181  bindOneOfIf(os, builtin, "<");
182  break;
183  }
184 }

219 void SPARQLQuery::doFilter(std::ostream &os, std::string_view varName, const std::shared_ptr<Atomic> &atomic, FilterType operatorType) {
220  os << '?' << varName;
221  switch (operatorType) {
222  case FilterType::LT:
223  os << " < ";
224  break;
225  case FilterType::GT:
226  os << " > ";
227  break;
228  case FilterType::LEQ:
229  os << " <= ";
230  break;
231  case FilterType::GEQ:
232  os << " >= ";
233  break;
234  case FilterType::EQ:
235  os << " = ";
236  break;
237  case FilterType::NEQ:
238  os << " != ";
239  break;
240  }
241  if (atomic->isNumeric()) {
242  os << '"' << *atomic << '"' << "^^";
243  iri(os, xsdTypeToIRI(std::static_pointer_cast<XSDAtomic>(atomic)->xsdType()));
244  }
245  else if (atomic->isString()) {
246  os << *atomic << "^^";
247  iri(os, xsdTypeToIRI(std::static_pointer_cast<XSDAtomic>(atomic)->xsdType()));
248  } else {
249  os << atomic->stringForm();
250  }
251  os << " ";
252 }

260 void SPARQLQuery::negationViaOptional(std::ostream &os, const FramedTriplePattern &triplePattern) {
261  if (triplePattern.objectTerm()->isVariable()) {
262  bool hasObjectOperator = optional(os, triplePattern);
263  os << "FILTER (";
264  if (hasObjectOperator) {

266  auto atomic = std::static_pointer_cast<Atomic>(triplePattern.objectTerm());

268  } else {
269  os << " !BOUND(";

305 bool SPARQLQuery::where(std::ostream &os, const FramedTriplePattern &triplePattern) {
306 
307  where(os, triplePattern.subjectTerm());
308  where(os, triplePattern.propertyTerm());
309  if (triplePattern.objectTerm()->isVariable() || triplePattern.objectOperator() == FilterType::EQ) {
310  if (!triplePattern.objectTerm()->isVariable() && triplePattern.objectVariable()) {
311  where(os, triplePattern.objectVariable());
312  os << ". ";
313  lastVar_ = triplePattern.objectVariable()->name();
314  return true;
315  } else {
316  where(os, triplePattern.objectTerm());
317  os << ". ";
318  return false;
319  }
320  } else if (triplePattern.objectVariable()) {
321  // value, operator and variable are provided in the pattern
322  lastVar_ = triplePattern.objectVariable()->name();
323  variables_.insert(triplePattern.objectVariable()->name());
324  os << "?" << lastVar_ << " . ";
325  return true;
326  } else {
327  // we need to introduce a temporary variable to handle value expressions like `<(5)` such
328  // that they can be filtered after the match.
329  static const std::string adhocVarPrefix = "v_adhoc";
330  std::stringstream varName_os;
331  varName_os << adhocVarPrefix << (varCounter_++);
332  lastVar_ = varName_os.str();
333  os << "?" << lastVar_ << " . ";
334  return true;
335  }
336 }

338 void SPARQLQuery::where(std::ostream &os, const TermPtr &term) {
339  switch (term->termType()) {
340  case TermType::VARIABLE: {
341  auto var = std::static_pointer_cast<Variable>(term);
342  variables_.insert(var->name());
343  os << "?" << var->name() << " ";
344  break;
345  }
346  case TermType::ATOMIC: {
347  if (term->isIRI()) {
348  iri(os, std::static_pointer_cast<Atomic>(term)->stringForm());
349  break;
350  } else if (term->isBlank()) {
351  os << "_:" << std::static_pointer_cast<Atomic>(term)->stringForm() << " ";
352  break;
353  }
354  }
355  default:
356  if (term->isNumeric() || term->isString()) {
357  auto xsdAtomic = std::static_pointer_cast<XSDAtomic>(term);
358  if (xsdAtomic->xsdType() == XSDType::BOOLEAN) {
359  auto numeric = std::static_pointer_cast<Numeric>(term);
360  os << (numeric->asBoolean() ? "\"true\"" : "\"false\"");
361  } else if (xsdAtomic->isString()) {
362  os << *term;
363  } else {
364  os << '"' << *term << '"';
365  }
366  os << "^^";
367  iri(os, xsdTypeToIRI(xsdAtomic->xsdType()));
368  os << " ";
369  }
370  break;
371  }
372 }
