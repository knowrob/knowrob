17 template<class T>
18 std::shared_ptr<T> createConnectiveFormula(
19  const FormulaPtr &phi1, const FormulaPtr &phi2, FormulaType type) {
20  // attempt to merge if phi1 or phi2 is already a connective formula of the given type
21  if (phi1->type() == type) {
22  auto *c1 = (T *) phi1.get();
23  if (phi2->type() == type) {
24  auto *c2 = (T *) phi2.get();
25  std::vector<FormulaPtr> args(c1->formulae().size() + c2->formulae().size());
26  for (int i = 0; i < c1->formulae().size(); i++) args[i] = c1->formulae()[i];
27  for (int i = 0; i < c2->formulae().size(); i++) args[i + c1->formulae().size()] = c2->formulae()[i];
28  return std::make_shared<T>(args);
29  } else {
30  std::vector<FormulaPtr> args(c1->formulae().size() + 1);
31  for (int i = 0; i < c1->formulae().size(); i++) args[i] = c1->formulae()[i];
32  args[c1->formulae().size()] = phi2;
33  return std::make_shared<T>(args);
34  }
35  } else if (phi2->type() == type) {
36  auto *c2 = (T *) phi2.get();
37  std::vector<FormulaPtr> args(c2->formulae().size() + 1);
38  args[0] = phi1;
39  for (int i = 0; i < c2->formulae().size(); i++) args[i + 1] = c2->formulae()[i];
40  return std::make_shared<T>(args);
41  } else {
42  return std::make_shared<T>(T(std::vector<FormulaPtr>({phi1, phi2})));
43  }
44 }

126 void ReasoningGraph::disjunction(const ReasoningGraph &other) {
127  // gather initial nodes of this graph without successors
128  std::list<NodePtr> simpleNodes;
129  for (auto &node1: initialNodes_) {
130  if (node1->successors_.empty()) simpleNodes.push_back(node1);
131  }
132  // iterate over initial nodes of the other graph.
133  // if the node has no successors it may be combinable into a disjunctive query
134  // with a node in simpleNodes.
135  for (auto &node2: other.initialNodes()) {
136  if (node2->successors_.empty()) {
137  bool hasNode = false;
138  for (auto &simple1: simpleNodes) {
139  if (simple1->canBeCombinedWith(node2, CAPABILITY_DISJUNCTIVE_QUERIES)) {
140  std::set<std::shared_ptr<DefinedReasoner>> reasonerChoices;
141  for (auto &reasonerChoice: node2->reasonerAlternatives_)
142  if (simple1->reasonerAlternatives_.find(reasonerChoice) != simple1->reasonerAlternatives_.end())
143  reasonerChoices.insert(reasonerChoice);
144 
145  auto phi = createConnectiveFormula<Disjunction>(
146  simple1->phi_, node2->phi_, FormulaType::DISJUNCTION);
147  auto disjunctiveNode = std::make_shared<Node>(
148  phi, reasonerChoices, PredicateType::FORMULA);
149  initialNodes_.remove(simple1);
150  initialNodes_.push_back(disjunctiveNode);
151  simpleNodes.remove(simple1);
152  simpleNodes.push_back(disjunctiveNode);
153  hasNode = true;
154  break;
155  }
156  }
157  if (hasNode) continue;
158  }
159  initialNodes_.push_back(node2);
160  }
161 }

163 void ReasoningGraph::conjunction(const ReasoningGraph &other) {
164  // create a copy of terminal nodes list as it is modified below
165  auto terminals = terminalNodes();
166  // look at each pair of terminal node of this, and initial node of the other graph.
167  // if possible, create a conjunctive query node, else add edges between the
168  // nodes to connect both graphs.
169  for (auto &terminal1: terminals) {
170  for (auto &initial2: other.initialNodes()) {
171  if (terminal1->canBeCombinedWith(initial2, CAPABILITY_CONJUNCTIVE_QUERIES)) {
172  // create a conjunctive node (terminal1 AND initial2)
173  addConjunctiveNode(terminal1, initial2);
174  } else if (isEdgeNeeded(terminal1, initial2)) {
175  // terminal1 and initial2 cannot be combined via conjunction.
176  // but it could be that one of the nodes refers to a builtin
177  // in which case the edge is omittable for some cases.
178  addSuccessor(terminal1, initial2);
179  }
180  }
181  // remove dangling node in this graph, i.e. terminal nodes not connected
182  // to the other graph.
183  // NOTE: dangling nodes in the other graph are not referred to in this graph, so can be ignored
184  if (terminal1->successors_.empty()) removeNode(terminal1);
185  }
186 }

188 void ReasoningGraph::addConjunctiveNode(const NodePtr &a, const NodePtr &b) {
189  std::set<std::shared_ptr<DefinedReasoner>> reasonerChoices;
190  for (auto &r1: a->reasonerAlternatives_)
191  if (b->reasonerAlternatives_.find(r1) != b->reasonerAlternatives_.end()) reasonerChoices.insert(r1);
192  // create a conjunctive node
193  auto phi = createConnectiveFormula<Conjunction>(
194  a->phi_, b->phi_, FormulaType::CONJUNCTION);
195  auto conjunctiveNode = std::make_shared<Node>(
196  phi, reasonerChoices, PredicateType::FORMULA);
197 
198  // connect the node to the rest of the graph
199  if (a->predecessors_.empty()) {
200  initialNodes_.push_back(conjunctiveNode);
201  } else {
202  for (auto &x: a->predecessors_)
203  addSuccessor(x, conjunctiveNode);
204  }
205  if (b->successors_.empty()) {
206  terminalNodes_.push_back(conjunctiveNode);
207  } else {
208  for (auto &x: b->successors_)
209  addSuccessor(conjunctiveNode, x);
210  }
211 }

213 bool ReasoningGraph::isEdgeNeeded(const NodePtr &a, const NodePtr &b) {
214  if (a->predicateType_ == PredicateType::BUILT_IN) {
215  // no edge needed if phi_a is a builtin supported by reasoner_b
216  if (b->isBuiltinSupported(*(Predicate *) a->phi_.get()))
217  return false;
218  }
219  if (b->predicateType_ == PredicateType::BUILT_IN) {
220  // no edge needed if phi_b is a builtin supported by reasoner_a
221  if (a->isBuiltinSupported(*(Predicate *) b->phi_.get()))
222  return false;
223  }
224  return true;
225 }
