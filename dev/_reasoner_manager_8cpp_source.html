51 std::shared_ptr<NamedReasoner> ReasonerManager::loadPlugin(const boost::property_tree::ptree &config) {
52  // get a reasoner factory
53  std::shared_ptr<ReasonerFactory> factory = findFactory(config);
54  // make sure factory was found above
55  if (!factory) throw ReasonerError("failed to load a reasoner.");
56  // create a reasoner id, or use name property
57  std::string reasonerID = getPluginID(factory, config);
58  KB_INFO("Using reasoner `{}` with type `{}`.", reasonerID, factory->name());
59 
60  // create a new reasoner instance
61  auto reasoner = factory->create(reasonerID);
62  // reasoner need to have a reference to the reasoner manager such that
63  // predicates can be defined that interact with the KB
64  reasoner->value()->setReasonerManager(this);
65  reasoner->value()->setReasonerName(reasonerID);
66 
67  auto backendName = config.get_optional<std::string>("data-backend");
68  if (backendName.has_value()) {
69  auto definedBackend = backendManager_->getPluginWithID(backendName.value());
70  if (definedBackend) {
71  setDataBackend(reasoner->value(), definedBackend->value());
72  } else {
73  throw ReasonerError("Reasoner `{}` refers to unknown data-backend `{}`.", reasonerID, backendName.value());
74  }
75  } else {
76  // check if reasoner implements DataBackend interface
77  auto backend = std::dynamic_pointer_cast<DataBackend>(reasoner->value());
78  if (backend) {
79  setDataBackend(reasoner->value(), backend);
80  backendManager_->addPlugin(reasonerID, backend);
81  } else {
82  throw ReasonerError("Reasoner `{}` has no 'data-backend' configured.", reasonerID);
83  }
84  }
85  auto definedReasoner = addPlugin(reasonerID, reasoner->value());
86 
87  PropertyTree pluginConfig(&config);
88  if (!reasoner->value()->initializeReasoner(pluginConfig)) {
89  KB_WARN("Reasoner `{}` failed to loadConfig.", reasonerID);
90  } else {
91  // load the reasoner-specific data sources.
92  for (auto &dataSource: pluginConfig.dataSources()) {
93  if (!reasoner->value()->loadDataSource(dataSource)) {
94  KB_WARN("Reasoner `{}` failed to load data source {}.", reasonerID, dataSource->uri());
95  }
96  }
97  }
98 
99  return definedReasoner;
100 }

102 std::shared_ptr<NamedReasoner>
103 ReasonerManager::addPlugin(std::string_view reasonerID, const std::shared_ptr<Reasoner> &reasoner) {
104  if (pluginPool_.find(reasonerID) != pluginPool_.end()) {
105  KB_WARN("overwriting reasoner with name '{}'", reasonerID);
106  }
107  auto managedReasoner = std::make_shared<NamedReasoner>(reasonerID, reasoner);
108  pluginPool_.emplace(managedReasoner->name(), managedReasoner);
109  reasoner->setReasonerManager(this);
110  reasoner->setReasonerName(reasonerID);
111  initPlugin(managedReasoner);
112  // indicate that the origin `reasonerID` is a reasoner, and thus belongs to the session
113  backendManager_->vocabulary()->importHierarchy()->addDirectImport(
114  backendManager_->vocabulary()->importHierarchy()->ORIGIN_REASONER, reasonerID);
115 
116  // check if reasoner implements DataBackend interface
117  auto backend = std::dynamic_pointer_cast<DataBackend>(reasoner);
118  if (backend) {
119  setDataBackend(reasoner, backend);
120  backendManager_->addPlugin(reasonerID, backend);
121  }
122 
123  return managedReasoner;
124 }
