60 bool Cursor::next(const bson_t **doc, bool ignore_empty) {
61  if (cursor_ == nullptr) {
62  if (isAggregateQuery_) {
63  cursor_ = mongoc_collection_aggregate(
64  collection_->coll(), MONGOC_QUERY_NONE, query_, opts_, nullptr /* read_prefs */ );
65  } else {
66  cursor_ = mongoc_collection_find_with_opts(
67  collection_->coll(), query_, opts_, nullptr /* read_prefs */ );
68  }
69  if (limit_ > 0) {
70  mongoc_cursor_set_limit(cursor_, limit_);
71  }
72  // make sure cursor has no error after creation
73  bson_error_t err1;
74  if (mongoc_cursor_error(cursor_, &err1)) {
75  throw MongoException("cursor_error", err1);
76  }
77  }
78  // get next document
79  if (!mongoc_cursor_next(cursor_, doc)) {
80  // make sure cursor has no error after next has been called
81  bson_error_t err2;
82  if (mongoc_cursor_error(cursor_, &err2)) {
83  throw MongoException("cursor_error", err2);
84  }
