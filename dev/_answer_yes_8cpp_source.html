32 bool AnswerYes::isRicherThan(const AnswerYes &other) const {
33  if (isUncertain() != other.isUncertain()) {
34  // certain answer is richer than uncertain answer
35  return isUncertain() < other.isUncertain();
36  } else if (substitution_->size() != other.substitution_->size()) {
37  // answer with more substitutions is richer
38  return substitution_->size() > other.substitution_->size();
39  } else if (positiveGroundings_.size() != other.positiveGroundings_.size()) {
40  // answer with more positive groundings is richer
41  return positiveGroundings_.size() > other.positiveGroundings_.size();
42  } else if (negativeGroundings_.size() != other.negativeGroundings_.size()) {
43  // answer with more negative groundings is richer
44  return negativeGroundings_.size() > other.negativeGroundings_.size();

46  return false;

69 bool AnswerYes::mergeWith(const AnswerYes &other, bool ignoreInconsistencies) {
70  auto mergedBindings = std::make_shared<Bindings>(*substitution_);
71  if (ignoreInconsistencies) {
72  // insert all substitutions of other answer, possibly overwriting existing ones
73  *mergedBindings += *other.substitution_;
74  } else {
75  // unify substitutions
76  if (!mergedBindings->unifyWith(*other.substitution_)) {
77  // unification failed -> results cannot be combined
78  return false;
79  }
80  }
81  substitution_ = mergedBindings;
82 
83  if (!frame_->mergeWith(*other.frame_)) {
84  // merging frames failed -> results cannot be combined
85  return false;
86  }
87 
88  reasonerTerm_ = {};
89  // insert groundings of other answer
90  positiveGroundings_.insert(positiveGroundings_.end(),
91  other.positiveGroundings_.begin(), other.positiveGroundings_.end());
92  negativeGroundings_.insert(negativeGroundings_.end(),
93  other.negativeGroundings_.begin(), other.negativeGroundings_.end());
94

167  AnswerPtr mergePositiveAnswers(const AnswerYesPtr &a, const AnswerYesPtr &b, bool ignoreInconsistencies) {
168  if (!a || a->isGenericYes()) {
169  return b;
170  } else if (!b || b->isGenericYes()) {
171  return a;
172  } else {
173  AnswerYesPtr smaller, larger;
174  if (a->isRicherThan(*b)) {
175  smaller = b;
176  larger = a;
177  } else {
178  smaller = a;
179  larger = b;
180  }
181  auto mergedAnswer = std::make_shared<AnswerYes>(*larger);
182  if (mergedAnswer->mergeWith(*smaller, ignoreInconsistencies)) {
183  return mergedAnswer;
184  } else {
185  // merging failed
186  return {};
187  }
188  }
