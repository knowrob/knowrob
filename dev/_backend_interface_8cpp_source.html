17 std::shared_ptr<transaction::Transaction> BackendInterface::createTransaction(
18  const QueryableBackendPtr &queryable,
19  TransactionType transactionType,
20  BackendSelection transactionTargets,
21  const std::vector<std::shared_ptr<NamedBackend>> &backends) {
22  std::shared_ptr<transaction::Transaction> transaction;
23  if (transactionType == Insert) {
24  transaction = std::make_shared<transaction::Insert>(queryable, vocabulary());
25  } else {
26  transaction = std::make_shared<transaction::Remove>(queryable, vocabulary());
27  }
28  if (transactionTargets == Including) {
29  for (auto &backend: backends) {
30  transaction->addBackend(backend);
31  }
32  } else {
33  for (auto &definedBackend: backendManager_->plugins()) {
34  auto &backend = definedBackend.second->value();
35 
36  bool skip = false;
37  if (transactionTargets == Excluding) {
38  for (auto &excluded: backends) {
39  if (excluded && backend == excluded->value()) {
40  skip = true;
41  break;
42  }
43  }
44  }
45  if (skip) continue;
46  transaction->addBackend(definedBackend.second);
47  }
48  }
49  return transaction;
50 }

52 bool BackendInterface::removeAllWithOrigin(std::string_view origin) {
53  // remove all triples with a given origin from all backends.
54  std::vector<std::shared_ptr<ThreadPool::Runner>> transactions;
55  for (auto &it: backendManager_->plugins()) {
56  auto definedBackend = it.second;
57  // create a worker goal that performs the transaction
58  auto transaction = std::make_shared<ThreadPool::LambdaRunner>(
59  [definedBackend, origin](const ThreadPool::LambdaRunner::StopChecker &) {
60  if (definedBackend->value()->removeAllWithOrigin(origin)) {
61  // unset version of origin in backend
62  definedBackend->value()->setVersionOfOrigin(origin, std::nullopt);
63  } else {
64  KB_WARN("removal of triples with origin '{}' from backend '{}' failed!", origin,
65  definedBackend->name());
66  }
67  });
68  // push goal to thread pool
69  DefaultThreadPool()->pushWork(
70  transaction,
71  [definedBackend](const std::exception &exc) {
72  KB_ERROR("transaction failed for backend '{}': {}", definedBackend->name(), exc.what());
73  });
74  transactions.push_back(transaction);
75  }
76 
77  // wait for all transactions to finish
78  for (auto &transaction: transactions) transaction->join();
79 
80  // remove origin from import hierarchy
81  if (!vocabulary()->importHierarchy()->isReservedOrigin(origin)) {
82  vocabulary()->importHierarchy()->removeCurrentGraph(origin);
83  }
84 
85  return true;
86 }

88 bool BackendInterface::mergeInsert(const QueryableBackendPtr &backend, const FramedTriple &triple) {
89  auto pat = std::make_shared<FramedTriplePattern>(triple);
90  // Match triples where interval intersection is not empty
91  pat->setIsOccasionalTerm(groundable(Numeric::trueAtom()));
92  // Construct a merged triple
93  FramedTripleView mergedTriple(triple);
94  // TODO: query for individual name of reified triple and hand it to remove.
95  // can be done by using query of original backend, or if the query can be changed to return the id
96  std::vector<FramedTriplePtr> overlappingTriples;
97  match(backend, *pat, [&](const FramedTriplePtr &matchedTriple) {
98  if (mergedTriple.mergeFrame(*matchedTriple)) {
99  auto &x = overlappingTriples.emplace_back();
100  x.owned = true;
101  // try to take over ownership of matched triple
102  if (matchedTriple.owned) {
103  x.ptr = matchedTriple.ptr;
104  matchedTriple.owned = false;
105  } else {
106  x.ptr = new FramedTripleCopy(*matchedTriple);
107  }
108  }
109  });
110  if (!overlappingTriples.empty()) {
111  // remove overlapping triples if any
112  auto container = std::make_shared<ProxyTripleContainer>(&overlappingTriples);
113  createTransaction(backend, Remove)->commit(container);
114  }
115  // Insert the triple after merging with overlapping existing ones
116  createTransaction(backend, Insert)->commit(mergedTriple);
117  return true;
118 }

136 void BackendInterface::foreach(const QueryableBackendPtr &backend, const TripleVisitor &visitor) {
137  if (backend->supports(BackendFeature::TripleContext)) {
138  backend->foreach(visitor);
139  return;
140  }
141 
142  // fill a container that reverses a reification.
143  UnReificationContainer unReifiedTriples;
144  // the UnReificationContainer uses triple views, but memory of original triples can be
145  // lost in the loop. So we need to store the original triples in a vector, and create
146  // a view on them in the UnReificationContainer.
147  std::vector<FramedTriplePtr> originalTriples;
148  // finally loop over all original triples
149  backend->foreach([&](const FramedTriplePtr &triple) {
150  if (ReifiedTriple::isPartOfReification(*triple)) {
151  auto &copy = originalTriples.emplace_back();
152  copy.owned = true;
153  // try to take over ownership of triple
154  if (triple.owned) {
155  copy.ptr = triple.ptr;
156  triple.owned = false;
157  } else {
158  copy.ptr = new FramedTripleCopy(*triple.ptr);
159  }
160  unReifiedTriples.add(*copy.ptr);
161  } else {
162  visitor(triple);
163  }
164  });
165  // after looping over all original triples, also visit the un-reified ones
166  for (auto &triple: unReifiedTriples) {
167  visitor(triple);
168  }
169 }

171 void BackendInterface::batch(const QueryableBackendPtr &backend, const TripleHandler &callback) {
172  if (backend->supports(BackendFeature::TripleContext)) {
173  backend->batch(callback);
174  return;
175  }
176 
177  auto batchSize = GlobalSettings::batchSize();
178  // fill a container that reverses a reification.
179  UnReificationContainer unReifiedTriples;
180  // take over ownership of triples in batches that need to be reified.
181  // note: reified triples could be split into multiple batches which makes
182  // the collapsing of them more difficult.
183  // to this end we defer the collapsing until the batchDirect call has completed
184  // while taking over ownership of the reified triples to avoid copies and allow
185  // the use of views in the UnReificationContainer.
186  std::vector<FramedTriplePtr> reificationTriples;
187  auto batch = std::make_shared<TripleViewBatch>(batchSize);
188  backend->batch([&](const TripleContainerPtr &triples) {
189  for (auto &triple: *triples) {
190  if (ReifiedTriple::isPartOfReification(*triple.ptr)) {
191  // take over ownership of triple
192  if (triple.owned) {
193  triple.owned = false;
194  auto &newOwner = reificationTriples.emplace_back(triple.ptr);
195  unReifiedTriples.add(*newOwner.ptr);
196  } else {
197  auto &copy = reificationTriples.emplace_back(new FramedTripleCopy(*triple.ptr));
198  unReifiedTriples.add(*copy.ptr);
199  }
200  } else {
201  batch->add(triple);
202  if (batch->size() >= batchSize) {
203  callback(batch);
204  batch->reset();
205  }
206  }
207  }
208  });
209  for (auto &triple: unReifiedTriples) {
210  batch->add(triple);
211  if (batch->size() >= batchSize) {
212  callback(batch);
213  batch->reset();
214  }
215  }
216  if (!batch->empty()) {
217  callback(batch);
218  }
219 }

221 static void setReifiedVariables( // NOLINT(misc-no-recursion)
222  const std::shared_ptr<GraphTerm> &t,
223  const std::map<std::string_view, VariablePtr> &variables) {
224  switch (t->termType()) {
225  case knowrob::GraphTermType::Pattern: {
226  auto &pattern = std::static_pointer_cast<GraphPattern>(t)->value();
227  if (!pattern->propertyTerm() || !pattern->propertyTerm()->isAtomic()) break;
228  auto atomic = std::static_pointer_cast<Atomic>(pattern->propertyTerm());
229  auto needle = variables.find(atomic->stringForm());
230  if (needle != variables.end()) {
231  pattern->setObjectVariable(needle->second);
232  }
233  break;
234  }
235  case knowrob::GraphTermType::Union:
236  case knowrob::GraphTermType::Sequence: {
237  auto connective = std::static_pointer_cast<GraphConnective>(t);
238  for (auto &term: connective->terms()) {
239  setReifiedVariables(term, variables);
240  }
241  break;
242  }
243  case knowrob::GraphTermType::Builtin:
244  break;
245  };
246 }

248 void BackendInterface::match(const QueryableBackendPtr &backend, const FramedTriplePattern &q,
249  const TripleVisitor &visitor) const {
250  static auto ctx = std::make_shared<QueryContext>();
251  if (backend->supports(BackendFeature::TripleContext)) {
252  backend->match(q, visitor);
253  } else {
254  auto flags = ReifiedQuery::getReificationFlags(q);
255  if (flags & IncludeOriginal) {
256  backend->match(q, visitor);
257  }
258  if (flags & IncludeReified) {
259  static auto v_o = std::make_shared<Variable>("o");
260  static auto v_begin = std::make_shared<Variable>("begin");
261  static auto v_end = std::make_shared<Variable>("end");
262  static auto v_confidence = std::make_shared<Variable>("confidence");
263  static auto v_uncertain = std::make_shared<Variable>("uncertain");
264  static auto v_occasional = std::make_shared<Variable>("occasional");
265 
266  auto reified = std::make_shared<ReifiedQuery>(q, vocabulary(), true);
267  // insert variables for contextual parameters
268  setReifiedVariables(reified->term(), {
269  {reification::hasBeginTime->stringForm(), v_begin},
270  {reification::hasEndTime->stringForm(), v_end},
271  {reification::hasConfidence->stringForm(), v_confidence},
272  {reification::isUncertain->stringForm(), v_uncertain},
273  {reification::isOccasional->stringForm(), v_occasional}
274  });
275 
276  backend->query(reified, [&](const BindingsPtr &bindings) {
277  FramedTriplePtr triple;
278  triple.ptr = new FramedTripleView();
279  triple.owned = true;
280 
281  if (q.instantiateInto(*triple, bindings)) {
282  if (bindings->contains(v_begin->name())) {
283  auto &t_begin = bindings->get(v_begin->name());
284  if (t_begin->isNumeric()) {
285  triple->setBegin(std::static_pointer_cast<Numeric>(t_begin)->asDouble());
286  }
287  }
288  if (bindings->contains(v_end->name())) {
289  auto &t_end = bindings->get(v_end->name());
290  if (t_end->isNumeric()) {
291  triple->setEnd(std::static_pointer_cast<Numeric>(t_end)->asDouble());
292  }
293  }
294  if (bindings->contains(v_confidence->name())) {
295  auto &t_confidence = bindings->get(v_confidence->name());
296  if (t_confidence->isNumeric()) {
297  triple->setConfidence(std::static_pointer_cast<Numeric>(t_confidence)->asDouble());
298  }
299  }
300  if (bindings->contains(v_uncertain->name())) {
301  auto &t_uncertain = bindings->get(v_uncertain->name());
302  if (t_uncertain->isNumeric()) {
303  triple->setIsUncertain(std::static_pointer_cast<Boolean>(t_uncertain)->asBoolean());
304  }
305  } else {
306  triple->setIsUncertain(false);
307  }
308  if (bindings->contains(v_occasional->name())) {
309  auto &t_occasional = bindings->get(v_occasional->name());
310  if (t_occasional->isNumeric()) {
311  triple->setIsOccasional(std::static_pointer_cast<Boolean>(t_occasional)->asBoolean());
312  }
313  } else {
314  triple->setIsOccasional(false);
315  }
316  visitor(triple);
317  }
318  });
319  }
320  }

337 void BackendInterface::pushIntoCursor(const QueryableBackendPtr &backend, const GraphPathQueryPtr &q,
338  const TokenBufferPtr &resultStream) const {
339  auto expanded = backend->expand(q);
340  auto channel = TokenStream::Channel::create(resultStream);
341  try {
342  bool hasPositiveAnswer = false;
343  query(backend, expanded->expanded, [&](const BindingsPtr &bindings) {
344  channel->push(backend->yes(expanded, bindings));
345  hasPositiveAnswer = true;
346  });
347  if (!hasPositiveAnswer) {
348  channel->push(backend->no(q));
349  }
350  channel->push(EndOfEvaluation::get());
351  }
352  catch (const std::exception &e) {
353  // make sure EOS is pushed to the stream
354  channel->push(EndOfEvaluation::get());
355  throw;
356  }
