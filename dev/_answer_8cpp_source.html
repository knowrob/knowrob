107  bool AnswerComparator::operator()(const AnswerPtr &v0, const AnswerPtr &v1) const {
108  if (!v0) {
109  if (!v1) return false;
110  else return true;
111  } else if (!v1 || v0 == v1) {
112  return false;
113  } else if (v0->isCertain() != v1->isCertain()) {
114  return v0->isCertain() < v1->isCertain();
115  } else if (v0->isPositive() != v1->isPositive()) {
116  return v0->isPositive() < v1->isPositive();
117  } else if (v0->isNegative() != v1->isNegative()) {
118  return v0->isNegative() > v1->isNegative();
119  } else {
120  return v0->hash() < v1->hash();
121  }
