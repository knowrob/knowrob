115 static bool bson_iter_append_array(bson_iter_t *iter, PlTail *pl_array)
116 {
117  static PlAtom ATOM_true("true");
118  static PlAtom ATOM_false("false");
119 
120  if(BSON_ITER_HOLDS_UTF8(iter)) {
121  pl_array->append(PlCompound("string",
122  PlTerm(bson_iter_utf8(iter, nullptr))));
123  }
124  else if(BSON_ITER_HOLDS_DOCUMENT(iter)) {
125  const uint8_t *data = nullptr;
126  uint32_t len = 0;
127  bson_iter_document(iter, &len, &data);
128  bson_t *doc = bson_new_from_data(data, len);
129  pl_array->append(bson_to_term(doc));
130  }
131  else if(BSON_ITER_HOLDS_ARRAY(iter)) {
132  const uint8_t *array = nullptr;
133  uint32_t array_len = 0;
134  bson_iter_array(iter, &array_len, &array);
135  bson_t *inner_doc = bson_new_from_data(array, array_len);
136  bson_iter_t inner_iter;
137  PlTerm inner_term;
138  PlTail inner_array(inner_term);
139  if(bson_iter_init(&inner_iter, inner_doc)) {
140  while(bson_iter_next(&inner_iter)) {
141  bson_iter_append_array(&inner_iter,&inner_array);
142  }
143  }
144  inner_array.close();
145  pl_array->append(PlCompound("array",inner_term));
146  }
147  else if(BSON_ITER_HOLDS_DOUBLE(iter)) {
148  pl_array->append(PlCompound("double",
149  PlTerm(bson_iter_double(iter))));
150  }
151  else if(BSON_ITER_HOLDS_DECIMAL128(iter)) {
152  bson_decimal128_t v_decimal128;
153  bson_iter_decimal128(iter, &v_decimal128);
154  bson_visit_decimal128(iter, &v_decimal128, pl_array);
155  }
156  else if(BSON_ITER_HOLDS_INT32(iter)) {
157  pl_array->append(PlCompound("int",
158  PlTerm((long)bson_iter_int32(iter))));
159  }
160  else if(BSON_ITER_HOLDS_INT64(iter)) {
161  pl_array->append(PlCompound("int",
162  PlTerm((long)bson_iter_int32(iter))));
163  }
164  else if(BSON_ITER_HOLDS_BOOL(iter)) {
165  if(bson_iter_bool(iter)) {
166  pl_array->append(PlCompound("bool", PlTerm(ATOM_true)));
167  }
168  else {
169  pl_array->append(PlCompound("bool", PlTerm(ATOM_false)));
170  }
171  }
172  else if(BSON_ITER_HOLDS_DATE_TIME(iter)) {
173  double sec_since_epoch = ((double)bson_iter_date_time(iter))/1000.0;
174  pl_array->append(PlCompound("time", PlTerm(sec_since_epoch)));
175  }
176  else {
177  bson_type_t iter_t = bson_iter_type(iter);
178  std::cout << "WARN: unsupported array type '" << iter_t << "'" << std::endl;
179  return false;
180  }
181  return true;
182 }

223 PlTerm bson_to_term(const bson_t *doc)
224 {
225  static bson_visitor_t visitor = get_bson_visitor();
226  bson_iter_t iter;
227  PlTerm term;
228  PlTail out_list(term);
229  if (bson_iter_init(&iter, doc)) {
230  if(bson_iter_visit_all(&iter, &visitor, &out_list)) {
231  bson_error_t err;
232  bson_set_error(&err,
233  MONGOC_ERROR_BSON,
234  MONGOC_ERROR_BSON_INVALID,
235  "BSON iteration prematurely stopped.");
236  throw knowrob::mongo::MongoException("bson", err);
237  }
238  }
239  out_list.close();
240  return term;
241 }

243 static bool bsonpl_append_typed(bson_t *doc, const char *key, const PlTerm &term, bson_error_t *err)
244 {
245  static PlAtom ATOM_array("array");
246  static PlAtom ATOM_time("time");
247  static PlAtom ATOM_id("id");
248  static PlAtom ATOM_regex("regex");
249  static PlAtom ATOM_double("double");
250  static PlAtom ATOM_bool("bool");
251  static PlAtom ATOM_string("string");
252  static PlAtom ATOM_integer("integer");
253  static PlAtom ATOM_int("int");
254  static PlAtom ATOM_true("true");
255  static PlAtom ATOM_false("false");
256 
257  const PlAtom type_atom(term.name());
258  const PlTerm &pl_value = term[1];
259  if(type_atom == ATOM_string) {
260  BSON_APPEND_UTF8(doc, key, (char*)pl_value);
261  }
262  else if(type_atom == ATOM_id) {
263  bson_oid_t oid;
264  bson_oid_init_from_string(&oid, (char*)pl_value);
265  BSON_APPEND_OID(doc,key,&oid);
266  }
267  else if(type_atom == ATOM_double) {
268  // NOTE: must use canonical form for "Infinity"
269  bson_decimal128_t dec;
270  bson_decimal128_from_string ((char*)pl_value, &dec);
271  BSON_APPEND_DECIMAL128(doc,key,&dec);
272  }
273  else if(type_atom == ATOM_time) {
274  BSON_APPEND_DATE_TIME(doc,key,(unsigned long long)(1000.0*((double)pl_value)));
275  }
276  else if(type_atom == ATOM_int || type_atom == ATOM_integer) {
277  BSON_APPEND_INT32(doc,key,(int)pl_value);
278  }
279  else if(type_atom == ATOM_bool) {
280  if(pl_value == ATOM_true) {
281  BSON_APPEND_BOOL(doc,key,true);
282  }
283  else if(pl_value == ATOM_false) {
284  BSON_APPEND_BOOL(doc,key,false);
285  }
286  else {
287  BSON_APPEND_BOOL(doc,key,(bool)((int)pl_value));
288  }
289  }
290  else if(type_atom == ATOM_regex) {
291  BSON_APPEND_REGEX(doc,key,(char*)pl_value,"msi");
292  }
293  else if(type_atom == ATOM_array) {
294  PlTail pl_list(pl_value);
295  PlTerm pl_member;
296  bson_t bson_array;
297  int counter=0;
298  const char *index_key;
299  char index_str[16];
300  bool status=true;
301 
302  BSON_APPEND_ARRAY_BEGIN(doc, key, &bson_array);
303  while(pl_list.next(pl_member)) {
304  bson_uint32_to_string(counter, &index_key, index_str, sizeof index_str);
305  if(!bsonpl_append(&bson_array,index_key,pl_member,err)) {
306  status=false;
307  break;
308  }
309  counter+=1;
310  }
311  bson_append_array_end(doc, &bson_array);
312  return status;
313  }
314  else {
315  bson_set_error(err,
316  MONGOC_ERROR_BSON,
317  MONGOC_ERROR_BSON_INVALID,
318  "invalid type: %s", (char*)term
319  );
320  return false;
321  }
322  return true;
323 }

325 bool bsonpl_append(bson_t *doc, const char *key, const PlTerm &term, bson_error_t *err)
326 {
327  if(PL_is_list((term_t)term)) {
328  // append a document if term is a list
329  bson_t *nested_doc = bson_new();
330  if(bsonpl_concat(nested_doc, term, err)) {
331  BSON_APPEND_DOCUMENT(doc, key, nested_doc);
332  bson_destroy(nested_doc);
333  return true;
334  }
335  else {
336  bson_destroy(nested_doc);
337  return false;
338  }
339  }
340  else if(std::string_view(term.name()) == "bson_t") {
341  // the term contains a sub-pipeline encoded as bson_t* object which can be concatenated
342  // with the pipeline under construction
343  if(term.arity()!=1) {
344  bson_set_error(err,
345  MONGOC_ERROR_BSON,
346  MONGOC_ERROR_BSON_INVALID,
347  "invalid bson_t term -- arity should be 1: %s", (char*)term
348  );
349  return false;
350  }
351  const PlTerm &bson_value = term[1];
352  auto *sub_bson = (bson_t*)bson_value.operator void *();
353  if(sub_bson) {
354  bool success = bson_concat(doc, sub_bson);
355  bson_destroy(sub_bson);
356  return success;
357  }
358  else {
359  bson_set_error(err,
360  MONGOC_ERROR_BSON,
361  MONGOC_ERROR_BSON_INVALID,
362  "invalid bson_t term -- void pointer is null: %s", (char*)term
363  );
364  return false;
365  }
366  }
367  else {
368  // append typed value otherwise
369  return bsonpl_append_typed(doc,key,term,err);
370  }
371 }

373 bool bsonpl_concat(bson_t *doc, const PlTerm &term, bson_error_t *err)
374 {
375  // the term given to bson_from_term must be a list
376  PlTail list(term);
377  PlTerm member;
378  if(!list.next(member)) {
379  // the term is a list []
380  return true;
381  }
382  else if(PL_is_list((term_t)member)) {
383  // the term is a list [[..]|_]
384  do {
385  if(!bsonpl_concat(doc,member,err)) {
386  return false;
387  }
388  } while(list.next(member));
389  return true;
390  }
391  else {
392  // the term is a list [key,value]
393  const char *key = (char*)member;
394  list.next(member);
395  return bsonpl_append(doc,key,member,err);
396  }
