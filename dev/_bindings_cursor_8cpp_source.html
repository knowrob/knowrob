25 void BindingsCursor::setSubstitution(const std::shared_ptr<Bindings> &bindings) {
26  while (bson_iter_next(&varIter_)) {
27  auto var = std::make_shared<Variable>(bson_iter_key(&varIter_));
28 
29  if (!bson_iter_recurse(&varIter_, &valIter_)) continue;
30  if (!bson_iter_find(&valIter_, "val")) continue;
31 
32  // read the value of the variable
33  switch (bson_iter_type(&valIter_)) {
34  case BSON_TYPE_UTF8: {
35  auto utf8 = bson_iter_utf8(&valIter_, nullptr);
36  // note: currently mongo KG does not store the type of the literal,
37  // so we cannot trivially distinguish between IRI and literal and need to guess here.
38  switch (rdfNodeTypeGuess(utf8)) {
39  case RDFNodeType::BLANK:
40  bindings->set(var, std::make_shared<Blank>(utf8));
41  break;
42  case RDFNodeType::IRI:
43  bindings->set(var, std::make_shared<IRIAtom>(utf8));
44  break;
45  case RDFNodeType::LITERAL:
46  bindings->set(var, std::make_shared<StringView>(utf8));
47  break;
48  }
49  break;
50  }
51  case BSON_TYPE_INT32:
52  bindings->set(var, std::make_shared<Integer>(bson_iter_int32(&valIter_)));
53  break;
54  case BSON_TYPE_INT64:
55  bindings->set(var, std::make_shared<Long>(bson_iter_int64(&valIter_)));
56  break;
57  case BSON_TYPE_BOOL:
58  bindings->set(var, std::make_shared<Boolean>(bson_iter_bool(&valIter_)));
59  break;
60  case BSON_TYPE_DOUBLE:
61  bindings->set(var, std::make_shared<Double>(bson_iter_double(&valIter_)));
62  break;
63  default:
64  KB_WARN("unsupported type {} for predicate arguments.", bson_iter_type(&valIter_));
65  break;
66  }
67  }
68 }

70 bool BindingsCursor::nextBindings(const std::shared_ptr<Bindings> &bindings) {
71  if (!next(&resultDocument_)) return false;
72  if (!bson_iter_init(&resultIter_, resultDocument_)) return false;
73 
74  while (bson_iter_next(&resultIter_)) {
75  std::string_view resultKey(bson_iter_key(&resultIter_));
76  // read variables from "v_VARS". each sub-field is named according to the field of
77  // a variable.
78  if (resultKey == "v_VARS" && bson_iter_recurse(&resultIter_, &varIter_)) {
79  setSubstitution(bindings);
80  }
81  }
82 
83  return true;
