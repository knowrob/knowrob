25 bool Unifier::unify(const TermPtr &t0, const TermPtr &t1) //NOLINT
26 {
27  if (t1->termType() == TermType::VARIABLE) {
28  return unify(std::static_pointer_cast<Variable>(t1), t0);
29  } else {
30  switch (t0->termType()) {
31  case TermType::VARIABLE:
32  return unify(std::static_pointer_cast<Variable>(t0), t1);
33  case TermType::ATOMIC:
34  // if one of the terms is an atomic, the other must be an equal atomic
35  return *t0 == *t1;
36  case TermType::FUNCTION: {
37  // n-ary functions with n>0 only unify with other n-ary functions
38  if (t1->termType() != TermType::FUNCTION) {
39  return false;
40  }
41  auto *p0 = (Function *) t0.get();
42  auto *p1 = (Function *) t1.get();
43  // tests for functor equality, and matching arity
44  if (*p0->functor() != *p1->functor() ||
45  p0->arity() != p1->arity()) {
46  return false;
47  }
48  // unify all arguments
49  for (int i = 0; i < p0->arity(); ++i) {
50  if (!unify(p0->arguments()[i], p1->arguments()[i])) {
51  return false;
52  }
53  }
54  break;
55  }
56  }
57  }
58 
59  return true;
60 }

76 TermPtr Unifier::apply() {
77  if (!exists_) {
78  // no unifier exists
79  return Bottom::get()->functor();
80  } else if (t0_->isGround() || t1_->termType() == TermType::VARIABLE) {
81  // empty unifier, or only substitutions in t1
82  return t0_;
83  } else if (t1_->isGround() || t0_->termType() == TermType::VARIABLE) {
84  // only substitutions in t0
85  return t1_;
86  } else if (t0_->termType() == TermType::FUNCTION) {
87  // both t0_ and t1_ contain variables.
88  auto &p = (
89  t0_->variables().size() < t1_->variables().size() ?
90  t0_ : t1_);
91  return applyBindings(p, *this);
92  } else {
93  KB_WARN("something went wrong.");
94  return Bottom::get()->functor();
95  }
96 }
