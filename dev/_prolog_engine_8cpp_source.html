37 bool PrologEngine::initializeWorker() {
38  // call PL_thread_attach_engine once initially for each worker thread
39  if (PL_thread_attach_engine(nullptr) < 0) {
40  // `-1` indicates an error, and `-2` that Prolog is compiled without multithreading support

42  return false;
43  } else {
44  // if PL_thread_attach_engine()>0, then the Prolog ID for the thread was returned

67 void PrologEngine::initializeProlog() {
68  if (isPrologInitialized_) return;
69  // toggle on flag
70  isPrologInitialized_ = true;
71 
72  // PL_initialise changes the locale. This can have bad consequences.
73  // I experienced unit tests for Redland failing because of this!
74  // Seems like it is related to parsing of numbers. Setting
75  // LC_NUMERIC to "C" before PL_initialise seems to fix the problem.
76  setenv("LC_NUMERIC", "C", 1);
77 
78  static int pl_ac = 0;
79  static char *pl_av[5];
80  pl_av[pl_ac++] = getNameOfExecutable();
81  // '-g true' is used to suppress the welcome message
82  pl_av[pl_ac++] = (char *) "-g";
83  pl_av[pl_ac++] = (char *) "true";
84  // Inhibit any signal handling by Prolog
85  pl_av[pl_ac++] = (char *) "--signals=false";
86  pl_av[pl_ac] = nullptr;
87  PL_initialise(pl_ac, pl_av);
88  KB_DEBUG("Prolog has been initialized.");
89 
90  // register some foreign predicates, i.e. cpp functions that are used to evaluate predicates.
91  // note: the predicates are loaded into module "user"
92  PL_register_foreign("knowrob_register_namespace",
93  2, (pl_function_t) pl_rdf_register_namespace2, 0);
94  PL_register_foreign("url_resolve",
95  2, (pl_function_t) url_resolve2, 0);
96  PL_register_foreign("log_message", 2, (pl_function_t) prolog::log_message2, 0);
97  PL_register_foreign("log_message", 4, (pl_function_t) prolog::log_message4, 0);
98  PL_register_extensions_in_module("algebra", prolog::PL_extension_algebra);
99  PL_register_extensions_in_module("semweb", prolog::PL_extension_semweb);
100  KB_DEBUG("common foreign Prolog modules have been registered.");
101 
102  consult(std::filesystem::path("reasoner") / "prolog" / "__init__.pl", "user", false);
103  KB_DEBUG("KnowRob __init__.pl has been consulted.");
104 }

173 void PrologEngine::query(const GoalFactory &goalFactory, const BindingsHandler &callback) {
174  pushGoalAndJoin(std::make_shared<LambdaRunner>(
175  [&goalFactory, &callback](const LambdaRunner::StopChecker &hasStopRequest) {
176  auto goal = goalFactory();
177  auto qid = goal.openQuery(prologQueryFlags);
178  while (!hasStopRequest() && goal.nextSolution(qid)) {
179  auto bindings = std::make_shared<Bindings>();
180  for (auto &var: goal.vars()) {
181  if (PL_term_type(var.second) == PL_VARIABLE) {
182  continue;
183  }
184  bindings->set(std::make_shared<Variable>(var.first),
185  PrologTerm::toKnowRobTerm(var.second));
186  }
187  callback(bindings);
188  }
189  PL_close_query(qid);
190  }));
191 }
