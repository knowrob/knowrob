91 static inline std::shared_ptr<MongologReasoner> getMongologReasoner(term_t t_reasonerManager,
92  term_t t_reasonerModule) {
93  auto definedReasoner = PrologReasoner::getDefinedReasoner(t_reasonerManager, t_reasonerModule);
94  if (!definedReasoner) {
95  KB_ERROR("unable to find reasoner with id '{}' (manager id: {}).",
96  *PrologTerm::toKnowRobTerm(t_reasonerModule),
97  *PrologTerm::toKnowRobTerm(t_reasonerManager));
98  return {};
99  }
100  auto reasoner = definedReasoner->value();
101  auto mongolog = std::dynamic_pointer_cast<MongologReasoner>(reasoner);
102  if (!mongolog) {
103  KB_ERROR("reasoner with id '{}' (manager id: {}) is not a mongolog reasoner.",
104  *PrologTerm::toKnowRobTerm(t_reasonerModule),
105  *PrologTerm::toKnowRobTerm(t_reasonerManager));
106  }
107  return mongolog;
108 }

138 foreign_t pl_assert_triple_cpp9(term_t t_reasonerManager,
139  term_t t_reasonerModule,
140  term_t t_subjectTerm,
141  term_t t_propertyTerm,
142  term_t t_objectTerm,
143  term_t t_graphTerm,
144  term_t t_beginTerm,
145  term_t t_endTerm,
146  term_t t_confidenceTerm) {
147  auto mongolog = getMongologReasoner(t_reasonerManager, t_reasonerModule);
148  if (mongolog) {
149  FramedTripleView tripleData;
150 
151  // "s" field
152  auto subjectTerm = PrologTerm::toKnowRobTerm(t_subjectTerm);
153  if (subjectTerm->termType() != TermType::ATOMIC) throw QueryError("invalid subject term {}", *subjectTerm);
154  tripleData.setSubject(((Atomic *) subjectTerm.get())->stringForm());
155 
156  // "p" field
157  auto propertyTerm = PrologTerm::toKnowRobTerm(t_propertyTerm);
158  if (propertyTerm->termType() != TermType::ATOMIC) throw QueryError("invalid property term {}", *propertyTerm);
159  tripleData.setPredicate(((Atomic *) propertyTerm.get())->stringForm());
160 
161  // "o" field
162  auto objectTerm = PrologTerm::toKnowRobTerm(t_objectTerm);
163  if(objectTerm->termType() == TermType::ATOMIC) {
164  auto atomic = std::static_pointer_cast<Atomic>(objectTerm);
165  if (atomic->isNumeric()) {
166  tripleData.setXSDValue(atomic->stringForm(),
167  std::static_pointer_cast<Numeric>(atomic)->xsdType());
168  } else if (atomic->isIRI()) {
169  tripleData.setObjectIRI(atomic->stringForm());
170  } else if (atomic->isBlank()) {
171  tripleData.setObjectBlank(atomic->stringForm());
172  } else {
173  tripleData.setStringValue(atomic->stringForm());
174  }
175  } else {
176  throw QueryError("object term {} of triple has an invalid type", *objectTerm);
177  }
178 
179  // "g" field
180  TermPtr graphTerm;
181  if (!PL_is_variable(t_graphTerm)) {
182  graphTerm = PrologTerm::toKnowRobTerm(t_graphTerm);
183  if (graphTerm->termType() != TermType::ATOMIC) throw QueryError("invalid property term {}", *graphTerm);
184  tripleData.setGraph(((Atomic *) graphTerm.get())->stringForm());
185  } else {
186  tripleData.setGraph(mongolog->reasonerManager().kb()->vocabulary()->importHierarchy()->defaultGraph());
187  }
188 
189  // "c" field
190  TermPtr confidenceTerm;
191  if (!PL_is_variable(t_confidenceTerm)) {
192  confidenceTerm = PrologTerm::toKnowRobTerm(t_confidenceTerm);
193  if (confidenceTerm->isNumeric()) {
194  tripleData.setConfidence(std::static_pointer_cast<Numeric>(confidenceTerm)->asDouble());
195  } else {
196  throw QueryError("invalid confidence term {}", *confidenceTerm);
197  }
198  }
199 
200  // "b" field
201  TermPtr beginTerm;
202  if (!PL_is_variable(t_beginTerm)) {
203  beginTerm = PrologTerm::toKnowRobTerm(t_beginTerm);
204  if (beginTerm->isNumeric()) {
205  tripleData.setBegin(std::static_pointer_cast<Numeric>(beginTerm)->asDouble());
206  } else {
207  throw QueryError("invalid begin term {}", *beginTerm);
208  }
209  }
210 
211  // "e" field
212  TermPtr endTerm;
213  if (!PL_is_variable(t_endTerm)) {
214  endTerm = PrologTerm::toKnowRobTerm(t_endTerm);
215  if (endTerm->isNumeric()) {
216  tripleData.setEnd(std::static_pointer_cast<Numeric>(endTerm)->asDouble());
217  } else {
218  throw QueryError("invalid end term {}", *endTerm);
219  }
220  }
221 
222  mongolog->reasonerManager().kb()->edb()->mergeInsert(mongolog->knowledgeGraph(), tripleData);
223  //mongolog->kb()->insertOne(tripleData);
224  return true;
225  } else {
226  KB_WARN("[mongolog] unable to assert triple: reasoner not found");
227  return false;
228  }
