131 void QueryTree::expandNextNode()
132 {
133  auto next = openNodes_.top();
134  next->isOpen = false;
135  openNodes_.pop();
136 
137  switch(next->formula->type()) {
138  case FormulaType::PREDICATE:
139  case FormulaType::MODAL:
140  for(Node *leaf : getLeafs(next)) {
141  if(hasCompletePath(leaf)) {
142  paths_.emplace_back();
143  auto &path = paths_.back();
144  constructPath(leaf, path);
145  }
146  }
147  break;
148 
149  case FormulaType::CONJUNCTION: {
150  auto *formula = (Conjunction*)next->formula.get();
151  if(next->isNegated) {
152  for(Node *leaf : getLeafs(next)) {
153  for(auto &phi : formula->formulae()) {
154  createNode(leaf, phi, true);
155  }
156  }
157  }
158  else {
159  for(Node *leaf : getLeafs(next)) {
160  Node *parent = leaf;
161  for (auto &phi: formula->formulae()) {
162  parent = createNode(parent, phi, false);
163  }
164  }
165  }
166  break;
167  }
168 
169  case FormulaType::DISJUNCTION: {
170  auto *formula = (Disjunction*)next->formula.get();
171  if(next->isNegated) {
172  for(Node *leaf : getLeafs(next)) {
173  Node *parent = leaf;
174  for (auto &phi: formula->formulae()) {
175  parent = createNode(parent, phi, true);
176  }
177  }
178  }
179  else {
180  for(Node *leaf : getLeafs(next)) {
181  for (auto &phi: formula->formulae()) {
182  createNode(leaf, phi, false);
183  }
184  }
185  }
186  break;
187  }
188 
189  case FormulaType::IMPLICATION: {
190  auto *formula = (Implication*)next->formula.get();
191  if(next->isNegated) {
192  for(Node *leaf : getLeafs(next)) {
193  Node *parent = leaf;
194  parent = createNode(parent, formula->antecedent(), false);
195  createNode(parent, formula->consequent(), true);
196  }
197  }
198  else {
199  for(Node *leaf : getLeafs(next)) {
200  createNode(leaf, formula->antecedent(), true);
201  createNode(leaf, formula->consequent(), false);
202  }
203  }
204  break;
205  }
206 
207  case FormulaType::NEGATION: {
208  auto *formula = (Negation*)next->formula.get();
209  for(Node *leaf : getLeafs(next)) {
210  createNode(leaf, formula->negatedFormula(), !next->isNegated);
211  }
212  break;
213  }
214  }
