45 size_t Predicate::hash() const {
46  static const auto GOLDEN_RATIO_HASH = static_cast<size_t>(0x9e3779b9);
47  auto seed = static_cast<size_t>(0);
48 
49  // Combine the hashes.
50  // The function (a ^ (b + GOLDEN_RATIO_HASH + (a << 6) + (a >> 2))) is known to
51  // give a good distribution of hash values across the range of size_t.
52  //
53  seed ^= std::hash<std::string_view>{}(functor_->stringForm()) + GOLDEN_RATIO_HASH + (seed << 6) + (seed >> 2);
54  for (auto &arg: arguments_) {
55  seed ^= arg->hash() + GOLDEN_RATIO_HASH + (seed << 6) + (seed >> 2);
56  }
57 
58  return seed;
59 }
