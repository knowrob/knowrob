170 bool RedlandModel::initializeBackend() {
171  if (isInitialized()) {
172  KB_WARN("RedlandModel already initialized.");
173  return false;
174  }
175 
176  // initialize the world if it hasn't been set before
177  if (!world_) {
178  ownedWorld_ = world_ = librdf_new_world();
179  if (!world_) {
180  throw BackendError("Failed to create Redland world.");
181  }
182  librdf_world_open(world_);
183  librdf_world_set_logger(world_, nullptr, logRedlandMessage);
184  }
185 
186  // initialize the storage
187  storageOptions_ = getStorageOptions();
188  const char *storageOptions = storageOptions_.empty() ? nullptr : storageOptions_.c_str();
189  const char *storageType = getStorageTypeString(storageType_);
190  const char *storageName = origin_.has_value() ? origin_.value().c_str() : "knowrob";
191  KB_INFO("[redland] using storage of type \"{}\" with name \"{}\" and options \"{}\".",
192  storageType, storageName, storageOptions);
193  storage_ = librdf_new_storage(world_,
194  storageType,
195  storageName,
196  storageOptions);
197  if (!storage_) {
198  throw BackendError("Failed to create Redland storage for type \"{}\".", storageType);
199  }
200 
201  // initialize the model
202  static const char *modelOptions = nullptr;
203  model_ = librdf_new_model(world_, storage_, modelOptions);
204  if (!model_) {
205  librdf_free_storage(storage_);
206  storage_ = nullptr;
207  throw BackendError("Failed to create Redland model for storage type \"{}\".", storageType);
208  }
209 
210  // initialize URIs
211  for (int i = 0; i != static_cast<int>(XSDType::LAST); i++) {
212  xsdURIs_[i].set(world_, xsdTypeToIRI((XSDType) i));
213  }
214 
215  return true;
216 }

237  if (o_storage) {

242  storageType_ = RedlandStorageType::MYSQL;

255 std::string RedlandModel::getStorageOptions() const {
256  using OptPair_o = std::pair<std::string, std::optional<std::string>>;
257  std::vector<std::pair<std::string, std::string>> opts;
258  for (auto &option: {
259  OptPair_o{"host", host_},
260  OptPair_o{"port", port_},
261  OptPair_o{"database", database_},
262  OptPair_o{"user", user_},
263  OptPair_o{"password", password_},
264  OptPair_o{"dir", storageDir_}}) {
265  if (option.second.has_value()) {
266  opts.emplace_back(option.first, option.second.value());
267  }
268  }
269  // "hash-type" is required by "hashes" storage type
270  if(hashType_.has_value()) {
271  switch (hashType_.value()) {
272  case RedlandHashType::MEMORY:
273  opts.emplace_back("hash-type", "memory");
274  break;
275  case RedlandHashType::BDB:
276  opts.emplace_back("hash-type", "bdb");
277  break;
278  }
279  }
280  // context support currently is required which limits the storage types we can use.
281  // but "memory" and "hashes" can do it.
282  opts.emplace_back("contexts", "yes");
283  std::stringstream ss;
284  for (int i = 0; i < opts.size(); i++) {
285  ss << opts[i].first << "='" << opts[i].second << '\'';
286  if (i < opts.size() - 1) {
287  ss << ",";
288  }
289  }
290  return ss.str();
291 }

308 static inline TermPtr termFromNode(librdf_node *node) {
309  if (!node) return nullptr;
310 
311  switch (librdf_node_get_type(node)) {
312  case LIBRDF_NODE_TYPE_RESOURCE: {
313  auto uri = librdf_node_get_uri(node);
314  if (uri) {
315  return std::make_shared<IRIAtom>((const char *) librdf_uri_as_string(uri));
316  }
317  break;
318  }
319  case LIBRDF_NODE_TYPE_BLANK:
320  return std::make_shared<Blank>(
321  (const char *) librdf_node_get_blank_identifier(node));
322  case LIBRDF_NODE_TYPE_LITERAL: {
323  auto literal_value = (const char *) librdf_node_get_literal_value(node);
324  auto datatype_uri = librdf_node_get_literal_value_datatype_uri(node);
325  TermPtr knowrobTerm;
326  if (datatype_uri) {
327  auto u_uri_str = librdf_uri_to_string(datatype_uri);
328  std::string_view uri_str((const char *) u_uri_str);
329  if (xsd::isDoubleType(uri_str)) {
330  knowrobTerm = std::make_shared<Double>(std::stod(literal_value));
331  } else if (xsd::isIntegerType(uri_str)) {
332  knowrobTerm = std::make_shared<Integer>(std::stoll(literal_value));
333  } else if (xsd::isBooleanType(uri_str)) {
334  knowrobTerm = std::make_shared<Integer>(std::string_view(literal_value) == "true");
335  }
336  free(u_uri_str);
337  }
338  if (!knowrobTerm) {
339  knowrobTerm = std::make_shared<StringView>(literal_value);
340  }
341  return knowrobTerm;
342  }
343  default:
344  break;
345  }
346  return nullptr;
347 }

416 void RedlandModel::batch(const TripleHandler &callback) const {
417  auto triples = std::make_shared<RaptorContainer>(GlobalSettings::batchSize());
418  auto contexts = librdf_model_get_contexts(model_);
419 
420  while (!librdf_iterator_end(contexts)) {
421  auto context = (librdf_node *) librdf_iterator_get_object(contexts);
422  auto stream = librdf_model_find_statements_in_context(
423  model_, nullptr, context);
424 
425  while (!librdf_stream_end(stream)) {
426  auto statement = librdf_stream_get_object(stream);
427  triples->add(statement->subject, statement->predicate, statement->object, context);
428  if (triples->size() >= GlobalSettings::batchSize()) {
429  triples->shrink();
430  callback(triples);
431  triples->reset();
432  }
433  librdf_stream_next(stream);
434  }
435  librdf_free_stream(stream);
436  librdf_iterator_next(contexts);
437  }
438  // Clean up
439  if (triples->size() > 0) {
440  triples->shrink();
441  callback(triples);
442  }
443  librdf_free_iterator(contexts);
444 }

454 void RedlandModel::match(const FramedTriplePattern &query, const TripleVisitor &visitor) {
455  auto triples = std::make_shared<RaptorContainer>(1);
456  auto rdf_query = librdf_new_statement(world_);
457  knowrobToRaptor(query, rdf_query);
458 
459  auto stream = librdf_model_find_statements(model_, rdf_query);
460  while (!librdf_stream_end(stream)) {
461  auto rdf_statement = librdf_stream_get_object(stream);
462  triples->add(rdf_statement->subject, rdf_statement->predicate, rdf_statement->object);
463  auto &triple = *triples->begin();
464 
465  if (query.objectOperator() != FilterType::EQ) {
466  if (query.filter(*triple)) visitor(triple);
467  } else {
468  visitor(triple);
469  }
470  triples->reset();
471  librdf_stream_next(stream);
472  }
473  librdf_free_stream(stream);
474 }

476 bool RedlandModel::sparql(std::string_view queryString, const BindingsHandler &callback) const {
477  auto queryObj = librdf_new_query(
478  world_,
479  QUERY_LANGUAGE_SPARQL.data(),
480  nullptr,
481  (unsigned char *) queryString.data(),
482  nullptr);
483  if (!queryObj) {
484  KB_WARN("Failed to create query `{}` for model \"{}\".", queryString, storageName_);
485  return false;
486  }
487  auto results = librdf_query_execute(queryObj, model_);
488  if (!results) {
489  KB_WARN("Failed to execute query `{}` for model \"{}\".", queryString, storageName_);
490  librdf_free_query(queryObj);
491  return false;
492  }
493  while (!librdf_query_results_finished(results)) {
494  auto bindings = std::make_shared<Bindings>();
495 
496  // read bindings
497  int bindings_count = librdf_query_results_get_bindings_count(results);
498  for (int i = 0; i < bindings_count; ++i) {
499  auto name = std::string_view(librdf_query_results_get_binding_name(results, i));
500  if (name.empty() || name[0] == '_') {
501  // skip variables that start with '_'.
502  // afaik, there is no way to prevent these from being returned when "*" is
503  // used in select pattern.
504  continue;
505  }
506 
507  auto node = librdf_query_results_get_binding_value(results, i);
508  if (!node) {
509  // this indicates that a variable appears in select pattern, but it was not grounded
510  // in the query. this is not an error, but rather means the grounding was OPTIONAL.
511  // we can just skip this variable.
512  continue;
513  }
514 
515  auto knowrobTerm = termFromNode(node);
516  if (knowrobTerm) {
517  bindings->set(std::make_shared<Variable>(name), knowrobTerm);
518  } else {
519  KB_WARN("Failed to process binding for variable \"{}\".", name);
520  }
521  librdf_free_node(node);
522  }
523  callback(bindings);
524  librdf_query_results_next(results);
525  }
526  librdf_free_query_results(results);
527  librdf_free_query(queryObj);
528  return true;
529 }

554 raptor_term *RedlandModel::knowrobToRaptor(const TermPtr &term) {
555  if (!term || !term->isAtomic()) {
556  return nullptr;
557  }
558  auto atomic = std::static_pointer_cast<Atomic>(term);
559  if (term->isIRI()) {
560  return KNOWROB_RDF_NEW_URI(atomic->stringForm());
561  } else if (term->isBlank()) {
562  return KNOWROB_RDF_NEW_BLANK(atomic->stringForm());
563  } else if (term->isString()) {
564  return KNOWROB_RDF_NEW_LITERAL(atomic->stringForm().data(), xsdURI(XSDType::STRING));
565  } else if (term->isNumeric()) {
566  auto numeric = std::static_pointer_cast<Numeric>(atomic);
567  return KNOWROB_RDF_NEW_LITERAL(atomic->stringForm().data(), xsdURI(numeric->xsdType()));
568  }
569  KB_WARN("Failed to convert term {} to raptor term.", *term);
570  return nullptr;
571 }

589 void RedlandModel::knowrobToRaptor(const FramedTriple &triple, raptor_statement *raptorTriple) {
590  raptor_term *subject, *predicate, *object;
591  predicate = KNOWROB_RDF_NEW_URI(triple.predicate());
592  if (triple.isSubjectBlank()) {
593  subject = KNOWROB_RDF_NEW_BLANK(triple.subject());
594  } else {
595  subject = KNOWROB_RDF_NEW_URI(triple.subject());
596  }
597  if (triple.isObjectBlank()) {
598  object = KNOWROB_RDF_NEW_BLANK(triple.valueAsString());
599  } else if (triple.isObjectIRI()) {
600  object = KNOWROB_RDF_NEW_URI(triple.valueAsString());
601  } else if (triple.xsdType()) {
602  switch (triple.xsdType().value()) {
603  case XSDType::STRING:
