86 void PrologBackend::batch(const TripleHandler &callback) const {
87  // :- rdf(?subject, ?property, ?object, ?origin).
88  // Note: Prolog querying works via backtracking over term_t, and this
89  // changes the value of the term_t in the Prolog engine such that imo we cannot
90  // avoid copying the results into a container -- as opposed to mapping the memory
91  // which was allocated by the Prolog engine during querying.
92  // A copy of the term_t could be created though, but this seems pointless, so here
93  // we just create new KnowRob terms with their own memory allocation.
94  static auto var_s = std::make_shared<Variable>("s");
95  static auto var_p = std::make_shared<Variable>("p");
96  static auto var_o = std::make_shared<Variable>("o");
97  static auto var_g = std::make_shared<Variable>("g");
98  auto triples = std::make_shared<TripleViewBatch>(GlobalSettings::batchSize());
99 
100  PROLOG_ENGINE_QUERY(
101  PrologTerm(rdf_quad,
102  PrologTerm(var_s),
103  PrologTerm(var_p),
104  PrologTerm(var_o),
105  PrologTerm(var_g)),
106  [&](const BindingsPtr &bindings) {
107  auto val_s = bindings->getAtomic(var_s->name());
108  auto val_p = bindings->getAtomic(var_p->name());
109  auto val_o = bindings->getAtomic(var_o->name());
110  auto val_g = bindings->getAtomic(var_g->name());
111  if(!val_s || !val_p || !val_o) {
112  KB_WARN("Failed to retrieve triple from Prolog: missing grounding.");
113  return;
114  }
115 
116  FramedTriplePtr triple_ptr;
117  triple_ptr.ptr = new FramedTripleCopy(val_s->stringForm(), val_p->stringForm());
118  if (val_o->isNumeric() || val_o->isString()) {
119  auto xsd_o = std::static_pointer_cast<XSDAtomic>(val_o);
120  triple_ptr.ptr->setXSDValue(xsd_o->stringForm(), xsd_o->xsdType());
121  } else if (val_o->isIRI()) {
122  triple_ptr.ptr->setObjectIRI(val_o->stringForm());
123  } else if (val_o->isBlank()) {
124  triple_ptr.ptr->setObjectBlank(val_o->stringForm());
125  }
126  if (val_g) {
127  triple_ptr.ptr->setGraph(val_g->stringForm());
128  }
129  triple_ptr.owned = true;
130 
131  triples->add(triple_ptr);
132  if (triples->size() >= GlobalSettings::batchSize()) {
133  callback(triples);
134  triples->reset();
135  }
136  });
137 
138  // Clean up
139  if (triples->size() > 0) {
140  callback(triples);
141  }
142 }
