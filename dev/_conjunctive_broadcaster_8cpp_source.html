22 void ConjunctiveBroadcaster::push(Channel &channel, const TokenPtr &tok) {
23  if (tok->type() == TokenType::ANSWER_TOKEN) {
24  auto answer = std::static_pointer_cast<const Answer>(tok);
25  if (answer->isPositive()) {
26  const uint32_t channelID = channel.id();
27  // need to lock the whole push as genCombinations uses an iterator over the buffer.
28  std::lock_guard<std::mutex> lock(buffer_mutex_);
29 
30  // add to the buffer for later combinations
31  // replace other answer with same hash if present
32  buffer_[channelID][answer->hash()] = answer;
33 
34  // generate combinations with other channels if each channel
35  // buffer has some content.
36  if (buffer_.size() == channels_.size()) {
37  if (channels_.size() == 1) {
38  // not needed to generate combinations
39  TokenBroadcaster::push(tok);
40  } else {
41  // generate all combinations and push combined messages
42  genCombinations(channelID, buffer_.begin(), answer);
43  }
44  }
45  } else if (answer->isNegative()) {
46  // do not combine negative answers like the positive ones above.
47  // only push "no" when receiving EOF while no positive answer has been produced.
48  negativeAnswers_.emplace_back(std::static_pointer_cast<const AnswerNo>(answer));
49  }
50  } else {
51  if (tok->indicatesEndOfEvaluation() && !hasSolution_) {
52  if(negativeAnswers_.size()==1) {
53  TokenBroadcaster::push(negativeAnswers_.front());
54  }
55  else {
56  auto no = std::make_shared<AnswerNo>();
57  for (auto &x: negativeAnswers_) {
58  no->mergeWith(*x);
59  }
60  TokenBroadcaster::push(no);
61  }
62  }
63  // pass through non-answer messages
64  TokenStream::push(channel, tok);
65  }
66 }

68 void ConjunctiveBroadcaster::genCombinations( //NOLINT
69  uint32_t pushedChannelID, AnswerMap::iterator it, AnswerPtr &combinedResult) {
70  if (it == buffer_.end()) {
71  // end reached, push combination
72  TokenBroadcaster::push(combinedResult);
73  hasSolution_ = true;
74  } else if (it->first == pushedChannelID) {
75  // pass through channel from which the new message was pushed
76  auto it1 = it;
77  ++it1;
78  genCombinations(pushedChannelID, it1, combinedResult);
79  } else if (it->second.size() == 1) {
80  // only a single message buffered from this channel
81  auto it1 = it;
82  ++it1;
83  auto merged = mergeAnswers(combinedResult,
84  it->second.begin()->second, ignoreInconsistentAnswers_);
85  if (merged) {
86  genCombinations(pushedChannelID, it1, merged);
87  }
88  } else {
89  // generate a combination for each buffered message
90  // note: the number of possible combinations grows exponentially with number of messages in channels
91  auto it1 = it;
92  ++it1;
93  for (auto &msg: it->second) {
94  auto merged = mergeAnswers(
95  combinedResult, msg.second, ignoreInconsistentAnswers_);
96  if (merged) {
97  genCombinations(pushedChannelID, it1, merged);
98  }
99  }
100  }

172  }));
173  auto answer21 = std::make_shared<AnswerYes>(std::make_shared<Bindings>(Bindings::VarMap{
174  {std::make_shared<Variable>("a"), QueryParser::parseFunction("p(2,Y)")}

177  {std::make_shared<Variable>("a"), QueryParser::parseFunction("p(2,2)")}

180  input1->push(answer11);

182  // expect that the combiner has one output "a=p(2,1)" unifying both inputs.
183  EXPECT_EQ(output->size(), 1);

189  EXPECT_TRUE(answer->isPositive());
190  EXPECT_FALSE(answer->isNegative());
