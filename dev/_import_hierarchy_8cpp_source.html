68 void ImportHierarchy::addDirectImport(std::string_view importerGraphName, std::string_view importedGraphName) {
69  auto &g_importer = getCurrentGraph(importerGraphName);
70  auto &g_imported = getCurrentGraph(importedGraphName);
71 
72  // if a system graph imports a session graph, remove the session graph from the session
73  // before adding to the system. And if a session graph imports a system graph, ignore the import.
74  if (isSystemOrigin(g_importer) && isSessionOrigin(g_imported)) {
75  removeCurrentGraph(importedGraphName);
76  } else if (isSessionOrigin(g_importer) && isSystemOrigin(g_imported)) {
77  KB_WARN("Ignoring session graph \"{}\" import of system graph \"{}\".", importerGraphName, importedGraphName);
78  return;
79  }
80 
81  KB_DEBUG("Graph \"{}\" imports \"{}\".", importerGraphName, importedGraphName);
82  // g_importer.directlyImports += [g_imported]
83  g_importer.directImports_.insert(&g_imported);
84  // g_importer.imports += g_imported.imports + [gb]
85  g_importer.imports_.insert(g_imported.imports_.begin(), g_imported.imports_.end());
86  g_importer.imports_.insert(&g_imported);
87  // for every graph gx that imports g_importer:
88  // gx.imports += (g_imported.imports + [g_imported])
89  for (auto &it: graphs_) {
90  if (it.second->imports_.count(&g_importer) > 0) {
91  it.second->imports_.insert(g_imported.imports_.begin(), g_imported.imports_.end());
92  it.second->imports_.insert(&g_imported);
93  }
94  }
95 }

97 void ImportHierarchy::removeCurrentGraph(std::string_view graphName) {
98  auto it = graphs_.find(graphName);
99  if (it == graphs_.end()) return;
100  auto &ga = *it->second;
101 
102  // for every graph gx that directly imports ga
103  for (auto &jt: graphs_) {
104  auto &gx = *jt.second;
105  if (gx.directImports_.count(&ga) > 0) {
106  // gx.directlyImports -= [ga]
107  gx.directImports_.erase(&ga);
108  gx.imports_.erase(&ga);
109  // gx.directlyImports += [ga.directImports]
110  gx.directImports_.insert(ga.directImports_.begin(), ga.directImports_.end());
111  }
112  }
113  // for every graph gy that imports ga:
114  for (auto &kt: graphs_) {
115  auto &gy = *kt.second;
116  if (gy.imports_.count(&ga) > 0) {
117  // gy.imports -= [ga]
118  gy.imports_.erase(&ga);
119  }
120  }
121 
122  // finally delete the node
123  graphs_.erase(it);
124 }
