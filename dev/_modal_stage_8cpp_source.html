14 ModalStage::ModalStage(
15  KnowledgeBase *kb,
16  const std::shared_ptr<ModalFormula> &modal,
17  const QueryContextPtr &ctx)
18  : TypedQueryStage(ctx, modal->modalFormula(),
19  [this](auto &&PH1) { return submitQuery(std::forward<decltype(PH1)>(PH1)); }),
20  kb_(kb),
21  modalFormula_(modal) {
22  // this modal stage could be embedded into another modal expression.
23  // however, forming a mental model of what a human might think about what the robot thinks about him is an
24  // additional level which we do not consider (yet).
25  // However, expressions like `P(Bp & Bq)` with iterations over different modality types can be used in queries.
26 
27  auto modalContext = std::make_shared<QueryContext>(
28  *ctx, modalFormula_->modalOperator());
29  nestedContext_ = modalContext;
30 
31  auto &modalOperator = modalFormula_->modalOperator();
32 
33  // modify the query context for the evaluation of the modal.
34  switch (modalOperator->modalType()) {
35  case ModalType::KNOWLEDGE:
36  case ModalType::BELIEF:
37  // epistemic states of different agents are stored in a single KG where triples are labeled by the agent.
38  // These states are assumed to be generated from the perspective of a "self", i.e. the agent
39  // that runs the knowledge base and its information about the other agents.
40  if (ctx->selector.perspective.has_value() &&
41  ctx->selector.perspective.value() != Perspective::getEgoPerspective()) {
42  // For now higher-level epistemic states are not allowed in queries,
43  // e.g. `B_a1(B_a2(x))` where `a1` is not the agent that runs the knowledge base is an example
44  // of such a higher-order query which is not allowed.
45  // `B_self(B_a2(x))` is ok though, and changes the agent label of the KG selector to `a2` below.
46  throw QueryError("epistemic formula {} is embedded within epistemic context of another agent.",
47  *modalFormula_);
48  }
49  modalContext->selector.perspective = modalOperator->perspective();
50  modalContext->selector.uncertain = !modalOperator->isModalNecessity();
51  if (modalContext->selector.uncertain) {
52  modalContext->selector.confidence = modalOperator->confidence();
53  }
54  break;
55 
56  case ModalType::ALWAYS:
57  case ModalType::SOMETIMES:
58  modalContext->selector.occasional = !modalOperator->isModalNecessity();
59  modalContext->selector.begin = modalOperator->begin();
60  modalContext->selector.end = modalOperator->end();
61  break;
62  }
63 }

65 TokenBufferPtr ModalStage::submitQuery_K(const FormulaPtr &phi) {
66  // This case corresponds to "I know that `phi` is true".
67  // It is assumed that all the knowledge of an agent is stored in an EDB or can be derived
68  // from EDB-facts through top-down methods.
69  // Hence, `phi` can be evaluated with the constraint that all positive literals can be grounded
70  // in or derived from EDB-facts that are marked as "certain".
71  // NOTE: At a later point it might be interesting to connect this evaluation to a simulation model
72  // where the world is simulated from the perspective of another agent.
73  return kb_->submitQuery(phi, nestedContext_);
74 }

76 TokenBufferPtr ModalStage::submitQuery_B(const FormulaPtr &phi) {
77  // This case corresponds to "I believe that `phi` is true".
78  // This is interpreted as a "best guess", and it is assumed that there is only one
79  // best guess at a time which is materialized in the EDB.
80  // Hence, `phi` can be evaluated with the constraint that all positive literals can be grounded
81  // in or derived from EDB-facts that are marked as "certain" or "uncertain".
82  return kb_->submitQuery(phi, nestedContext_);
83 }

85 TokenBufferPtr ModalStage::submitQuery_P(const FormulaPtr &phi) {
86  // This case corresponds to "At some point in the past `phi` was true".
87  // It is assumed here that the past is materialized in the EDB already.
88  // One use case being the NEEMs of the EASE project that represent episodes
89  // of robot actions that happened in the past.
90  // Hence, `phi` can be evaluated with the constraint that all positive literals can be grounded
91  // in or derived from EDB-facts whose temporal intersection is non-empty:
92  // NOTE: At a later point it might be interesting to consider this case and below as
93  // access to long term memory where the KG is not materialized locally.
94  // e.g. using the NEEMHub infrastructure of EASE directly for query evaluation.
95  return kb_->submitQuery(phi, nestedContext_);
96 }

106 TokenBufferPtr ModalStage::submitQuery(const FormulaPtr &phi) {
107  switch (modalFormula_->modalOperator()->modalType()) {
108  case ModalType::KNOWLEDGE:
109  return submitQuery_K(phi);
110  case ModalType::BELIEF:
111  return submitQuery_B(phi);
112  case ModalType::ALWAYS:
113  return submitQuery_H(phi);
114  case ModalType::SOMETIMES:
115  return submitQuery_P(phi);
116  }
117  throw QueryError("unexpected modality type in query '{}'.", *modalFormula_);
118 }
