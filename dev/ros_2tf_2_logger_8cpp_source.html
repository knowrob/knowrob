69 bool TFLogger::ignoreTransform(const geometry_msgs::TransformStamped &ts0)
70 {
71  const std::string &child = ts0.child_frame_id;
72  if(!memory_.has_transform(child)) {
73  // it's a new frame
74  return false;
75  }
76  if(memory_.is_managed_frame(child)) {
77  // managed frames are asserted into DB back-end directly
78  return true;
79  }
80  const geometry_msgs::TransformStamped &ts1 = memory_.get_transform(child);
81  // do not ignore in case parent frame has changed
82  const std::string &parent0 = ts0.header.frame_id;
83  const std::string &parent1 = ts1.header.frame_id;
84  if(parent0.compare(parent1)!=0) {
85  return false;
86  }
87  // tests whether temporal distance exceeds threshold
88  if(timeThreshold_>0.0) {
89  double timeDistance = (
90  (ts0.header.stamp.sec * 1000.0 + ts0.header.stamp.nsec / 1000000.0) -
91  (ts1.header.stamp.sec * 1000.0 + ts1.header.stamp.nsec / 1000000.0)) / 1000.0;
92  if(timeDistance<0 || timeDistance > timeThreshold_) {
93  return false;
94  }
95  }
96  // tests whether vectorial distance exceeds threshold
97  const geometry_msgs::Vector3 &pos0 = ts0.transform.translation;
98  const geometry_msgs::Vector3 &pos1 = ts1.transform.translation;
99  double vectorialDistance = sqrt(
100  ((pos0.x - pos1.x) * (pos0.x - pos1.x)) +
101  ((pos0.y - pos1.y) * (pos0.y - pos1.y)) +
102  ((pos0.z - pos1.z) * (pos0.z - pos1.z)));
103  if(vectorialDistance > vectorialThreshold_) {
104  return false;
105  }
106  // tests whether angular distance exceeds threshold
107  const geometry_msgs::Quaternion &rot0 = ts0.transform.rotation;
108  const geometry_msgs::Quaternion &rot1 = ts1.transform.rotation;
109  tf::Quaternion q1(rot0.x, rot0.y, rot0.z, rot0.w);
110  tf::Quaternion q2(rot1.x, rot1.y, rot1.z, rot1.w);
111  float angularDistance = 2.0 * fabs(q1.angle(q2));
112  if(angularDistance > angularThreshold_) {
113  return false;
114  }
115  return true;
116 }
