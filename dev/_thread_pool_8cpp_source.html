50 void ThreadPool::pushWork(const std::shared_ptr<ThreadPool::Runner> &goal, ExceptionHandler exceptionHandler) {
51  {
52  std::lock_guard<std::mutex> scoped_lock(workMutex_);
53  goal->setExceptionHandler(std::move(exceptionHandler));
54  workQueue_.push(goal);
55 
56  uint32_t numAliveThreads = workerThreads_.size() - numFinishedThreads_;
57  uint32_t numAvailableThreads = numAliveThreads - numActiveWorker_;
58  if (numAvailableThreads == 0) {

61  workerThreads_.push_back(new Worker(this));
62  }
63  }
64  }

66  workCV_.notify_one();

93 void ThreadPool::Worker::run() {
94  KB_DEBUG("Worker started.");
95  // let the pool do some thread specific initialization
96  if (!threadPool_->initializeWorker()) {
97  KB_ERROR("Worker initialization failed.");
98  isTerminated_ = true;
99  threadPool_->numFinishedThreads_ += 1;
100  return;
101  }
102  KB_DEBUG("Worker initialized.");
103  hasTerminateRequest_ = false;
104  threadPool_->numActiveWorker_ += 1;
105 
106  // loop until the application exits
107  while (!hasTerminateRequest_) {
108  // wait for a claim
109  {
110  KB_DEBUG("Worker going to sleep.");
111  std::unique_lock<std::mutex> lk(threadPool_->workMutex_);
112  threadPool_->numActiveWorker_ -= 1;
113  threadPool_->workCV_.wait(lk, [this] {
114  return hasTerminateRequest_ || !threadPool_->workQueue_.empty();
115  });
116  threadPool_->numActiveWorker_ += 1;
117  KB_DEBUG("Worker woke up.");
118  }
119  if (hasTerminateRequest_) {
120  KB_DEBUG("Worker has terminate request.");
121  break;
122  }
123 
124  // pop work from queue
125  auto goal = threadPool_->popWork();
126  // do the work
127  if (goal) {
128  KB_DEBUG("Worker has a new goal.");
129  goal->runInternal();
130  KB_DEBUG("Work finished.");
131  }
132  }
133 
134  isTerminated_ = true;
135  // tell the thread pool that a worker thread exited
136  if (threadPool_->finalizeWorker_) {
137  threadPool_->finalizeWorker_();
138  }
139  // note: counter indicates that there are finished threads in workerThreads_ list.
140  threadPool_->numFinishedThreads_ += 1;
141  threadPool_->numActiveWorker_ -= 1;
142  KB_DEBUG("Worker terminated.");
143 }
