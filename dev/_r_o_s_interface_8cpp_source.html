51 ROSInterface::applyModality(const GraphQueryMessage &query,
52  FormulaPtr phi) {
53  FormulaPtr mFormula = std::move(phi);
54 
55  // Add epistemic operator
56  if (query.epistemicOperator == GraphQueryMessage::BELIEF) {
57  if (!query.aboutAgentIRI.empty()) {
58  if (query.confidence != 1.0){
59  mFormula = std::make_shared<ModalFormula>(
60  BeliefModality::B(query.aboutAgentIRI, query.confidence), mFormula);
61  } else {
62  mFormula = std::make_shared<ModalFormula>(
63  BeliefModality::B(query.aboutAgentIRI), mFormula);
64  }
65  }
66  } else if (query.epistemicOperator == GraphQueryMessage::KNOWLEDGE) {
67  if (!query.aboutAgentIRI.empty()) {
68  mFormula = std::make_shared<ModalFormula>(
69  KnowledgeModality::K(query.aboutAgentIRI),mFormula);
70  }
71  }
72  // Add temporal operator
73  if (query.temporalOperator == GraphQueryMessage::SOME_PAST) {
74  if (query.minPastTimestamp != GraphQueryMessage::UNSPECIFIED_TIMESTAMP
75  || query.maxPastTimestamp != GraphQueryMessage::UNSPECIFIED_TIMESTAMP) {
76  if (query.minPastTimestamp == GraphQueryMessage::UNSPECIFIED_TIMESTAMP) {
77  mFormula = std::make_shared<ModalFormula>(
78  PastModality::P(TimeInterval(std::nullopt,
79  query.maxPastTimestamp)),mFormula);
80  } else if (query.maxPastTimestamp == GraphQueryMessage::UNSPECIFIED_TIMESTAMP) {
81  mFormula = std::make_shared<ModalFormula>(
82  PastModality::P(TimeInterval(query.minPastTimestamp,
83  std::nullopt)),mFormula);
84  } else {
85  mFormula = std::make_shared<ModalFormula>(
86  PastModality::P(TimeInterval(query.minPastTimestamp,
87  query.maxPastTimestamp)),mFormula);
88  }
89  } else {
90  mFormula = std::make_shared<ModalFormula>(
91  PastModality::P(),mFormula);
92  }
93  } else if (query.temporalOperator == GraphQueryMessage::ALL_PAST) {
94  if (query.minPastTimestamp != GraphQueryMessage::UNSPECIFIED_TIMESTAMP
95  || query.maxPastTimestamp != GraphQueryMessage::UNSPECIFIED_TIMESTAMP) {
96  if (query.minPastTimestamp == GraphQueryMessage::UNSPECIFIED_TIMESTAMP) {
97  mFormula = std::make_shared<ModalFormula>(
98  PastModality::H(TimeInterval(std::nullopt,
99  query.maxPastTimestamp)),mFormula);
100  } else if (query.maxPastTimestamp == GraphQueryMessage::UNSPECIFIED_TIMESTAMP) {
101  mFormula = std::make_shared<ModalFormula>(
102  PastModality::H(TimeInterval(query.minPastTimestamp,
103  std::nullopt)),mFormula);
104  } else {
105  mFormula = std::make_shared<ModalFormula>(
106  PastModality::H(TimeInterval(query.minPastTimestamp,
107  query.maxPastTimestamp)),mFormula);
108  }
109  } else {
110  mFormula = std::make_shared<ModalFormula>(
111  PastModality::H(),mFormula);
112  }
113  }
114  return mFormula;
115 }

117 GraphAnswerMessage ROSInterface::createGraphAnswer(std::shared_ptr<const Answer> answer) {
118  const SubstitutionPtr &substitution = answer->substitution();
119  GraphAnswerMessage graphAnswer;
120  for (const auto& pair : *substitution) {
121  KeyValuePair kvpair;
122  kvpair.key = pair.first.name();
123  TermPtr term = pair.second;
124  // Stringstream for list terms
125  std::stringstream ss;
126 
127  switch(term->type()) {
128  case TermType::STRING:
129  kvpair.type = KeyValuePair::TYPE_STRING;
130  kvpair.value_string = ((StringTerm*)term.get())->value();
131  break;
132  case TermType::DOUBLE:
133  kvpair.type = KeyValuePair::TYPE_FLOAT;
134  kvpair.value_float = ((DoubleTerm*)term.get())->value();
135  break;
136  case TermType::INT32:
137  kvpair.type = KeyValuePair::TYPE_INT;
138  kvpair.value_int = ((Integer32Term*)term.get())->value();
139  break;
140  case TermType::LONG:
141  kvpair.type = KeyValuePair::TYPE_LONG;
142  kvpair.value_long = ((LongTerm*)term.get())->value();
143  break;
144  case TermType::LIST:
145  kvpair.type = KeyValuePair::TYPE_LIST;
146  ss << ((ListTerm*)term.get());
147  kvpair.value_list = ss.str();
148  break;
149  case TermType::VARIABLE:
150  // TDDO: Can a variable be in the answer?
151  case TermType::PREDICATE:
152  // TDDO: Can a predicate be in the answer?
153  break;
154  case TermType::MODAL_OPERATOR:
155  throw std::runtime_error("Modal Operator is not allowed as TermType in the answer." );
156  }
157  graphAnswer.substitution.push_back(kvpair);
158  }
159  return graphAnswer;
160 }

162 void ROSInterface::executeAskAllCB(const askallGoalConstPtr& goal)
163 {
164 
165  // Implement your action here
166  FormulaPtr phi(QueryParser::parse(goal->query.queryString));
167 
168  FormulaPtr mPhi = applyModality(goal->query, phi);
169 
170  auto resultStream = kb_.submitQuery(mPhi, QUERY_FLAG_ALL_SOLUTIONS);
171  auto resultQueue = resultStream->createQueue();
172 
173  int numSolutions_ = 0;
174  askallResult result;
175  while(true) {
176  auto nextResult = resultQueue->pop_front();
177 
178  if(AnswerStream::isEOS(nextResult)) {
179  break;
180  }
181  else {
182  if (nextResult->substitution()->empty()) {
183  numSolutions_ = 1;
184  break;
185  } else {
186  // Push one answer
187  GraphAnswerMessage answer = createGraphAnswer(nextResult);
188  result.answer.push_back(answer);
189  numSolutions_ += 1;
190  // publish feedback
191  askallFeedback feedback;
192  feedback.numberOfSolutions = numSolutions_;
193  askall_action_server_.publishFeedback(feedback);
194  }
195  }
196  }
197 
198  if(numSolutions_ == 0) {
199  result.status = askallResult::FALSE;
200  } else {
201  result.status = askallResult::TRUE;
202  }
203  askall_action_server_.setSucceeded(result);
204 }

277 void ROSInterface::executeTellCB(const tellGoalConstPtr &goal) {
278  FormulaPtr phi(QueryParser::parse(goal->query.queryString));
279 
280  FormulaPtr mPhi = applyModality(goal->query, phi);
281 
282  const QueryTree qt(phi);
283  if(qt.numPaths()>1) {
284  throw QueryError("Disjunctions are not allowed in assertions. "
285  "Appears in statement {}.", *phi);
286  }
287  else if(qt.numPaths()==0) {
288  throw QueryError("Invalid assertion: '{}'", *phi);
289  }
290 
291  std::vector<StatementData> data(qt.begin()->literals().size());
292  std::vector<RDFLiteralPtr> buf(qt.begin()->literals().size());
293  uint32_t dataIndex = 0;
294  for(auto &lit : qt.begin()->literals()) {
295  buf[dataIndex] = RDFLiteral::fromLiteral(lit);
296  data[dataIndex++] = buf[dataIndex]->toStatementData();
297  }
298 
299  tellResult result;
300  tellFeedback feedback;
301  if(kb_.insert(data)) {
302  result.status = tellResult::TRUE;
303  std::cout << "success, " << dataIndex << " statement(s) were asserted." << "\n";
304  }
305  else {
306  result.status = tellResult::TELL_FAILED;
307  std::cout << "assertion failed." << "\n";
308  }
309  feedback.finished = true;
310  tell_action_server_.publishFeedback(feedback);
311  tell_action_server_.setSucceeded(result);
312 }
