211 void TFRepublisher::advance_cursor()
212 {
213  double this_time = time_;
214  if(has_new_goal_) {
215  has_new_goal_ = false;
216  has_next_ = false;
217  set_initial_poses(time_min_);
218  create_cursor(time_min_);
219  }
220  else if(has_been_skipped_) {
221  has_been_skipped_ = false;
222  has_next_ = false;
223  // special reset mode because cursor needs to be recreated with proper start time
224  skip_reset_ = true;
225  // load initial poses
226  set_initial_poses(this_time);
227  // create cursor with documents from this_time to time_max_
228  create_cursor(this_time);
229  }
230  if(reset_) {
231  reset_ = false;
232  has_next_ = false;
233  // load initial poses to avoid problems with objects sticking at the position
234  // where they were at the end of the loop.
235  set_initial_poses(time_min_);
236  if(skip_reset_) {
237  skip_reset_ = false;
238  create_cursor(time_min_);
239  }
240  else {
241  reset_cursor();
242  }
243  }
244  //
245  while(1) {
246  // check if the cursor has an error.
247  // if this is the case, reset next loop.
248  bson_error_t cursor_error;
249  if (mongoc_cursor_error (cursor_, &cursor_error)) {
250  ROS_ERROR("[TFRepublisher] mongo cursor error: %s. Resetting..", cursor_error.message);
251  reset_ = true;
252  break;
253  }
254 
255  if(has_next_) {
256  double t_next = (ts_.header.stamp.sec * 1000.0 +
257  ts_.header.stamp.nsec / 1000000.0) / 1000.0;
258  if(t_next > this_time) {
259  // the next transform is too far in the future
260  break;
261  }
262  // push the next transform
263 #if CLEAR_MEMORY_AFTER_PUBLISH
264  memory_.set_managed_transform(ts_);
265 #else
266  memory_.set_transform(ts_);
267 #endif
268  }
269  // read the next transform
270  const bson_t *doc;
271  if(cursor_!=NULL && mongoc_cursor_next(cursor_,&doc)) {
272  read_transform(doc);
273  has_next_ = true;
274  }
275  else {
276  has_next_ = false;
277  break;
278  }
279  }
