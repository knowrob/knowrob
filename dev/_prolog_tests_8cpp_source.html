22 void PrologTestsBase::runPrologTests(
23  const std::shared_ptr<knowrob::PrologReasoner> &reasoner,
24  const std::string &target) {
25  bool hasResult = false;
26  int numTests = 0;
27  int numFailedTests = 0;
28 
29  for (const auto &t: reasoner->runTests(PrologEngine::getPrologPath(target))) {
30  hasResult = true;
31  numTests += 1;
32 
33  // each result is a term element/3
34  ASSERT_EQ(t->termType(), TermType::FUNCTION);
35  auto *pElem = (Function *) t.get();
36  ASSERT_EQ(*pElem->functor(), *Atom::Tabled("element"));
37  ASSERT_EQ(pElem->arity(), 3);
38  ASSERT_EQ(*(pElem->arguments()[0]), *Atom::Tabled("testcase"));
39 
40  // the second argument has the form:
41  // [name=::string, file=::string, line=::int, time=::double]
42  OptionList trace(pElem->arguments()[1]);
43  ASSERT_TRUE(trace.contains("file"));
44  ASSERT_TRUE(trace.contains("line"));
45  ASSERT_TRUE(trace.contains("name"));
46  const auto &name = trace.getString("name", "");
47  const char *file = trace.getString("file", "").data();
48  const long line = trace.getLong("line", 0);
49 
50  // the third argument is a list of failures.
51  auto *failureList = (ListTerm *) pElem->arguments()[2].get();
52  if (!failureList->isNIL()) {
53  numFailedTests += 1;
54  }
55  for (const auto &failureTerm: (*failureList)) {
56  ASSERT_EQ(failureTerm->termType(), TermType::FUNCTION);
57  // each failure is a term element/3
58  auto *errElem = (Function *) failureTerm.get();
59  ASSERT_EQ(*errElem->functor(), *Atom::Tabled("element"));
60  ASSERT_EQ(errElem->arity(), 3);
61  ASSERT_EQ(*(errElem->arguments()[0]), *Atom::Tabled("failure"));
62 
63  OptionList errOpts(errElem->arguments()[1]);
64  ASSERT_TRUE(errOpts.contains("type"));
65  ASSERT_TRUE(errOpts.contains("message"));
66 
67  std::ostringstream message_os;
68  message_os << errOpts.getString("type", "");
69  message_os << ": ";
70  message_os << errOpts.getString("message", "");
71  auto message = message_os.str();
72 
73  std::ostringstream summary_os;
74  summary_os << "test: ";
75  summary_os << name;
76  auto summary = summary_os.str();
77 
78  // the second argument has the form: [ type=error|failure, message='...' ]
79  // the third argument is the exact same message, which is a bit strange.
80  //ADD_FAILURE_AT(file,line) << message;
81  GTEST_MESSAGE_AT_(file, line, summary.c_str(), \
82  testing::TestPartResult::kNonFatalFailure) << unescapeString(message);
83  }
84  }
85  EXPECT_TRUE(hasResult);
86  KB_INFO1(target.c_str(), 1, "[plunit] {}/{} tests succeeded for target '{}'.", (numTests - numFailedTests), numTests, target);
87 }
