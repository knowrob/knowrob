35 void MongoTriplePattern::append(bson_t *selectorDoc,
36  const FramedTriplePattern &tripleExpression,
37  bool b_isTaxonomicProperty,
38  const std::shared_ptr<ImportHierarchy> &importHierarchy) {
39  // "s"" field
40  MongoTerm::append(selectorDoc,
41  "s", tripleExpression.subjectTerm());
42  // "p" field
43  MongoTerm::append(selectorDoc,
44  (b_isTaxonomicProperty ? "p" : "p*"),
45  tripleExpression.propertyTerm());
46  // "o" field
47  const char *objectOperator = getOperatorString(tripleExpression.objectOperator());
48  MongoTerm::append(selectorDoc,
49  (b_isTaxonomicProperty ? "o*" : "o"),
50  tripleExpression.objectTerm(),
51  objectOperator);
52  // "g" field
53  appendGraphSelector(selectorDoc, tripleExpression, importHierarchy);
54  // epistemic fields
55  appendEpistemicSelector(selectorDoc, tripleExpression);
56  // temporal fields
57  appendTimeSelector(selectorDoc, tripleExpression);

97 void MongoTriplePattern::appendEpistemicSelector(bson_t *selectorDoc, const FramedTriplePattern &tripleExpression) {
98  static const bool allowConfidenceNullValues = true;
99  auto &ct = tripleExpression.confidenceTerm();
100  auto &at = tripleExpression.perspectiveTerm();
101  auto &u = tripleExpression.isUncertainTerm();
102 
103  if (!((u.has_grounding() && u.grounded()->asBoolean()) || u.has_variable())) {
104  // note: null value of "uncertain" field is seen as value "false"
105  MongoTerm::append(
106  selectorDoc,
107  "uncertain",
108  Numeric::falseAtom(),
109  nullptr,
110  true);
111  }
112 
113  if (at) {
114  if (at->termType() == TermType::ATOMIC) {
115  MongoTerm::append(
116  selectorDoc,
117  "agent",
118  *at,
119  nullptr,
120  false);
121  } else {
122  KB_WARN("agent term {} has unexpected type", *at);
123  }
124  } else {
125  // make sure agent field is undefined: { agent: { $exists: false } }
126  // note: null value of agent field is seen as "self", i.e. the agent running the knowledge base
127  bson_t agentDoc;
128  BSON_APPEND_DOCUMENT_BEGIN(selectorDoc, "agent", &agentDoc);
129  BSON_APPEND_BOOL(&agentDoc, "$exists", false);
130  bson_append_document_end(selectorDoc, &agentDoc);
131  }
132 
133  if (ct) {
134  if (ct->termType() == TermType::ATOMIC) {
135  // note: null value of confidence is seen as larger than the requested threshold
136  MongoTerm::append(
137  selectorDoc,
138  "confidence",
139  *ct,
140  MONGO_OPERATOR_GTE,
141  true);
142  } else {
143  KB_WARN("confidence term {} has unexpected type", *ct);
144  }
145  }

206 static inline void appendSetVariable(bson_t *doc, std::string_view varName, std::string_view newValue) {
207  const auto varKey = MongoTerm::variableKey(varName);
208  const auto varValue = std::string("$") + varKey;
209  bson_t condDoc, condArray, notOperator, notArray;
210  // Only conditionally apply new value in case it has not been grounded before.
211  // varKey: {$cond: [ { $not: [ varValue ] }, newValue, varValue ]}
212  BSON_APPEND_DOCUMENT_BEGIN(doc, varKey.c_str(), &condDoc);
213  BSON_APPEND_ARRAY_BEGIN(&condDoc, "$cond", &condArray);
214  {
215  BSON_APPEND_DOCUMENT_BEGIN(&condArray, "0", &notOperator);
216  BSON_APPEND_ARRAY_BEGIN(&notOperator, "$not", &notArray);
217  BSON_APPEND_UTF8(&notArray, "0", varValue.c_str());
218  bson_append_array_end(&notOperator, &notArray);
219  bson_append_document_end(&condArray, &notOperator);
220 
221  BSON_APPEND_UTF8(&condArray, "1", newValue.data());
222  BSON_APPEND_UTF8(&condArray, "2", varValue.c_str());
223  }
224  bson_append_array_end(&condDoc, &condArray);
225  bson_append_document_end(doc, &condDoc);
226 }

228 static inline void appendMatchVariable(bson_t *doc,
229  std::string_view fieldValue,
230  std::string_view varName,
231  const TripleLookupData &lookupData) {
232  const auto varKey = MongoTerm::variableKey(varName);
233  const auto varLetValue = std::string("$$") + varKey;
234  const auto matchOperator = (fieldValue.back() == '*' ? "$in" : "$eq");
235  bson_t exprDoc, orArr, notDoc, notArr, matchDoc, matchArr;
236 
237  BSON_APPEND_DOCUMENT_BEGIN(doc, "$expr", &exprDoc);
238 
239  if (lookupData.knownGroundedVariables.count(varName) > 0) {
240  // variable is known to have a grounding in the v_VARS field
241  // {$expr: { matchOperator: [varLetValue, fieldValue] } }
242  BSON_APPEND_ARRAY_BEGIN(&exprDoc, matchOperator, &matchArr);
243  BSON_APPEND_UTF8(&matchArr, "0", varLetValue.c_str());
244  BSON_APPEND_UTF8(&matchArr, "1", fieldValue.data());
245  bson_append_array_end(&exprDoc, &matchArr);
246  } else {
247  // {$expr: {$or: [ { $not: [ varLetValue ] }, { matchOperator: [varLetValue, fieldValue] } ]}}
248  BSON_APPEND_ARRAY_BEGIN(&exprDoc, "$or", &orArr);
249 
250  // { $not: [ varLetValue ] }
251  BSON_APPEND_DOCUMENT_BEGIN(&orArr, "0", &notDoc);
252  BSON_APPEND_ARRAY_BEGIN(&notDoc, "$not", &notArr);
253  BSON_APPEND_UTF8(&notArr, "0", varLetValue.c_str());
254  bson_append_array_end(&notDoc, &notArr);
255  bson_append_document_end(&orArr, &notDoc);
256 
257  // { matchOperator: [varLetValue, fieldValue] }
258  BSON_APPEND_DOCUMENT_BEGIN(&orArr, "1", &matchDoc);
259  BSON_APPEND_ARRAY_BEGIN(&matchDoc, matchOperator, &matchArr);
260  BSON_APPEND_UTF8(&matchArr, "0", varLetValue.c_str());
261  BSON_APPEND_UTF8(&matchArr, "1", fieldValue.data());
262  bson_append_array_end(&matchDoc, &matchArr);
263  bson_append_document_end(&orArr, &matchDoc);
264 
265  bson_append_array_end(&exprDoc, &orArr);
266  }
267  bson_append_document_end(doc, &exprDoc);
268 }

270 static void setTripleVariables(Pipeline &pipeline, const TripleLookupData &lookupData) {
271  std::list<std::pair<const char *, Variable *>> varList;
272 
273  // the object can be specified as a triple (Value, Operator, ActualValue) such that
274  // a value constraint is given plus a variable where the actual value of the triple should be stored.
275  TermPtr objectVar = lookupData.expr->objectTerm();
276  if (objectVar && !objectVar->isVariable() && lookupData.expr->objectVariable()) {
277  objectVar = lookupData.expr->objectVariable();
278  }
279 
280  for (auto &it: {
281  std::make_pair("$next.s", lookupData.expr->subjectTerm()),
282  std::make_pair("$next.p", lookupData.expr->propertyTerm()),
283  std::make_pair("$next.o", objectVar),
284  std::make_pair("$next.graph", lookupData.expr->graphTerm().get()),
285  std::make_pair("$next.confidence", lookupData.expr->confidenceTerm().get()),
286  std::make_pair("$next.agent", lookupData.expr->perspectiveTerm().get()),
287  std::make_pair("$next.scope.time.since", lookupData.expr->beginTerm().get()),
288  std::make_pair("$next.scope.time.until", lookupData.expr->endTerm().get()),
289  std::make_pair("$next.uncertain", lookupData.expr->isUncertainTerm().get()),
290  std::make_pair("$next.occasional", lookupData.expr->isOccasionalTerm().get())
291  }) {
292  if (!it.second || it.second->termType() != TermType::VARIABLE) continue;
293  auto var = (Variable *) it.second.get();
294  // skip variables that were instantiated in previous steps
295  if (lookupData.knownGroundedVariables.count(var->name()) > 0) continue;
296  varList.emplace_back(it.first, var);
297  }
298 
299  if (!varList.empty()) {
300  auto setVariables = pipeline.appendStageBegin("$set");
301  for (auto &it: varList) {
302  appendSetVariable(setVariables, it.second->name(), it.first);
303  }
304  pipeline.appendStageEnd(setVariables);
305  }
306 }

308 static void nonTransitiveLookup(
309  Pipeline &pipeline,
310  const TripleStore &tripleStore,
311  const TripleLookupData &lookupData,
312  const semweb::PropertyPtr &definedProperty) {
313  bool b_isTaxonomicProperty = (definedProperty &&
314  tripleStore.vocabulary->isTaxonomicProperty(definedProperty->iri()));
315  char arrIndexStr[16];
316  const char *arrIndexKey;
317  uint32_t arrIndex;
318 
319  bool b_skipInputGroundings = false;
320  if (lookupData.expr->numVariables() == 0) {
321  // the triple expression has no variables
322  b_skipInputGroundings = true;
323  } else if (!lookupData.mayHasMoreGroundings) {
324  // all possible previous instantiations are known and stored in lookupData.knownGroundedVariables.
325  b_skipInputGroundings = true;
326  for (auto &exprTerm: {
327  lookupData.expr->subjectTerm(),
328  lookupData.expr->propertyTerm(),
329  lookupData.expr->objectTerm()
330  }) {
331  if (exprTerm->termType() != TermType::VARIABLE) continue;
332  auto var = (Variable *) exprTerm.get();
333  // skip if this variable cannot have a runtime grounding
334  if (lookupData.knownGroundedVariables.count(var->name()) > 0) {
335  // the expression contains a variable that is known to have received a grounding before
336  b_skipInputGroundings = false;
337  break;
338  }
339  }
340  }
341 
342  // filter out documents that do not match the triple pattern. this is done using $match or $lookup operators.
343  // first lookup matching documents and store them in the 'next' field
344  bson_t lookupArray, letDoc;
345  auto lookupStage = pipeline.appendStageBegin("$lookup");
346  BSON_APPEND_UTF8(lookupStage, "from", tripleStore.tripleCollection->name().data());
347  BSON_APPEND_UTF8(lookupStage, "as", "next");
348  if (!b_skipInputGroundings) {
349  // pass "v_VARS" field to lookup pipeline
350  BSON_APPEND_DOCUMENT_BEGIN(lookupStage, "let", &letDoc);
351  BSON_APPEND_UTF8(&letDoc, "v_VARS", "$v_VARS");
352  bson_append_document_end(lookupStage, &letDoc);
353  }
354  BSON_APPEND_ARRAY_BEGIN(lookupStage, "pipeline", &lookupArray);
355  {
356  Pipeline lookupPipeline(&lookupArray);
357 
358  auto matchStage = lookupPipeline.appendStageBegin("$match");
359  if (b_skipInputGroundings) {
360  MongoTriplePattern::append(matchStage, *lookupData.expr, b_isTaxonomicProperty,
361  tripleStore.vocabulary->importHierarchy());
362  } else {
363  // need to match with potential groundings of variables from previous steps these are stored in the "v_VARS" field.
364  bson_t andArray, tripleDoc, variablesDoc;
365  BSON_APPEND_ARRAY_BEGIN(matchStage, "$and", &andArray);
366  {
367  BSON_APPEND_DOCUMENT_BEGIN(&andArray, "0", &tripleDoc);
368  MongoTriplePattern::append(&tripleDoc, *lookupData.expr, b_isTaxonomicProperty,
369  tripleStore.vocabulary->importHierarchy());
370  bson_append_document_end(&andArray, &tripleDoc);
371 
372  // match triple values with previously grounded variables
373  arrIndex = 1;
374  for (auto &it: {
375  std::make_pair("$s", lookupData.expr->subjectTerm()),
376  std::make_pair(b_isTaxonomicProperty ? "$p" : "$p*", lookupData.expr->propertyTerm()),
377  std::make_pair(b_isTaxonomicProperty ? "$o*" : "$o", lookupData.expr->objectTerm())
378  }) {
379  if (it.second->termType() != TermType::VARIABLE) continue;
380  auto var = (Variable *) it.second.get();
381  // skip if this variable cannot have a runtime grounding
382  if (!lookupData.mayHasMoreGroundings &&
383  lookupData.knownGroundedVariables.count(var->name()) == 0)
384  continue;
385 
386  bson_uint32_to_string(arrIndex++,
387  &arrIndexKey, arrIndexStr, sizeof arrIndexStr);
388  BSON_APPEND_DOCUMENT_BEGIN(&andArray, arrIndexKey, &variablesDoc);
389  appendMatchVariable(&variablesDoc, it.first, var->name(), lookupData);
390  bson_append_document_end(&andArray, &variablesDoc);
391  }
392  }
393  bson_append_array_end(matchStage, &andArray);
394  }
395  lookupPipeline.appendStageEnd(matchStage);
396  // { $limit: maxNumOfTriples }
397  if (lookupData.maxNumOfTriples > 0) lookupPipeline.limit(lookupData.maxNumOfTriples);
398  }
399  bson_append_array_end(lookupStage, &lookupArray);
400  pipeline.appendStageEnd(lookupStage);
401 
402  // at this point the 'next' field holds an array of matching documents, if any.
403  if (lookupData.expr->isNegated()) {
404  // following closed-world assumption succeed if no solutions are found
405  pipeline.matchEmptyArray("next");
406  } else {
407  // Unwind, and set preserveEmpty=true if lookupData.expr->isOptional().
408  pipeline.unwind("$next", lookupData.expr->isOptional());
409  // project new variable groundings
410  setTripleVariables(pipeline, lookupData);
411  }
412 
413  // finally, remove next field: { $unset: "next" }
414  pipeline.unset("next");
415 }

417 static void transitiveLookup(
418  Pipeline &pipeline,
419  const TripleStore &tripleStore,
420  const TripleLookupData &lookupData,
421  const semweb::PropertyPtr &definedProperty) {
422  bool b_isTaxonomicProperty = (definedProperty &&
423  tripleStore.vocabulary->isTaxonomicProperty(definedProperty->iri()));
424  // start with object in case it is known to be grounded before at runtime
425  bool b_objectIsKnownGrounded = lookupData.expr->objectTerm()->isVariable() &&
426  lookupData.knownGroundedVariables.count(std::static_pointer_cast<Variable>(lookupData.expr->objectTerm())->name()) > 0;
427  bool b_startWithSubject = lookupData.expr->subjectTerm()->isGround() || !b_objectIsKnownGrounded;
428 
429  auto startTerm = (b_startWithSubject ?
430  lookupData.expr->subjectTerm() : lookupData.expr->objectTerm());
431  auto endTerm = (b_startWithSubject ?
432  lookupData.expr->objectTerm() : lookupData.expr->subjectTerm());
433 
434  // recursive lookup
435  bson_t restrictSearchDoc;
436  auto lookupStage = pipeline.appendStageBegin("$graphLookup");
437  BSON_APPEND_UTF8(lookupStage, "from", tripleStore.tripleCollection->name().data());
438  if (startTerm->termType() == TermType::ATOMIC) {
439  auto startString = (Atomic *) startTerm.get();
440  BSON_APPEND_UTF8(lookupStage, "startWith", startString->stringForm().data());
441  } else if (startTerm->termType() == TermType::VARIABLE) {
442  auto startVariable = (Variable *) startTerm.get();
443  auto startValue = std::string("$") + MongoTerm::variableKey(startVariable->name());
444  BSON_APPEND_UTF8(lookupStage, "startWith", startValue.c_str());
445  } else {
446  KB_WARN("Ignoring term {} with invalid type for graph lookup.", *startTerm);
447  }
448  BSON_APPEND_UTF8(lookupStage, "connectToField", b_startWithSubject ? "s" : "o");
449  BSON_APPEND_UTF8(lookupStage, "connectFromField", b_startWithSubject ? "p" : "s");
450  BSON_APPEND_UTF8(lookupStage, "as", "t_paths");
451  BSON_APPEND_UTF8(lookupStage, "depthField", "depth");
452  /* { restrictSearchWithMatch: { p*: Query_p, ... }" */
453  BSON_APPEND_DOCUMENT_BEGIN(lookupStage, "restrictSearchWithMatch", &restrictSearchDoc);
454  {
455  MongoTerm::append(&restrictSearchDoc,
456  (b_isTaxonomicProperty ? "p" : "p*"),
457  lookupData.expr->propertyTerm());
458  MongoTriplePattern::appendGraphSelector(&restrictSearchDoc, *lookupData.expr, tripleStore.vocabulary->importHierarchy());
459  MongoTriplePattern::appendEpistemicSelector(&restrictSearchDoc, *lookupData.expr);
460  MongoTriplePattern::appendTimeSelector(&restrictSearchDoc, *lookupData.expr);
461  }
462  bson_append_document_end(lookupStage, &restrictSearchDoc);
463  pipeline.appendStageEnd(lookupStage);
464 
465  // $graphLookup does not ensure order, so we need to order by recursion depth in a separate step
466  // t_sorted = { $sortArray: { input: "$t_paths", sort: { depth: 1 } } }
467  bson_t setSortedDoc, sortArrayDoc, sortByDoc;
468  auto setSortedStage = pipeline.appendStageBegin("$set");
469  BSON_APPEND_DOCUMENT_BEGIN(setSortedStage, "t_sorted", &setSortedDoc);
470  BSON_APPEND_DOCUMENT_BEGIN(&setSortedDoc, "$sortArray", &sortArrayDoc);
471  BSON_APPEND_UTF8(&sortArrayDoc, "input", "$t_paths");
472  BSON_APPEND_DOCUMENT_BEGIN(&sortArrayDoc, "$sortArray", &sortByDoc);
473  BSON_APPEND_INT32(&sortByDoc, "depth", 1);
474  bson_append_document_end(&sortArrayDoc, &sortByDoc);
475  bson_append_document_end(&setSortedDoc, &sortArrayDoc);
476  bson_append_document_end(setSortedStage, &setSortedDoc);
477  pipeline.appendStageEnd(setSortedStage);
478 
479  // { $set: { next: "$t_sorted" } }
480  auto setNext = pipeline.appendStageBegin("$set");
481  BSON_APPEND_UTF8(setNext, "next", "$t_sorted");
482  pipeline.appendStageEnd(setNext);
483 
484  // { $set: { start: { $arrayElemAt: ["$next", 0] } } }
485  bson_t setStartOperation, setStartArr;
486  auto setStart = pipeline.appendStageBegin("$set");
487  BSON_APPEND_DOCUMENT_BEGIN(setStart, "start", &setStartOperation);
488  BSON_APPEND_ARRAY_BEGIN(&setStartOperation, "$arrayElemAt", &setStartArr);
489  BSON_APPEND_UTF8(&setStartArr, "0", "$next");
490  BSON_APPEND_INT32(&setStartArr, "1", 0);
491  bson_append_array_end(&setStartOperation, &setStartArr);
492  bson_append_document_end(setStart, &setStartOperation);
493  pipeline.appendStageEnd(setStart);
494 
495  // { $unset: "t_paths" }, { $unset: "t_sorted" }
496  pipeline.unset("t_paths");
497  pipeline.unset("t_sorted");
498 
499  // iterate over results: { $unwind: "$next" }
500  pipeline.unwind("$next");
501 
502  // graph lookup uses "s" or "o" as start for recursive lookup but ignores the other.
503  // thus a matching must be performed for the results.
504  if (endTerm->termType() != TermType::VARIABLE) {
505  // FIXME: must add another match case for endTerm being a runtime grounded variable.
506  bson_t matchEndVal;
507  auto matchEnd = pipeline.appendStageBegin("$match");
508  BSON_APPEND_DOCUMENT_BEGIN(matchEnd, "next.o", &matchEndVal);
509  MongoTerm::append(&restrictSearchDoc,
510  (b_startWithSubject ? "next.o" : "next.s"),
511  endTerm);
512  bson_append_document_end(matchEnd, &matchEndVal);
513  pipeline.appendStageEnd(matchEnd);
514  }
515 
516  // project new variable groundings
517  setTripleVariables(pipeline, lookupData);
518  // remove next field again: { $unset: "next" }
519  pipeline.unset("next");
520 }

522 void mongo::lookupTriple(Pipeline &pipeline, const TripleStore &tripleStore, const TripleLookupData &lookupData) {
523  semweb::PropertyPtr definedProperty;
524  if (lookupData.expr->propertyTerm() && lookupData.expr->propertyTerm()->termType() == TermType::ATOMIC) {
525  auto propertyTerm = std::static_pointer_cast<Atomic>(lookupData.expr->propertyTerm());
526  definedProperty = tripleStore.vocabulary->getDefinedProperty(propertyTerm->stringForm());
527  }
528 
529  bool b_isTransitiveProperty = (definedProperty && definedProperty->hasFlag(
530  semweb::PropertyFlag::TRANSITIVE_PROPERTY));
531  if (b_isTransitiveProperty)
532  transitiveLookup(pipeline, tripleStore, lookupData, definedProperty);
533  else
534  nonTransitiveLookup(pipeline, tripleStore, lookupData, definedProperty);
535 }
