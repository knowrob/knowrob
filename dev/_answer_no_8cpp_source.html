34 bool AnswerNo::mergeWith(const AnswerNo &other) {
35  reasonerTerm_ = {};
36  if (!frame_->mergeWith(*other.frame_)) {
37  // merging frames failed -> results cannot be combined
38  return false;
39  }
40  // insert groundings of other answer

42  other.positiveUngrounded_.begin(), other.positiveUngrounded_.end());

44  other.negativeUngrounded_.begin(), other.negativeUngrounded_.end());
