155 foreign_t sw_load_rdf_xml4(term_t t_manager, term_t t_reasoner, term_t t_url, term_t t_parentGraph) {
156  auto kb = getKnowledgeBase(t_manager, t_reasoner);
157  char *url, *parentGraph;
158  if (kb && PL_get_atom_chars(t_url, &url) && PL_get_atom_chars(t_parentGraph, &parentGraph)) {
159  URI ontologyURI(url);
160  auto ontologyFile = std::make_shared<OntologyFile>(ontologyURI, "rdf-xml");
161  ontologyFile->setParentOrigin(parentGraph);
162  if (kb->loadDataSource(ontologyFile)) {
163  return true;
164  }
165  }
166  return false;
167 }
