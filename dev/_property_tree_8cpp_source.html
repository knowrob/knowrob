40 void PropertyTree::loadProperty( //NOLINT(misc-no-recursion)
41  const std::string &key, const boost::property_tree::ptree &remainder) {
42  if (remainder.empty()) {
43  auto stringVal = remainder.get_value<std::string>();
44  properties_.emplace(key, std::make_shared<String>(stringVal));
45  } else {
46  for (const auto &key_val: remainder) {
47  if (key_val.first.empty()) {
48  // this indicates a list. Elements of lists must be atomic values.
49  auto &listData = key_val.second;
50  std::vector<TermPtr> elements(listData.size());
51  int index = 0;
52  for(const auto& l_key_val : listData) {
53  elements[index++] = std::make_shared<String>(l_key_val.second.get_value<std::string>());
54  }
55  properties_.emplace(key, std::make_shared<ListTerm>(elements));
56  } else {
57  auto key_t = key + delimiter_ + std::string(key_val.first);
58  loadProperty(key_t, key_val.second);
59  }
60  }
61  }

73 TermPtr PropertyTree::createKeyTerm(std::string_view key) const {
74  TermPtr last_key, next_key;
75 
76  size_t pos_start = 0, pos_end, delim_len = delimiter_.length();
77  std::string_view token;
78  std::vector<std::string_view> res;
79 
80  while ((pos_end = key.find(delimiter_, pos_start)) != std::string::npos) {
81  token = key.substr(pos_start, pos_end - pos_start);
82  pos_start = pos_end + delim_len;
83  next_key = Atom::Tabled(token);
84  if (last_key) {
85  last_key = std::make_shared<Function>(Function(delimiter_, {last_key, next_key}));
86  } else {
87  last_key = next_key;
88  }
89  res.push_back(token);
90  }
91  if (!last_key) {
92  last_key = Atom::Tabled(key);
93  }
94 
95  return last_key;
96 }
