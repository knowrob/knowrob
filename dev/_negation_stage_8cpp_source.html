38 bool PredicateNegationStage::succeeds(const AnswerYesPtr &answer) {
39  // here the idea is to issue individual queries for each negative literal.
40  // each query produces an AnswerBufferPtr object.
41  // the set of all results can then be checked for a certain property, e.g. that none or all
42  // of the queries produced a result.
43  std::vector<TokenBufferPtr> results;
44 
45  auto kg = kb_->getBackendForQuery();
46 
47  for (auto &pat: negatedLiterals_) {
48  auto pat1 = applyBindings(pat, *answer->substitution());
49 
50  // create an instance of the pattern based on given substitution
51  auto instance = std::make_shared<FramedTriplePattern>(
52  pat1->predicate(), pat1->isNegated());
53  instance->setTripleFrame(ctx_->selector);
54  // for now evaluate positive variant of the pattern.
55  // NOTE: for open-world semantics this cannot be done. open-world reasoner
56  // would need to receive negative literal instead.
57  instance->setIsNegated(false);
58 
59  // check if the EDB contains positive lit, if so negation cannot be true
60  results.push_back(kb_->edb()->getAnswerCursor(
61  kg, std::make_shared<GraphPathQuery>(instance, ctx_)));
62 
63  // next check if positive lit is an IDB predicate, if so negation cannot be true.
64  // get list of reasoner that define the literal
65  auto l_property = instance->propertyTerm();
66  if (!l_property->isAtom()) {
67  KB_WARN("PredicateNegationStage: property term is not an atom, skipping.");
68  continue;
69  }
70  auto l_property_a = std::static_pointer_cast<Atomic>(l_property);
71  auto l_reasoner = kb_->reasonerManager()->getReasonerForRelation(PredicateIndicator(l_property_a->stringForm(), 2));
72  for (auto &r: l_reasoner) {
73  results.push_back(r->submitQuery(instance, ctx_));
74  }
75  }
76 
77  // go through results, ensure *all* sub-queries failed, only then the negation stage succeeds.
78  for (auto &result: results) {
79  auto resultQueue = result->createQueue();
80  auto firstResult = resultQueue->pop_front();
81  if (firstResult->type() == TokenType::ANSWER_TOKEN) {
82  if (std::static_pointer_cast<const Answer>(firstResult)->isPositive()) {
83  return false;
84  }
85  }
86  }
87 
88  return true;
89 }

97 bool ModalNegationStage::succeeds(const AnswerYesPtr &answer) {
98  // here the idea is to issue individual queries for each negated modal.
99  // for now the positive variants of the modals is evaluated, the modal negation is thought to be
100  // true if none of the positive variants can be grounded.
101 
102  auto outputStream = std::make_shared<TokenBuffer>();
103 
104  // run "submitQuery" for each negated modal, and collect results in outputStream
105  for (auto &modal: negatedModals_) {
106  auto modalInstance = applyBindings(modal, *answer->substitution());
107  auto modalOutput = kb_->submitQuery(modalInstance, ctx_);
108  modalOutput >> outputStream;
109  modalOutput->stopBuffering();
110  }
111 
112  // wait for and pop first message from output stream
113  auto resultQueue = outputStream->createQueue();
114  auto firstResult = resultQueue->pop_front();
115  // "EOS" indicates that no solution was found for any of the modals
116  return firstResult->indicatesEndOfEvaluation();
117 }
