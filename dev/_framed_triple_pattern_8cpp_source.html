205 static bool filterString(std::string_view value, const FramedTriplePattern &query) {
206  auto &q_term = query.objectTerm();
207  if (!q_term->isAtomic()) return false;
208  auto q_atomic = std::static_pointer_cast<Atomic>(q_term);
209  switch (query.objectOperator()) {
210  case FilterType::EQ:
211  return value == q_atomic->stringForm();
212  case FilterType::NEQ:
213  return value != q_atomic->stringForm();
214  case FilterType::LEQ:
215  return value <= q_atomic->stringForm();
216  case FilterType::GEQ:
217  return value >= q_atomic->stringForm();
218  case FilterType::GT:
219  return value > q_atomic->stringForm();
220  case FilterType::LT:
221  return value < q_atomic->stringForm();
222  }
223  return false;
224 }

245 bool FramedTriplePattern::filter(const FramedTriple &triple) const {
246  if (triple.isObjectIRI() || triple.isObjectBlank()) {
247  return filterString(triple.valueAsString(), *this);
248  } else if (triple.xsdType()) {
249  if (triple.xsdType().value() == XSDType::STRING) {
250  return filterString(triple.valueAsString(), *this);
251  }
252  auto &q_term = objectTerm();
253  if (!q_term->isNumeric()) return false;
254  auto q_numeric = std::static_pointer_cast<Numeric>(q_term);
255  switch (triple.xsdType().value()) {
256  case knowrob::XSDType::BOOLEAN:
257  return filterNumeric(triple.valueAsBoolean(), q_numeric->asBoolean(), objectOperator());
258  case knowrob::XSDType::DOUBLE:
259  return filterNumeric(triple.valueAsDouble(), q_numeric->asDouble(), objectOperator());
260  case knowrob::XSDType::FLOAT:
261  return filterNumeric(triple.valueAsFloat(), q_numeric->asFloat(), objectOperator());
262  case knowrob::XSDType::NON_NEGATIVE_INTEGER:
263  case knowrob::XSDType::INTEGER:
264  return filterNumeric(triple.valueAsInt(), q_numeric->asInteger(), objectOperator());
265  case knowrob::XSDType::LONG:
266  return filterNumeric(triple.valueAsLong(), q_numeric->asLong(), objectOperator());
267  case knowrob::XSDType::SHORT:
268  return filterNumeric(triple.valueAsShort(), q_numeric->asShort(), objectOperator());
269  case knowrob::XSDType::UNSIGNED_LONG:
270  return filterNumeric(triple.valueAsUnsignedLong(), q_numeric->asUnsignedLong(), objectOperator());
271  case knowrob::XSDType::UNSIGNED_INT:
272  return filterNumeric(triple.valueAsUnsignedInt(), q_numeric->asUnsignedInteger(), objectOperator());
273  case knowrob::XSDType::UNSIGNED_SHORT:
274  return filterNumeric(triple.valueAsUnsignedShort(), q_numeric->asUnsignedShort(), objectOperator());
275  case knowrob::XSDType::STRING:
276  case knowrob::XSDType::LAST:
277  break;
278  }
279  }
280  return false;
281 }

284 FramedTriplePattern::instantiateInto(FramedTriple &triple, const std::shared_ptr<const Bindings> &bindings) const {
285  // return a flag that indicates if s/p/o were assigned successfully.
286  bool hasMissingSPO = false;
287  // handle subject
288  if (subjectTerm_) {
289  auto &s = subjectTerm_->isVariable() ?
290  bindings->get(std::static_pointer_cast<Variable>(subjectTerm_)->name()) :
291  subjectTerm_;
292  if (!s) {
293  hasMissingSPO = true;
294  } else if (s->isBlank()) {
295  triple.setSubjectBlank(std::static_pointer_cast<Blank>(s)->stringForm());
296  } else if (s->termType() == TermType::ATOMIC) {
297  triple.setSubject(std::static_pointer_cast<Atomic>(s)->stringForm());
298  } else {
299  hasMissingSPO = true;
300  }
301  } else {
302  hasMissingSPO = true;
303  }
304  // handle property
305  if (propertyTerm_) {
306  auto &p = propertyTerm_->isVariable() ?
307  bindings->get(std::static_pointer_cast<Variable>(propertyTerm_)->name()) :
308  propertyTerm_;
309  if (p && p->termType() == TermType::ATOMIC) {
310  triple.setPredicate(std::static_pointer_cast<Atomic>(p)->stringForm());
311  } else {
312  hasMissingSPO = true;
313  }
314  } else {
315  hasMissingSPO = true;
316  }
317  // handle object
318  if (objectTerm_ || objectVariable_) {
319  auto &o = (objectTerm_ && objectTerm_->isVariable()) ?
320  bindings->get(std::static_pointer_cast<Variable>(objectTerm_)->name()) :
321  (objectVariable_ && objectVariable_->isVariable()) ?
322  bindings->get(std::static_pointer_cast<Variable>(objectVariable_)->name()) :
323  objectTerm_;
324 
325  if (!o) {
326  hasMissingSPO = true;
327  } else if (o->isNumeric()) {
328  auto numeric = std::static_pointer_cast<Numeric>(o);
329  triple.setXSDValue(numeric->stringForm(), numeric->xsdType());
330  } else if (o->termType() == TermType::ATOMIC) {
331  auto atom = std::static_pointer_cast<Atomic>(o);
332  if (atom->isIRI()) {
333  triple.setObjectIRI(atom->stringForm());
334  } else if (atom->isBlank()) {
335  triple.setObjectBlank(atom->stringForm());
336  } else {
337  triple.setStringValue(atom->stringForm());
338  }
339  } else {
340  hasMissingSPO = true;
341  }
342  } else {
343  hasMissingSPO = true;
344  }
345  // handle optional properties
346  if (graphTerm_) {
347  auto &g = graphTerm_->isVariable() ?
348  bindings->get(std::static_pointer_cast<Variable>(*graphTerm_)->name()) :
349  *graphTerm_;
350  if (g && g->termType() == TermType::ATOMIC) {
351  triple.setGraph(std::static_pointer_cast<Atomic>(g)->stringForm());
352  }
353  }
354  if (perspectiveTerm_) {
355  auto &a = perspectiveTerm_->isVariable() ?
356  bindings->get(std::static_pointer_cast<Variable>(*perspectiveTerm_)->name()) :
357  *perspectiveTerm_;
358  if (a && a->termType() == TermType::ATOMIC) {
359  triple.setPerspective(std::static_pointer_cast<Atomic>(a)->stringForm());
360  }
361  }
362  if (confidenceTerm_) {
363  auto &c = confidenceTerm_->isVariable() ?
364  bindings->get(std::static_pointer_cast<Variable>(*confidenceTerm_)->name()) :
365  *confidenceTerm_;
366  if (c && c->isNumeric()) {
367  triple.setConfidence(std::static_pointer_cast<Numeric>(c)->asDouble());
368  }
369  }
370  if (beginTerm_) {
371  auto &b = beginTerm_->isVariable() ?
372  bindings->get(std::static_pointer_cast<Variable>(*beginTerm_)->name()) :
373  *beginTerm_;
374  if (b && b->isNumeric()) {
375  triple.setBegin(std::static_pointer_cast<Numeric>(b)->asDouble());
376  }
377  }
378  if (endTerm_) {
379  auto &e = endTerm_->isVariable() ?
380  bindings->get(std::static_pointer_cast<Variable>(*endTerm_)->name()) :
381  *endTerm_;
382  if (e && e->isNumeric()) {
383  triple.setEnd(std::static_pointer_cast<Numeric>(e)->asDouble());
384  }
385  }
386  if (isOccasional_) {
387  auto &o = isOccasional_->isVariable() ?
388  bindings->get(std::static_pointer_cast<Variable>(*isOccasional_)->name()) :
389  *isOccasional_;
390  if (o && o->isNumeric()) {
391  triple.setIsOccasional(std::static_pointer_cast<Numeric>(o)->asBoolean());
392  }
393  }
394  if (isUncertain_) {
395  auto &u = isUncertain_->isVariable() ?
396  bindings->get(std::static_pointer_cast<Variable>(*isUncertain_)->name()) :
397  *isUncertain_;
398  if (u && u->isNumeric()) {
399  triple.setIsUncertain(std::static_pointer_cast<Numeric>(u)->asBoolean());
400  }
401  }
402 
403  return !hasMissingSPO;
404 }
