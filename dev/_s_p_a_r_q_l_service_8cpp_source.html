15 SPARQLService::SPARQLService(const URI &uri, std::string_view format)
16  : DataService(uri, format),
17  OntologySource(),
18  origin_(DataSource::getNameFromURI(uri())) {
19  // create a Redland model for the SPARQL endpoint
20  // Note that the model has interfaces for insert and remove triples, these won't
21  // interact with the SPARQL endpoint, but with the local storage of the model.
22  model_.setStorageType(RedlandStorageType::MEMORY);
23  model_.setOrigin(origin_);
24  if (!model_.initializeBackend()) {
25  throw BackendError("Failed to initialize Redland backend for SPARQL endpoint at \"{}\".", uri());
26  }
27  // set up the model to interface with the SPARQL endpoint
28  if (!model_.load(uri, semweb::tripleFormatFromString(format))) {
29  throw BackendError("Failed to load URI of SPARQL endpoint at \"{}\".", uri());
30  }
31 }
