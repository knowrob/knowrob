36 std::string DataSource::getVersionFromURI(const std::string &uriString) {
37  fs::path p(uriString);
38 
39  // check if it is a local existing file and use file modification time
40  // as version in this case.
41  if (exists(p)) {
42  std::ostringstream oss;
43  auto stamp = last_write_time(p);
44  auto tt = to_time_t(stamp);
45  auto tm = *std::localtime(&tt);
46  oss << std::put_time(&tm, "%c");
47  return oss.str();
48  }
49 
50  // try to extract version from URI
51  auto versionString = p.parent_path().filename();
52  if (isVersionString(versionString)) {
53  return versionString;
54  }
55 
56  // fallback to use the current day as version, thus causing
57  // a reload each day.
58  {
59  std::ostringstream oss;
60  auto t = std::time(nullptr);
61  auto tm = *std::localtime(&t);
62  oss << std::put_time(&tm, "%d-%m-%Y");
63  return oss.str();
64  }
65 }

67 bool DataSource::isVersionString(const std::string &versionString) {
68  // parser rules
69  qi::rule<std::string::const_iterator> numbers = (
70  (+qi::digit >> '.' >> +qi::digit >> '.' >> +qi::digit) |
71  (+qi::digit >> '.' >> +qi::digit));
72  // parse input
73  auto first = versionString.begin();
74  auto last = versionString.end();
75  bool r = qi::phrase_parse(first,
76  last,
77  (('v' >> numbers) | numbers),
78  ascii::space);
79  // return true if parser succeeded
80  return (first == last && r);
81 }
