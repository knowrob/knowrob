67 std::vector<std::string> Collection::distinctValues(std::string_view key) {
68  bson_error_t err;
69  bson_t reply;
70  bson_t *command = BCON_NEW("distinct", BCON_UTF8(name_.c_str()), "key", BCON_UTF8(key.data()));
71  bool success = mongoc_database_command_simple(
72  db_, command, nullptr, &reply, &err);
73  if (!success) {
74  bson_destroy(command);
75  bson_destroy(&reply);
76  throw MongoException("distinct_failed", err);
77  }
78 
79  std::vector<std::string> values;
80  bson_iter_t iter;
81  if (bson_iter_init(&iter, &reply)) {
82  const bson_value_t *value;
83  if (bson_iter_find(&iter, "values") && BSON_ITER_HOLDS_ARRAY(&iter)) {
84  bson_iter_t array;
85  bson_iter_recurse(&iter, &array);
86  while (bson_iter_next(&array)) {
87  value = bson_iter_value(&array);
88  values.emplace_back(value->value.v_utf8.str);
89  }
90  }
91  }
92  bson_destroy(command);
93  bson_destroy(&reply);
94  return values;
95 }
