153  std::shared_ptr<PluginLibrary<T>> loadSharedLibrary(std::string_view path) {
154  auto absPath = std::filesystem::absolute(path);
155  auto it = loadedPlugins_.find(absPath);
156  if (it == loadedPlugins_.end()) {
157  auto p = std::make_shared<PluginLibrary<T>>(absPath.c_str());
158  auto jt = loadedPlugins_.insert(std::pair<std::string,
159  std::shared_ptr<PluginLibrary<T>>>(absPath, p));
160  if (jt.first->second->loadDLL()) {
161  return jt.first->second;
162  }
163  } else if (it->second->isLoaded()) {
164  return it->second;
165  }
166  KB_WARN("Failed to open plugin library at path '{}'.", path);
167  return {};
168  }

186  std::shared_ptr<PluginFactory<T>> findFactory(const boost::property_tree::ptree &config) {
187  auto lib = config.get_optional<std::string>("lib");
188  auto module = config.get_optional<std::string>("module");
189  auto type = config.get_optional<std::string>("type");
190 
191  if (lib.has_value()) {
192  return loadSharedLibrary(lib.value());
193  } else if (module.has_value()) {
194  if (type.has_value()) {
195  return loadPythonModule(module.value(), type.value());
196  } else {
197  KB_WARN("modules require type key in settings, but it's missing for module '{}'.", module.value());
198  }
199  } else if (type.has_value()) {
200  // map type name to a factory
201  const auto &it = pluginFactories().find(type.value());
202  if (it == pluginFactories().end()) {
203  KB_WARN("no factory registered for plugin type '{}'.", type.value());
204  } else {
205  return it->second;
206  }
207  } else {
208  KB_WARN("missing 'type', 'lib' and 'module' key in plugin config.");
209  }
210  return {};
211  }
