22 void MongoTaxonomy::update(
23  const std::vector<StringPair> &subClassAssertions,
24  const std::vector<StringPair> &subPropertyAssertions) {
25  // below performs the server-side data transformation for updating hierarchy relations
26  // such as rdf::type.
27  // However, there are many steps for large ontologies so this might consume some time.
28  // TODO: list of parents could be supplied as a constant in aggregation queries below.
29  // currently parents are computed in the query, maybe it would be a bit faster using a constant
30  // baked into the query.
31 
32  bson_t pipelineDoc = BSON_INITIALIZER;
33 
34  // update class hierarchy.
35  // unfortunately must be done step-by-step as it is undefined yet in mongo
36  // if it's possible to access $merge results in following pipeline iterations via e.g. $lookup.
37  for (auto &assertion: subClassAssertions) {
38  bson_reinit(&pipelineDoc);
39 
40  bson_t pipelineArray;
41  BSON_APPEND_ARRAY_BEGIN(&pipelineDoc, "pipeline", &pipelineArray);
42  Pipeline pipeline(&pipelineArray);
43  updateHierarchyO(pipeline,
44  tripleCollection_->name(),
45  rdfs::subClassOf->stringForm(),
46  assertion.first,
47  assertion.second);
48  bson_append_array_end(&pipelineDoc, &pipelineArray);
49 
50  oneCollection_->evalAggregation(&pipelineDoc);
51  }
52 
53  // update property hierarchy.
54  // unfortunately must be done step-by-step as it is undefined yet in mongo
55  // if it's possible to access $merge results in following pipeline iterations via e.g. $lookup.
56  std::set<std::string_view> visited;
57  for (auto &assertion: subPropertyAssertions) {
58  visited.insert(assertion.first);
59  bson_reinit(&pipelineDoc);
60 
61  bson_t pipelineArray;
62  BSON_APPEND_ARRAY_BEGIN(&pipelineDoc, "pipeline", &pipelineArray);
63  Pipeline pipeline(&pipelineArray);
64  updateHierarchyO(pipeline,
65  tripleCollection_->name(),
66  rdfs::subPropertyOf->stringForm(),
67  assertion.first,
68  assertion.second);
69  bson_append_array_end(&pipelineDoc, &pipelineArray);
70 
71  oneCollection_->evalAggregation(&pipelineDoc);
72  }
73 
74  // update property assertions
75  // TODO: below steps are independent, and could run in parallel.
76  // could bake an array of properties into pipeline,
77  // or rather use a bulk operation.
78  for (auto &newProperty: visited) {
79  bson_reinit(&pipelineDoc);
80 
81  bson_t pipelineArray;
82  BSON_APPEND_ARRAY_BEGIN(&pipelineDoc, "pipeline", &pipelineArray);
83  Pipeline pipeline(&pipelineArray);
84  updateHierarchyP(pipeline,
85  tripleCollection_->name(),
86  rdfs::subPropertyOf->stringForm(),
87  newProperty);
88  bson_append_array_end(&pipelineDoc, &pipelineArray);
89 
90  oneCollection_->evalAggregation(&pipelineDoc);
91  }
92 
93  bson_destroy(&pipelineDoc);
94 }

96 void MongoTaxonomy::lookupParents(
97  Pipeline &pipeline,
98  const std::string_view &collection,
99  const std::string_view &entity,
100  const std::string_view &relation) {
101  // lookup parent hierarchy.
102  // e.g. for subClassOf these are all o* values of subClassOf documents of entity
103  bson_t lookupArray;
104  auto lookupStage = pipeline.appendStageBegin("$lookup");
105  BSON_APPEND_UTF8(lookupStage, "from", collection.data());
106  BSON_APPEND_UTF8(lookupStage, "as", "directParents");
107  BSON_APPEND_ARRAY_BEGIN(lookupStage, "pipeline", &lookupArray);
108  {
109  Pipeline lookupPipeline(&lookupArray);
110  // { $match: { s: $entity, p: $relation } }
111  auto matchStage = lookupPipeline.appendStageBegin("$match");
112  BSON_APPEND_UTF8(matchStage, "s", entity.data());
113  BSON_APPEND_UTF8(matchStage, "p", relation.data());
114  lookupPipeline.appendStageEnd(matchStage);
115  // { $project: { "o*": 1 } }
116  lookupPipeline.project("o*");
117  // { $unwind: "$o*" }
118  lookupPipeline.unwind("$o*");
119  }
120  bson_append_array_end(lookupStage, &lookupArray);
121  pipeline.appendStageEnd(lookupStage);
122 
123  // convert "parents" field from list of documents to list of strings:
124  // { $set { parents: { $map: { input: "$parents", "in": "$$this.o*" } } } }
125  bson_t parentsDoc, mapDoc;
126  auto setStage = pipeline.appendStageBegin("$set");
127  BSON_APPEND_DOCUMENT_BEGIN(setStage, "directParents", &parentsDoc);
128  {
129  BSON_APPEND_DOCUMENT_BEGIN(&parentsDoc, "$map", &mapDoc);
130  {
131  BSON_APPEND_UTF8(&mapDoc, "input", "$directParents");
132  BSON_APPEND_UTF8(&mapDoc, "in", "$$this.o*");
133  }
134  bson_append_document_end(&parentsDoc, &mapDoc);
135  }
136  bson_append_document_end(setStage, &parentsDoc);
137  pipeline.appendStageEnd(setStage);
138 }

140 void MongoTaxonomy::updateHierarchyO(
141  Pipeline &pipeline,
142  const std::string_view &collection,
143  const std::string_view &relation,
144  const std::string_view &newChild,
145  const std::string_view &newParent) {
146  // lookup hierarchy into array field "parents"
147  lookupParents(pipeline, collection, newParent, relation);
148  // add newParent to parents array:
149  // { $set: { parents: { $concatArrays: [ "$parents", [$newParent] ] } } }
150  pipeline.addToArray("directParents", "$directParents", newParent);
151 
152  // lookup documents that include the child in the p* field
153  bson_t lookupArray;
154  auto lookupStage = pipeline.appendStageBegin("$lookup");
155  BSON_APPEND_UTF8(lookupStage, "from", collection.data());
156  BSON_APPEND_UTF8(lookupStage, "as", "doc");
157  BSON_APPEND_ARRAY_BEGIN(lookupStage, "pipeline", &lookupArray);
158  {
159  Pipeline lookupPipeline(&lookupArray);
160  // { $match: { "o*": $newChild } }
161  auto matchStage = lookupPipeline.appendStageBegin("$match");
162  BSON_APPEND_UTF8(matchStage, "o*", newChild.data());
163  lookupPipeline.appendStageEnd(matchStage);
164  // { $project: { "o*": 1 } }
165  lookupPipeline.project("o*");
166  }
167  bson_append_array_end(lookupStage, &lookupArray);
168  pipeline.appendStageEnd(lookupStage);
169  pipeline.unwind("$doc");
170 
171  // add parents to the doc.o* field
172  // { $set: { "doc.o*": { $setUnion: [ "$doc.o*", "$parents" ] } } }
173  pipeline.setUnion("doc.o*", {"$doc.o*", "$directParents"});
174  // make the "doc" field the new root
175  pipeline.replaceRoot("$doc");
176  // merge the result into the collection
177  pipeline.merge(collection);
178 }
