53  bool loadModule() {
54  // try to make sure that the module can be imported.
55  // the modules can be addressed either via an absolute path,
56  // or via a relative path in the KnowRob project.
57  std::filesystem::path modulePath(modulePath_);
58  if (!std::filesystem::exists(modulePath)) {
59  KB_ERROR("Module '{}' does not exist.", modulePath_.c_str());
60  return false;
61  }
62 
63  try {
64  // Make sure module can be imported by possibly extending the sys.path
65  // of the Python interpreter.
66  auto importString = knowrob::py::addToSysPath(modulePath);
67  pyModule_ = boost::python::import(importString.c_str());
68  if (pyModule_.is_none()) {
69  KB_ERROR("Failed to import module '{}'.", modulePath_.c_str());
70  return false;
71  }
72  pyPluginType_ = pyModule_.attr(pluginType_.c_str());
73  } catch (const boost::python::error_already_set &) {
74  throw PythonError();
75  }
