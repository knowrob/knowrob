37 void TokenStream::push(Channel &channel, const TokenPtr &tok) {
38  if (tok->indicatesEndOfEvaluation()) {
39  bool doPushMsg = true;
40  if (isOpened()) {
41  if (channel.hasValidIterator()) {
42  // prevent channels from being created while processing EOS message
43  std::lock_guard<std::mutex> lock(channel_mutex_);
44  // close this stream if no channels are left
45  channels_.erase(channel.iterator_);
46  channel.invalidateIterator();
47  doPushMsg = channels_.empty();
48  } else {
49  KB_WARN("ignoring attempt to write to a channel with a singular iterator.");
50  }
51  }
52  // send EOS on this stream if no channels are left
53  if (doPushMsg) {
54  push(tok);
55  isOpened_ = false;
56  }
57  } else if (!isOpened()) {
58  KB_WARN("ignoring attempt to write to a closed stream.");
59  } else {
60  push(tok);
61  }
62 }
