94 std::shared_ptr<AnswerYes> QueryableBackend::yes(const GraphQueryExpansionPtr &expanded, const BindingsPtr &bindings) {
95  static const auto edbTerm = Atom::Tabled("EDB");
96 
97  auto positiveAnswer = std::make_shared<AnswerYes>(bindings);
98  // Indicate that EDB has computed the grounding.
99  positiveAnswer->setReasonerTerm(edbTerm);
100  // Apply query context to the answer for some parameters.
101  positiveAnswer->applyFrame(expanded->original->ctx()->selector);
102 
103  // Add predicate groundings to the answer
104  for (auto &rdfLiteral: expanded->original->path()) {
105  auto p = rdfLiteral->predicate();
106  auto p_instance = applyBindings(p, *positiveAnswer->substitution());
107  positiveAnswer->addGrounding(
108  std::static_pointer_cast<Predicate>(p_instance),
109  positiveAnswer->frame(),
110  rdfLiteral->isNegated());
111  }
112 
113  // The answer is uncertain if any of the groundings is uncertain.
114  positiveAnswer->setIsUncertain(
115  std::any_of(expanded->u_vars.begin(), expanded->u_vars.end(), [&](const VariablePtr &v) {
116  auto &u_v = bindings->get(v->name());
117  return (u_v && u_v->isNumeric() && std::static_pointer_cast<Numeric>(u_v)->asBoolean());
118  }), std::nullopt);
119 
120  // The answer is occasional if any of the groundings has occasional=true flag
121  positiveAnswer->setIsOccasionallyTrue(
122  std::any_of(expanded->o_vars.begin(), expanded->o_vars.end(), [&](const VariablePtr &v) {
123  auto &o_v = bindings->get(v->name());
124  return (o_v && o_v->isNumeric() && std::static_pointer_cast<Numeric>(o_v)->asBoolean());
125  }));
126 
127  return positiveAnswer;
128 }

130 std::shared_ptr<AnswerNo> QueryableBackend::no(const GraphPathQueryPtr &q) {
131  static const auto edbTerm = Atom::Tabled("EDB");
132 
133  // send one negative answer if no positive answer was found
134  auto negativeAnswer = std::make_shared<AnswerNo>();
135  negativeAnswer->setReasonerTerm(edbTerm);
136  // Apply query context "origin" and "perspective" to the answer if any
137  negativeAnswer->applyFrame(q->ctx()->selector);
138  // the answer is uncertain as we only were not able to obtain a positive answer
139  // which does not mean that there is no positive answer.
140  negativeAnswer->setIsUncertain(true, std::nullopt);
141 
142  // Add ungrounded literals to negative answer.
143  // But at the moment the information is not provided by EDBs, would be difficult to implement e.g. in MongoDB.
144  // Well at least we know if the query is only a single triple pattern.
145  if (q->path().size() == 1) {
146  negativeAnswer->addUngrounded(q->path().front()->predicate(),
147  q->path().front()->isNegated());
148  }
149  return negativeAnswer;
150 }

152 static std::shared_ptr<GraphTerm>
153 expand_pattern(const std::shared_ptr<GraphPattern> &q, GraphQueryExpansion &ctx) {
154  const auto &p = q->value();
155  ctx.counter += 1;
156 
157  // Find out if an additional variable must be added for the "isUncertain" flag.
158  // If a variable exists, rather use the existing one.
159  bool needsUncertainVar = false;
160  if (p->isUncertainTerm()) {
161  if (**p->isUncertainTerm() == *Numeric::trueAtom()) {
162  // For each possibly uncertain triple, add a variable to the query for the isUncertain flag.
163  needsUncertainVar = true;
164  } else if (p->isUncertainTerm()->isVariable()) {
165  // In case uncertain term is already a variable, remember it.
166  ctx.u_vars.emplace_back(std::static_pointer_cast<Variable>(*p->isUncertainTerm()));
167  }
168  }
169 
170  // Find out if an additional variable must be added for the "isOccasional" flag.
171  // If a variable exists, rather use the existing one.
172  bool needsOccasionalVar = false;
173  if (p->isOccasionalTerm()) {
174  if (**p->isOccasionalTerm() == *Numeric::trueAtom()) {
175  // For each possibly occasional triple, add a variable to the query for the isOccasional flag.
176  needsOccasionalVar = true;
177  } else if (p->isOccasionalTerm()->isVariable()) {
178  // In case occasional term is already a variable, remember it.
179  ctx.o_vars.emplace_back(std::static_pointer_cast<Variable>(*p->isOccasionalTerm()));
180  }
181  }
182 
183  // Find out if begin/end variables must be added for the computation of the time interval.
184  // This is always the case if the query uses SOMETIMES operator.
185  bool needsIntervalComputation = (ctx.query_ctx->selector.occasional);
186 
187  bool needsRewrite = needsUncertainVar || needsOccasionalVar || needsIntervalComputation;
188  if (!needsRewrite) return q;
189 
190  auto pat_expanded = std::make_shared<FramedTriplePattern>(*p);
191 
192  if (needsUncertainVar) {
193  static const std::string varPrefix = "_uncertain";
194  // insert a fresh variable for the isUncertain flag
195  auto u_var = std::make_shared<Variable>(varPrefix + std::to_string(ctx.counter));
196  pat_expanded->setIsUncertainTerm(groundable<Numeric>(u_var));
197  // also remember the variable in the context
198  ctx.u_vars.emplace_back(u_var);
199  }
200 
201  if (needsOccasionalVar) {
202  static const std::string varPrefix = "_occasional";
203  // insert a fresh variable for the isOccasional flag
204  auto o_var = std::make_shared<Variable>(varPrefix + std::to_string(ctx.counter));
205  pat_expanded->setIsOccasionalTerm(groundable<Numeric>(o_var));
206  // also remember the variable in the context
207  ctx.o_vars.emplace_back(o_var);
208  }
209 
210  VariablePtr triple_begin, triple_end;
211  if (needsIntervalComputation) {
212  // Add begin/end variables possibly replacing fixed values for begin/end in the query.
213  // This is ok assuming each pattern has the same query frame.
214  // FILTER will be performed through a builtin instead, so deleting the values is ok,
215  // SPARQL would insert ad-hoc variables anyway.
216  auto g_begin = p->beginTerm();
217  auto g_end = p->endTerm();
218 
219  if (g_begin.has_variable()) {
220  triple_begin = g_begin.variable();
221  } else {
222  static const std::string varPrefix = "_begin";
223  triple_begin = std::make_shared<Variable>(varPrefix + std::to_string(ctx.counter));
224  pat_expanded->setBeginTerm(groundable<Double>(triple_begin));
225  }
226 
227  if (g_end.has_variable()) {
228  triple_end = g_end.variable();
229  } else {
230  static const std::string varPrefix = "_end";
231  triple_end = std::make_shared<Variable>(varPrefix + std::to_string(ctx.counter));
232  pat_expanded->setEndTerm(groundable<Double>(triple_end));
233  }
234  }
235 
236  auto q_expanded = std::make_shared<GraphPattern>(pat_expanded);
237  std::shared_ptr<GraphTerm> outer_term = q_expanded;
238 
239  // If the query uses SOMETIMES operator, then we need to insert builtins to the query for
240  // the computation of the time interval.
241  if (needsIntervalComputation) {
242  auto seq = std::make_shared<GraphSequence>();
243  seq->addMember(q_expanded);
244 
245  // FILTER all triples that do not intersect with begin/end of query frame
246  if (ctx.query_ctx->selector.begin) {
247  auto ctx_begin = std::make_shared<Double>(ctx.query_ctx->selector.begin.value());
248  seq->addMember(GraphBuiltin::lessOrEqual(ctx_begin, triple_end));
249  }
250  if (ctx.query_ctx->selector.end) {
251  auto ctx_end = std::make_shared<Double>(ctx.query_ctx->selector.end.value());
252  seq->addMember(GraphBuiltin::greaterOrEqual(ctx_end, triple_begin));
253  }
254 
255  // Set accumulated begin and end time.
256  // But not all backends support variable re-assignment.
257  // For the ones that don't support it like (SPARQL), we need to introduce a new variable for each intersection computed.
258  VariablePtr next_i_begin, next_i_end;
259  if (ctx.with_reassignment) {
260  next_i_begin = ctx.accumulated_begin;
261  next_i_end = ctx.accumulated_end;
262  } else {
263  static const std::string varPrefix_begin = "_i_begin";
264  static const std::string varPrefix_end = "_i_end";
265  next_i_begin = std::make_shared<Variable>(varPrefix_begin + std::to_string(ctx.counter));
266  next_i_end = std::make_shared<Variable>(varPrefix_end + std::to_string(ctx.counter));
267  }
268  seq->addMember(GraphBuiltin::max(next_i_begin, ctx.accumulated_begin, triple_begin));
269  seq->addMember(GraphBuiltin::min(next_i_end, ctx.accumulated_end, triple_end));
270  ctx.accumulated_begin = next_i_begin;
271  ctx.accumulated_end = next_i_end;
272 
273  // Ensure that begin < end
274  seq->addMember(GraphBuiltin::less(ctx.accumulated_begin, ctx.accumulated_end));
275 
276  // Update outer term
277  outer_term = seq;
278  }
279 
280  return outer_term;
281 }

283 static std::shared_ptr<GraphTerm>
284 expand_term(const std::shared_ptr<GraphTerm> &q, GraphQueryExpansion &ctx) { // NOLINT
285  switch (q->termType()) {
286  case GraphTermType::Pattern:
287  return expand_pattern(std::static_pointer_cast<GraphPattern>(q), ctx);
288  case GraphTermType::Union:
289  case GraphTermType::Sequence: {
290  auto connective = std::static_pointer_cast<GraphConnective>(q);
291  std::vector<std::shared_ptr<GraphTerm>> expandedTerms(connective->terms().size());
292 
293  bool hasExpansion = false;
294  for (size_t i = 0; i < connective->terms().size(); ++i) {
295  auto expandedTerm = expand_term(connective->terms()[i], ctx);
296  expandedTerms[i] = expandedTerm;
297  hasExpansion = hasExpansion || (expandedTerm != connective->terms()[i]);
298  }
299 
300  if (hasExpansion) {
301  return std::make_shared<GraphSequence>(expandedTerms);
302  }
303  break;
304  }
305  case GraphTermType::Builtin:
306  break;
307  }
308  return q;
309 }

311 static GraphQueryPtr expand_query(const GraphQueryPtr &q, GraphQueryExpansion &ctx) {
312  // Initialize begin/end variables for the computation of the time interval.
313  static const auto var_begin = std::make_shared<Variable>("_begin");
314  static const auto var_end = std::make_shared<Variable>("_end");
315  ctx.accumulated_begin = var_begin;
316  ctx.accumulated_end = var_end;
317 
318  // Expand the query
319  auto expandedTerm = expand_term(q->term(), ctx);
320 
321  // If the query uses SOMETIMES operator, prepend initialization of the accumulated_begin and accumulated_end variables
322  // used to compute the intersection of triple time intervals.
323  if (ctx.query_ctx->selector.occasional) {
324  double b_min = ctx.query_ctx->selector.begin.value_or(0.0);
325  double e_max = ctx.query_ctx->selector.end.value_or(std::numeric_limits<double>::max());
326  auto set_b = GraphBuiltin::bind(var_begin, std::make_shared<Double>(b_min));
327  auto set_e = GraphBuiltin::bind(var_end, std::make_shared<Double>(e_max));
328 
329  if (expandedTerm->termType() == GraphTermType::Sequence) {
330  // Prepend to existing sequence
331  auto seq_terms = std::static_pointer_cast<GraphSequence>(expandedTerm)->terms();
332  seq_terms.insert(seq_terms.begin(), set_e);
333  seq_terms.insert(seq_terms.begin(), set_b);
334  expandedTerm = std::make_shared<GraphSequence>(seq_terms);
335  } else {
336  // Create a new sequence
337  auto seq = std::make_shared<GraphSequence>();
338  seq->addMember(set_b);
339  seq->addMember(set_e);
340  seq->addMember(expandedTerm);
341  expandedTerm = seq;
342  }
343  }
344 
345  if (expandedTerm == q->term()) {
346  return q;
347  } else {
348  return std::make_shared<GraphQuery>(expandedTerm, q->ctx());
349  }
350 }
