175 TokenBufferPtr PrologReasoner::submitQuery(const FramedTriplePatternPtr &literal, const QueryContextPtr &ctx) {
176  // context term options:
177  static const auto query_scope_f = "query_scope";
178  static const auto solution_scope_f = "solution_scope";
179  static const auto triple_f = "triple";
180 
181  auto answerBuffer = std::make_shared<TokenBuffer>();
182  auto outputChannel = TokenStream::Channel::create(answerBuffer);
183 
184  // create runner that evaluates the goal in a thread with a Prolog engine
185  auto runner = std::make_shared<ThreadPool::LambdaRunner>(
186  [&](const ThreadPool::LambdaRunner::StopChecker &hasStopRequest) {
187  PrologTerm queryFrame, answerFrame;
188  putQueryFrame(queryFrame, ctx->selector);
189 
190  PrologTerm rdfGoal(*literal, triple_f);
191  // :- ContextTerm = [query_scope(...), solution_scope(Variable)]
192  PrologList contextTerm({
193  PrologTerm(query_scope_f, queryFrame),
194  PrologTerm(solution_scope_f, answerFrame)
195  });
196  // :- QueryGoal = ( b_setval(reasoner_module, reasonerName()),
197  // b_setval(reasoner_manager, managerID),
198  // reasoner_call(LiteralGoal, ContextTerm) ).
199  PrologTerm queryGoal = getReasonerQuery(
200  PrologTerm(callFunctor(), rdfGoal, contextTerm));
201 
202  auto qid = queryGoal.openQuery(prologQueryFlags);
203  bool hasSolution = false;
204  while (!hasStopRequest() && queryGoal.nextSolution(qid)) {
205  outputChannel->push(yes(literal, rdfGoal, answerFrame));
206  hasSolution = true;
207  if (ctx->queryFlags & QUERY_FLAG_ONE_SOLUTION) break;
208  }
209  PL_close_query(qid);
210  if (!hasSolution) outputChannel->push(no(literal));
211  outputChannel->push(EndOfEvaluation::get());
212  });
213 
214  // push goal and return
215  PrologEngine::pushGoal(runner, [literal, outputChannel](const std::exception &e) {
216  KB_WARN("an exception occurred for prolog query ({}): {}.", *literal, e.what());
217  outputChannel->close();
218  });
219 
220  return answerBuffer;
221 }

223 AnswerYesPtr PrologReasoner::yes(const FramedTriplePatternPtr &literal,
224  const PrologTerm &rdfGoal,
225  const PrologTerm &answerFrameTerm) {
226  KB_DEBUG("Prolog has a next solution.");
227 
228  // create bindings object
229  auto bindings = std::make_shared<Bindings>();
230  for (const auto &kv: rdfGoal.vars()) {
231  auto grounding = PrologTerm::toKnowRobTerm(kv.second);
232  if (grounding && grounding->termType() != TermType::VARIABLE) {
233  bindings->set(std::make_shared<Variable>(kv.first), PrologTerm::toKnowRobTerm(kv.second));
234  }
235  }
236 
237  // create an answer object given the bindings
238  auto yes = std::make_shared<AnswerYes>(bindings);
239  yes->setReasonerTerm(reasonerName());
240 
241  // set the solution scope, if reasoner specified it
242  auto answerFrame_rw = createAnswerFrame(answerFrameTerm);
243  GraphSelectorPtr answerFrame_ro;
244  if (answerFrame_rw) {
245  yes->setFrame(answerFrame_rw);
246  answerFrame_ro = answerFrame_rw;
247  } else {
248  answerFrame_ro = DefaultGraphSelector();
249  }
250 
251  // store instantiation of literal
252  auto p = literal->predicate();
253  auto p_instance = applyBindings(p, *yes->substitution());
254  yes->addGrounding(std::static_pointer_cast<Predicate>(p_instance), answerFrame_ro, literal->isNegated());
255 
256  return yes;
257 }

259 AnswerNoPtr PrologReasoner::no(const FramedTriplePatternPtr &rdfLiteral) {
260  KB_DEBUG("Prolog has no solution.");
261  // if no solution was found, indicate that via a NegativeAnswer.
262  auto negativeAnswer = std::make_shared<AnswerNo>();
263  negativeAnswer->setReasonerTerm(reasonerName());
264  // however, as Prolog cannot proof negations such an answer is always not well-founded
265  // and can be overruled by a well-founded one.
266  negativeAnswer->setIsUncertain(true, std::nullopt);
267  // as the query goal was only a single literal, we know that exactly this literal cannot be proven.
268  negativeAnswer->addUngrounded(
269  std::static_pointer_cast<Predicate>(rdfLiteral->predicate()),
270  rdfLiteral->isNegated());
271  return negativeAnswer;
272 }

275 bool PrologReasoner::putQueryFrame(PrologTerm &frameTerm, const GraphSelector &frame) {
276  // Map an GraphSelector to a Prolog dict term.
277  // The term has the form:
278  // { epistemicMode: knowledge|belief,
279  // temporalMode: sometimes|always,
280  // [agent: $name,]
281  // [since: $time,]
282  // [until: $time] }
283 
284  int numFrameKeys = 2;
285  if (frame.perspective.has_value()) numFrameKeys += 1;
286  if (frame.begin.has_value()) numFrameKeys += 1;
287  if (frame.end.has_value()) numFrameKeys += 1;
288 
289  // option: frame($dictTerm)
290  atom_t scopeKeys[numFrameKeys];
291  auto scopeValues = PL_new_term_refs(numFrameKeys);
292 
293  // epistemicMode: knowledge|belief
294  static const auto epistemicMode_a = PL_new_atom("epistemicMode");
295  static const auto knowledge_a = PL_new_atom("knowledge");
296  static const auto belief_a = PL_new_atom("belief");
297 
298  bool isAboutBelief = (frame.uncertain);
299  int keyIndex = 0;
300  scopeKeys[keyIndex] = epistemicMode_a;
301  if (!PL_put_atom(scopeValues, isAboutBelief ? belief_a : knowledge_a)) return false;
302 
303  // temporalMode: sometimes|always
304  static const auto temporalMode_a = PL_new_atom("temporalMode");
305  static const auto sometimes_a = PL_new_atom("sometimes");
306  static const auto always_a = PL_new_atom("always");
307 
308  bool isAboutSomePast = (frame.occasional);
309  scopeKeys[++keyIndex] = temporalMode_a;
310  if (!PL_put_atom(scopeValues + keyIndex, isAboutSomePast ? sometimes_a : always_a)) return false;
311 
312  // agent: $name
313  if (frame.perspective.has_value()) {
314  static const auto agent_a = PL_new_atom("agent");
315  scopeKeys[++keyIndex] = agent_a;
316  auto agent_iri = frame.perspective.value()->iri();
317  if (!PL_put_atom_chars(scopeValues + keyIndex, agent_iri.data())) return false;
318  }
319 
320  // since: $name
321  if (frame.begin.has_value()) {
322  static const auto since_a = PL_new_atom("since");
323  scopeKeys[++keyIndex] = since_a;
324  if (!PL_put_float(scopeValues + keyIndex, frame.begin.value())) return false;
325  }
326  // until: $name
327  if (frame.end.has_value()) {
328  static const auto until_a = PL_new_atom("until");
329  scopeKeys[++keyIndex] = until_a;
330  if (!PL_put_float(scopeValues + keyIndex, frame.end.value())) return false;
331  }
332 
333  return PL_put_dict(frameTerm(), 0, numFrameKeys, scopeKeys, scopeValues);
334 }

336 std::shared_ptr<GraphSelector> PrologReasoner::createAnswerFrame(const PrologTerm &plTerm) {
337  std::shared_ptr<GraphSelector> frame;
338 
339  if (PL_is_variable(plTerm())) {
340  // reasoner did not specify a solution scope
341  return {};
342  } else if (PL_is_dict(plTerm())) {
343  // the solution scope was generated as a Prolog dictionary
344  auto scope_val = PL_new_term_ref();
345 
346  frame = std::make_shared<GraphSelector>();
347 
348  // read "time" key, the value is expected to be a predicate `range(Since,Until)`
349  static const auto time_key = PL_new_atom("time");
350  if (PL_get_dict_key(time_key, plTerm(), scope_val)) {
351  term_t arg = PL_new_term_ref();
352  std::optional<TimePoint> v_since, v_until;
353  double val = 0.0;
354 
355  if (PL_get_arg(1, scope_val, arg) &&
356  PL_term_type(arg) == PL_FLOAT &&
357  PL_get_float(arg, &val) &&
358  val > 0.001) { frame->begin = val; }
359 
360  if (PL_get_arg(2, scope_val, arg) &&
361  PL_term_type(arg) == PL_FLOAT &&
362  PL_get_float(arg, &val)) { frame->end = val; }
363  }
364 
365  static const auto uncertain_key = PL_new_atom("uncertain");
366  if (PL_get_dict_key(uncertain_key, plTerm(), scope_val)) {
367  atom_t flagAtom;
368  if (PL_term_type(scope_val) == PL_ATOM && PL_get_atom(scope_val, &flagAtom)) {
369  if (flagAtom == PrologTerm::ATOM_true()) {
370  frame->uncertain = true;
371  }
372  }
373  }
374 
375  static const auto confidence_key = PL_new_atom("confidence");
376  if (PL_get_dict_key(confidence_key, plTerm(), scope_val)) {
377  double confidenceValue = 1.0;
378  if (PL_term_type(scope_val) == PL_FLOAT && PL_get_float(scope_val, &confidenceValue)) {
379  frame->confidence = confidenceValue;
380  if (confidenceValue > 0.999) {
381  frame->uncertain = false;
382  } else {
383  frame->uncertain = true;
384  }
385  }
386  }
387 
388  static const auto occasional_key = PL_new_atom("occasional");
389  if (PL_get_dict_key(occasional_key, plTerm(), scope_val)) {
390  atom_t flagAtom;
391  if (PL_term_type(scope_val) == PL_ATOM && PL_get_atom(scope_val, &flagAtom)) {
392  frame->occasional = (flagAtom == PrologTerm::ATOM_true());
393  }
394  }
395 
396  static const auto agent_key = PL_new_atom("agent");
397  if (PL_get_dict_key(agent_key, plTerm(), scope_val)) {
398  atom_t agentAtom;
399  if (PL_term_type(scope_val) == PL_ATOM && PL_get_atom(scope_val, &agentAtom)) {
400  frame->perspective = Perspective::get(PL_atom_chars(agentAtom));
401  }
402  }
403 
404  PL_reset_term_refs(scope_val);
405  } else {
406  KB_WARN("solution scope has an unexpected type (should be dict).");
407  }
408 
409  if (frame) {
410  return frame;
411  } else {
412  return {};
413  }
414 }
