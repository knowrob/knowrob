71 std::string FramedTriple::createStringValue() const {
72  static const auto a_true = "true";
73  static const auto a_false = "false";
74 
75  if (isObjectIRI() || isObjectBlank()) {
76  return std::string(valueAsString());
77  }
78 
79  if (xsdType()) {
80  switch (xsdType().value()) {
81  case XSDType::DOUBLE: {
82  std::ostringstream os;

85  }

94  case XSDType::LONG:
95  return std::to_string(valueAsLong());
96  case XSDType::SHORT:
97  return std::to_string(valueAsShort());
98  case XSDType::UNSIGNED_LONG:
99  return std::to_string(valueAsUnsignedLong());
100  case XSDType::UNSIGNED_INT:
101  return std::to_string(valueAsUnsignedInt());
102  case XSDType::UNSIGNED_SHORT:
103  return std::to_string(valueAsUnsignedShort());
104  case XSDType::BOOLEAN:

106  case XSDType::STRING:
107  return std::string(valueAsString());

284 bool FramedTriple::mergeFrame(const FramedTriple &other) {
285  bool sameBegin = begin() == other.begin();
286  bool sameEnd = end() == other.end();
287  bool sameTime = sameBegin && sameEnd;
288 
289  if (sameTime) {
290  // occasional can switch to always if both triples have the same time
291  setIsOccasional(isOccasional() && other.isOccasional());
292  // same for uncertain vs. certain
293  setIsUncertain(isUncertain() && other.isUncertain());
294  // plus we can take the max confidence
295  if (other.confidence().has_value() && confidence().has_value()) {
296  setConfidence(std::max(confidence().value(), other.confidence().value()));
297  } else if (!other.confidence().has_value()) {
298  confidence_ = std::nullopt;
299  }
300  } else {
301  // either both triples must be occasional or neither
302  if (isOccasional() != other.isOccasional()) return false;
303  // same for uncertain vs. certain
304  if (isUncertain() != other.isUncertain()) return false;
305  // also confidence must match in order to merge different time frames
306  if (confidence() != other.confidence()) return false;
307  // finally we can merge time frame
308  if (isOccasional()) {
309  if (begin() && other.begin()) {
310  setBegin(std::max(begin().value(), other.begin().value()));
311  } else if (other.begin()) {
312  setBegin(other.begin().value());
313  }
314  if (end() && other.end()) {
315  setEnd(std::min(end().value(), other.end().value()));
316  } else if (other.end()) {
317  setEnd(other.end().value());
318  }
319  } else {
320  if (begin() && other.begin()) {
321  setBegin(std::min(begin().value(), other.begin().value()));
322  } else if (!other.begin().has_value()) {
323  begin_ = std::nullopt;
324  }
325  if (end() && other.end()) {
326  setEnd(std::max(end().value(), other.end().value()));
327  } else if (!other.end().has_value()) {
328  end_ = std::nullopt;
329  }
330  }
331  }
332  return true;
333 }
