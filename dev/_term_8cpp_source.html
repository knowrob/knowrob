46 bool Term::operator==(const Term &other) const {
47  if (this == &other) return true;
48  if (termType() != other.termType()) return false;
49  switch (termType()) {
50  case TermType::ATOMIC:
51  return ((Atomic *) this)->isSameAtomic(*((Atomic *) &other));
52  case TermType::VARIABLE:
53  return ((Variable *) this)->isSameVariable(*((Variable *) &other));
54  case TermType::FUNCTION:
55  return ((Function *) this)->isSameFunction(*((Function *) &other));
56  }
57  return false;
58 }
