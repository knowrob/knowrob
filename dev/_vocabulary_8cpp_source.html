141 PropertyPtr Vocabulary::defineProperty(const std::shared_ptr<Property> &p) {
142  definedProperties_[p->iri()] = p;
143  definedReifications_[p->reification()->iri()] = p;
144  // define the reification of the property as a concept which inherits knowrob:ReifiedRelation
145  // note further superclasses will be added through Property::addDirectParent
146  auto reification = p->reification();
147  definedClasses_[reification->iri()] = reification;
148  reification->addDirectParent(defineClass(reification::ReifiedRelation->stringForm()));
149  return p;
150 }

192 void Vocabulary::setFrequency(const std::string_view &iri, uint64_t frequency) {
193  if (frequency == 0) {
194  frequency_.erase(iri);
195  } else {
196  auto p_it = definedProperties_.find(iri);
197  if (p_it != definedProperties_.end()) {
198  frequency_[p_it->second->iri()] = frequency;
199  return;
200  }
201  auto c_it = definedClasses_.find(iri);
202  if (c_it != definedClasses_.end()) {
203  frequency_[c_it->second->iri()] = frequency;
204  return;
205  }
206  KB_WARN("cannot set frequency of unknown resource '{}'", iri);
207  }
208 }
