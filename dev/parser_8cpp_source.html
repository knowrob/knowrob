41 PlTerm to_prolog_pose(const std::string &frame, const urdf::Pose& p) {
42  //
43  PlTerm pos;
44  PlTail posl(pos);
45  posl.append(p.position.x);
46  posl.append(p.position.y);
47  posl.append(p.position.z);
48  posl.close();
49  //
50  PlTerm rot;
51  PlTail rotl(rot);
52  rotl.append(p.rotation.x);
53  rotl.append(p.rotation.y);
54  rotl.append(p.rotation.z);
55  rotl.append(p.rotation.w);
56  rotl.close();
57  //
58  PlTerm pose;
59  PlTail l(pose);
60  l.append(frame.c_str());
61  l.append(pos);
62  l.append(rot);
63  l.close();
64  return pose;
65 }

67 PlCompound to_prolog_geometry(urdf::GeometryConstSharedPtr geom) {
68  if (!geom)
69  throw PlException(PlCompound("urdf_error", PlTerm("null_geometry")));
70 
71  switch(geom->type) {
72  case urdf::Geometry::BOX: {
73  urdf::BoxConstSharedPtr box = urdf::dynamic_pointer_cast<const urdf::Box>(geom);
74  PlTermv dim_term(3);
75  dim_term[0] = box->dim.x;
76  dim_term[1] = box->dim.y;
77  dim_term[2] = box->dim.z;
78  return PlCompound("box", dim_term);
79  }
80  case urdf::Geometry::SPHERE: {
81  urdf::SphereConstSharedPtr sphere = urdf::dynamic_pointer_cast<const urdf::Sphere>(geom);
82  PlTermv dim_term(1);
83  dim_term[0] = sphere->radius;
84  return PlCompound("sphere", dim_term);
85  }
86  case urdf::Geometry::CYLINDER : {
87  urdf::CylinderConstSharedPtr cylinder = urdf::dynamic_pointer_cast<const urdf::Cylinder>(geom);
88  PlTermv dim_term(2);
89  dim_term[0] = cylinder->radius;
90  dim_term[1] = cylinder->length;
91  return PlCompound("cylinder", dim_term);
92  }
93  case urdf::Geometry::MESH : {
94  urdf::MeshConstSharedPtr mesh = urdf::dynamic_pointer_cast<const urdf::Mesh>(geom);
95  PlTermv dim_term(2);
96  dim_term[0] = mesh->filename.c_str();
97  PlTail scale(dim_term[1]);
98  scale.append(mesh->scale.x);
99  scale.append(mesh->scale.y);
100  scale.append(mesh->scale.z);
101  scale.close();
102  return PlCompound("mesh", dim_term);
103  }
104  default:
105  throw PlException(PlCompound("urdf_error",
106  PlTerm("unsupported_geometry")));
107  }
108 }

110 PlTerm to_prolog_material(const urdf::MaterialSharedPtr& material) {
111  PlTermv mat_term(1);
112  PlTail args(mat_term[0]);
113  if(material) {
114  if(!material->texture_filename.empty()) {
115  PlTermv file_term(1);
116  file_term[0] = material->texture_filename.c_str();
117  args.append(PlCompound("texture_file",file_term));
118  }
119  //
120  PlTermv col_term(1);
121  PlTail col_list(col_term[0]);
122  col_list.append(material->color.r);
123  col_list.append(material->color.g);
124  col_list.append(material->color.b);
125  col_list.append(material->color.a);
126  col_list.close();
127  args.append(PlCompound("rgba", col_term));
128  }
129  args.close();
130  return PlCompound("material", mat_term);
131 }
