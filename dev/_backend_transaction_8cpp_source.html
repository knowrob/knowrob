18 static void setReificationVariable( // NOLINT(misc-no-recursion)
19  const std::shared_ptr<GraphTerm> &t,
20  const VariablePtr &variable) {
21  switch (t->termType()) {
22  case knowrob::GraphTermType::Pattern: {
23  auto &pattern = std::static_pointer_cast<GraphPattern>(t)->value();
24  pattern->setSubjectTerm(variable);
25  break;
26  }
27  case knowrob::GraphTermType::Union:
28  case knowrob::GraphTermType::Sequence: {
29  auto connective = std::static_pointer_cast<GraphConnective>(t);
30  for (auto &term: connective->terms()) {
31  setReificationVariable(term, variable);
32  }
33  break;
34  }
35  case knowrob::GraphTermType::Builtin:
36  break;
37  };
38 }

66 bool Transaction::commit(const FramedTriple &triple, const IRIAtomPtr &reifiedName) {
67  ReifiedTriplePtr reification;
68  bool success = true;
69  for (auto &definedBackend: backends_) {
70  auto &backend = definedBackend->value();
71  if (!backend->supports(BackendFeature::TripleContext) && ReifiedTriple::isReifiable(triple)) {
72  if (!reification) reification = std::make_shared<ReifiedTriple>(triple, vocabulary_, reifiedName);
73  for (auto &reified: *reification) {
74  success = success && doCommit(*reified.ptr, backend);
75  }
76  } else {
77  success = doCommit(triple, backend);
78  }
79  if (!success) break;
80  }
81  if (success) {
82  updateVocabulary(triple);
83  }
84  return success;
85 }

87 bool Transaction::commit(const TripleContainerPtr &triples) {
88  static auto v_reification = std::make_shared<Variable>("reification");
89  if (isRemoval_ && !queryable_->supports(BackendFeature::TripleContext)) {
90  // FIXME: The name lookup only works in case the queryable backend does store reified triples.
91  // (1) we could search for a queryable backend without context support and use this instead, or
92  // (2) we could store the reified names also in backends that support context, such they
93  // can be queried from any queryable backend.
94  // Note: the container type does not provide a size method because it internally uses a generator
95  // without knowing when it will end. Also, container with additional filtering could be implemented.
96  // So we need to resize the reifiedNames vector while looping over the triples, but we can use
97  // the default batch size as initial size.
98  auto estimatedSize = GlobalSettings::batchSize();
99  ReifiedNames reifiedNames = std::make_shared<std::vector<IRIAtomPtr>>();
100  reifiedNames->reserve(estimatedSize);
101  for (auto &triple: *triples) {
102  if (ReifiedTriple::isReifiable(*triple)) {
103  reifiedNames->push_back(queryReifiedName(*triple));
104  } else {
105  reifiedNames->push_back(nullptr);
106  }
107  }
108  // If fewer elements were added, resize the vector
109  if (reifiedNames->size() < estimatedSize) {
110  reifiedNames->resize(reifiedNames->size());
111  }
112  return commit(triples, reifiedNames);
113  } else {
114  return commit(triples, {});
115  }
116 }

118 bool Transaction::commit(const TripleContainerPtr &triples, const ReifiedNames &reifiedNames) {
119  TripleContainerPtr reified;
120  std::vector<std::shared_ptr<ThreadPool::Runner>> transactions;
121  bool success = true;
122 
123  auto vocabWorker = createTripleWorker(triples,
124  [this](const FramedTriplePtr &triple) { updateVocabulary(*triple); });
125 
126  for (auto &definedBackend: backends_) {
127  auto &backend = definedBackend->value();
128  const TripleContainerPtr *backendTriples;
129  if (!backend->supports(BackendFeature::TripleContext)) {
130  if (!reified) reified = std::make_shared<ReificationContainer>(triples, vocabulary_, reifiedNames);
131  backendTriples = &reified;
132  } else {
133  backendTriples = &triples;
134  }
135  auto worker = std::make_shared<ThreadPool::LambdaRunner>(
136  [&](const std::function<bool()> &) { success = success && doCommit(*backendTriples, backend); });
137  transactions.push_back(worker);
138 
139  DefaultThreadPool()->pushWork(worker,
140  [&definedBackend](const std::exception &exc) {
141  KB_ERROR("transaction failed for backend '{}': {}", definedBackend->name(),
142  exc.what());
143  });
144  }
145 
146  for (auto &transaction: transactions) transaction->join();
147  vocabWorker->join();
148 
149  return success;
150 }

155  auto perTripleWorker =

182 void Insert::updateVocabulary(const FramedTriple &triple) {
183  // keep track of imports, subclasses, and subproperties
184  if (isSubClassOfIRI(triple.predicate())) {
185  auto sub = vocabulary_->defineClass(triple.subject());
186  auto sup = vocabulary_->defineClass(triple.valueAsString());
187  sub->addDirectParent(sup);
188  } else if (isSubPropertyOfIRI(triple.predicate())) {
189  auto sub = vocabulary_->defineProperty(triple.subject());
190  auto sup = vocabulary_->defineProperty(triple.valueAsString());
191  sub->addDirectParent(sup);
192  } else if (isTypeIRI(triple.predicate())) {
193  vocabulary_->addResourceType(triple.subject(), triple.valueAsString());
194  // increase frequency in vocabulary
195  static std::set<std::string_view> skippedTypes = {
196  owl::Class->stringForm(),
197  owl::Restriction->stringForm(),
198  owl::NamedIndividual->stringForm(),
199  owl::AnnotationProperty->stringForm(),
200  owl::ObjectProperty->stringForm(),
201  owl::DatatypeProperty->stringForm(),
202  rdfs::Class->stringForm(),
203  rdf::Property->stringForm()
204  };
205  if (vocabulary_->isDefinedClass(triple.valueAsString()) &&
206  !skippedTypes.count(triple.valueAsString())) {
207  vocabulary_->increaseFrequency(triple.valueAsString());
208  }
209  } else if (isInverseOfIRI(triple.predicate())) {
210  auto p = vocabulary_->defineProperty(triple.subject());
211  auto q = vocabulary_->defineProperty(triple.valueAsString());
212  p->setInverse(q);
213  q->setInverse(p);
214  } else if (owl::imports->stringForm() == triple.predicate()) {
215  auto resolvedImport = URI::resolve(triple.valueAsString());
216  auto importedGraph = DataSource::getNameFromURI(resolvedImport);
217  if (triple.graph()) {
218  vocabulary_->importHierarchy()->addDirectImport(triple.graph().value(), importedGraph);
219  } else {
220  KB_WARN("import statement without graph");
221  }
222  } else if (vocabulary_->isObjectProperty(triple.predicate()) ||
223  vocabulary_->isDatatypeProperty(triple.predicate())) {
224  // increase frequency of property in vocabulary
225  vocabulary_->increaseFrequency(triple.predicate());
226  }
227 }
