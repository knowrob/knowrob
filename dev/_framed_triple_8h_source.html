374  template<typename ValT>
375  FramedTripleTemplate(std::string_view subject, std::string_view predicate, const ValT &object, XSDType type)
376  : FramedTriple(),
377  subject_(subject),
378  predicate_(predicate) {
379  if (type == XSDType::STRING) {

381  xsdType_ = type;

383  set(object, type);

387  explicit FramedTripleTemplate(const FramedTriple &other)
388  : FramedTriple() {
389  subject_ = other.subject();
390  predicate_ = other.predicate();
391  isOccasional_ = other.isOccasional();
392  isUncertain_ = other.isUncertain();
393  xsdType_ = other.xsdType();
394  graph_ = other.graph();
395  perspective_ = other.perspective();
396  begin_ = other.begin();
397  end_ = other.end();
398  confidence_ = other.confidence();
399 
400  if (!xsdType_) {
401  object_ = StringType(other.valueAsString());
402  } else {
403  switch (*xsdType_) {
404  case XSDType::STRING:
405  object_ = StringType(other.valueAsString());
406  break;
407  case XSDType::FLOAT:
408  object_ = other.valueAsFloat();
409  break;
410  case XSDType::DOUBLE:
411  object_ = other.valueAsDouble();
412  break;
413  case XSDType::LONG:
414  object_ = other.valueAsLong();
415  break;
416  case XSDType::NON_NEGATIVE_INTEGER:
417  case XSDType::INTEGER:
418  object_ = other.valueAsInt();
419  break;
420  case XSDType::BOOLEAN:
421  object_ = other.valueAsBoolean();
422  break;
423  case XSDType::SHORT:
424  object_ = other.valueAsShort();
425  break;
426  case XSDType::UNSIGNED_LONG:
427  object_ = other.valueAsUnsignedLong();
428  break;
429  case XSDType::UNSIGNED_INT:
430  object_ = other.valueAsUnsignedInt();
431  break;
432  case XSDType::UNSIGNED_SHORT:
433  object_ = other.valueAsUnsignedShort();
434  break;
435  case XSDType::LAST:
436  break;
437  }
438  }
