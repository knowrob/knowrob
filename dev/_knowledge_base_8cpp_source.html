102 void KnowledgeBase::initVocabulary() {
103  auto v_s = std::make_shared<Variable>("?s");
104  auto v_o = std::make_shared<Variable>("?o");
105 
106  for (auto &it: backendManager_->persistent()) {
107  auto backend = it.second;
108 
109  // initialize the import hierarchy
110  for (auto &origin: backend->getOrigins()) {
111  vocabulary_->importHierarchy()->addDirectImport(vocabulary_->importHierarchy()->ORIGIN_SYSTEM, origin);
112  }
113 
114  // iterate over all rdf:type assertions and add them to the vocabulary
115  backend->match(FramedTriplePattern(v_s, rdf::type, v_o),
116  [this](const FramedTriplePtr &triple) {
117  vocabulary_->addResourceType(triple->subject(), triple->valueAsString());
118  });
119  // iterate over all rdfs::subClassOf assertions and add them to the vocabulary
120  backend->match(FramedTriplePattern(v_s, rdfs::subClassOf, v_o),
121  [this](const FramedTriplePtr &triple) {
122  vocabulary_->addSubClassOf(triple->subject(), triple->valueAsString());
123  });
124  // iterate over all rdfs::subPropertyOf assertions and add them to the vocabulary
125  backend->match(FramedTriplePattern(v_s, rdfs::subPropertyOf, v_o),
126  [this](const FramedTriplePtr &triple) {
127  vocabulary_->addSubPropertyOf(triple->subject(), triple->valueAsString());
128  });
129  // iterate over all owl::inverseOf assertions and add them to the vocabulary
130  backend->match(FramedTriplePattern(v_s, owl::inverseOf, v_o),
131  [this](const FramedTriplePtr &triple) {
132  vocabulary_->setInverseOf(triple->subject(), triple->valueAsString());
133  });
134 
135  // query number of assertions of each property/class.
136  // this is useful information for optimizing the query planner.
137  std::vector<semweb::PropertyPtr> reifiedProperties;
138  backend->count([this, &reifiedProperties](std::string_view resource, uint64_t count) {
139  // special handling for reified relations: they are concepts, but do also increase the relation counter
140  auto reifiedProperty = vocabulary_->getDefinedReification(resource);
141  if (reifiedProperty) reifiedProperties.push_back(reifiedProperty);
142  vocabulary_->setFrequency(resource, count);
143  });
144  for (auto &p: reifiedProperties) {
145  vocabulary_->increaseFrequency(p->iri());
146  }
147  }
148 }

315 std::vector<RDFComputablePtr> KnowledgeBase::createComputationSequence(
316  const std::list<DependencyNodePtr> &dependencyGroup) const {
317  // Pick a node to start with.
318  auto comparator = IDBComparator(vocabulary());
319  DependencyNodePtr first;
320  RDFComputablePtr firstComputable;
321  for (auto &n: dependencyGroup) {
322  auto computable_n =
323  std::static_pointer_cast<RDFComputable>(n->literal());
324  if (!first || comparator(firstComputable, computable_n)) {
325  first = n;
326  firstComputable = computable_n;
327  }
328  }
329 
330  // remember visited nodes, needed for circular dependencies
331  // all nodes added to the queue should also be added to this set.
332  std::set<DependencyNode *> visited;
333  visited.insert(first.get());
334 
335  std::vector<RDFComputablePtr> sequence;
336  sequence.push_back(firstComputable);
337 
338  // start with a FIFO queue only containing first node
339  std::deque<std::shared_ptr<DependencyNodeQueue>> queue;
340  auto qn0 = std::make_shared<DependencyNodeQueue>(first);
341  queue.push_front(qn0);
342 
343  // loop until queue is empty and process exactly one successor of
344  // the top element in the FIFO in each step. If an element has no
345  // more successors, it can be removed from queue.
346  // Each successor creates an additional node added to the top of the FIFO.
347  while (!queue.empty()) {
348  auto front = queue.front();
349 
350  // get top successor node that has not been visited yet
351  DependencyNodePtr topNext;
352  while (!front->neighbors_.empty()) {
353  auto topNeighbor = front->neighbors_.top();
354  front->neighbors_.pop();
355 
356  if (visited.count(topNeighbor.get()) == 0) {
357  topNext = topNeighbor;
358  break;
359  }
360  }
361  // pop element from queue if all neighbors were processed
362  if (front->neighbors_.empty()) {
363  queue.pop_front();
364  }
365 
366  if (topNext) {
367  // push a new node onto FIFO
368  auto qn_next = std::make_shared<DependencyNodeQueue>(topNext);
369  queue.push_front(qn_next);
370  sequence.push_back(std::static_pointer_cast<RDFComputable>(topNext->literal()));
371  visited.insert(topNext.get());
372  }
373  }
374 
375  return sequence;
376 }

378 void KnowledgeBase::createComputationPipeline(
379  const std::shared_ptr<QueryPipeline> &pipeline,
380  const std::vector<RDFComputablePtr> &computableLiterals,
381  const std::shared_ptr<TokenBroadcaster> &pipelineInput,
382  const std::shared_ptr<TokenBroadcaster> &pipelineOutput,
383  const QueryContextPtr &ctx) const {
384  // This function generates a query pipeline for literals that can be computed
385  // (EDB-only literals are processed separately). The literals are part of one dependency group.
386  // They are sorted, and also evaluated in this order. For each computable literal there is at
387  // least one reasoner that can compute the literal. However, instances of the literal may also
388  // occur in the EDB. Hence, computation results must be combined with results of an EDB query
389  // for each literal.
390 
391  auto lastOut = pipelineInput;
392 
393  for (auto &lit: computableLiterals) {
394  auto stepInput = lastOut;
395  auto stepOutput = std::make_shared<TokenBroadcaster>();
396  pipeline->addStage(stepOutput);
397 
398  // --------------------------------------
399  // Construct a pipeline that grounds the literal in the EDB.
400  // But only add an EDB stage if the predicate was materialized in EDB before,
401  // or if the predicate is a variable.
402  // --------------------------------------
403  bool isEDBStageNeeded = true;
404  if (lit->propertyTerm() && lit->propertyTerm()->termType() == TermType::ATOMIC) {
405  isEDBStageNeeded = isMaterializedInEDB(
406  std::static_pointer_cast<Atomic>(lit->propertyTerm())->stringForm());
407  }
408  if (isEDBStageNeeded) {
409  auto edb = getBackendForQuery();
410  auto edbStage = std::make_shared<TypedQueryStage<FramedTriplePattern>>(
411  ctx,
412  lit,
413  [&](const FramedTriplePatternPtr &q) {
414  return edb_->getAnswerCursor(edb, std::make_shared<GraphPathQuery>(q, ctx));
415  });
416  edbStage->selfWeakRef_ = edbStage;
417  stepInput >> edbStage;
418  edbStage >> stepOutput;
419  pipeline->addStage(edbStage);
420  }
421 
422  // --------------------------------------
423  // Construct a pipeline that grounds the literal in the IDB.
424  // To this end add an IDB stage for each reasoner that defines the literal.
425  // --------------------------------------
426  for (auto &r: lit->reasonerList()) {
427  auto idbStage = std::make_shared<TypedQueryStage<FramedTriplePattern>>(
428  ctx, lit,
429  [&r, &ctx](const FramedTriplePatternPtr &q) {
430  return r->submitQuery(q, ctx);
431  });
432  idbStage->selfWeakRef_ = idbStage;
433  stepInput >> idbStage;
434  idbStage >> stepOutput;
435  pipeline->addStage(idbStage);
436  // TODO: what about the materialization of the predicate in EDB?
437  }
438 
439  // add a stage that consolidates the results of the EDB and IDB stages.
440  // in particular the case needs to be handled where none of the stages return
441  // 'true'. Also print a warning if two stages disagree but state they are confident.
442  auto consolidator = std::make_shared<DisjunctiveBroadcaster>();
443  stepOutput >> consolidator;
444  // TODO: are all these addStage calls really needed?
445  pipeline->addStage(consolidator);
446 
447  // --------------------------------------
448  // Optionally add a stage to the pipeline that drops all redundant result.
449  // The filter is applied here to remove redundancies early on directly after IDB and EDB
450  // results are combined.
451  // --------------------------------------
452  if (ctx->queryFlags & QUERY_FLAG_UNIQUE_SOLUTIONS) {
453  auto filterStage = std::make_shared<RedundantAnswerFilter>();
454  stepOutput >> filterStage;
455  lastOut = filterStage;
456  } else {
457  lastOut = stepOutput;
458  }
459  }
460 
461  lastOut >> pipelineOutput;
462 }

464 TokenBufferPtr KnowledgeBase::submitQuery(const GraphPathQueryPtr &graphQuery) {
465  auto &allLiterals = graphQuery->path();
466 
467  // --------------------------------------
468  // Construct a pipeline that holds references to stages.
469  // --------------------------------------
470  auto pipeline = std::make_shared<QueryPipeline>();
471 
472  // --------------------------------------
473  // split input literals into positive and negative literals.
474  // negative literals are evaluated in parallel after all positive literals.
475  // --------------------------------------
476  std::vector<FramedTriplePatternPtr> positiveLiterals, negativeLiterals;
477  for (auto &l: allLiterals) {
478  if (l->isNegated()) negativeLiterals.push_back(l);
479  else positiveLiterals.push_back(l);
480  }
481 
482  // --------------------------------------
483  // split positive literals into edb-only and computable.
484  // also associate list of reasoner to computable literals.
485  // --------------------------------------
486  std::vector<FramedTriplePatternPtr> edbOnlyLiterals;
487  std::vector<RDFComputablePtr> computableLiterals;
488  for (auto &l: positiveLiterals) {
489  auto property = l->propertyTerm();
490  if (!property->isAtom()) {
491  KB_WARN("Variable predicate in query not supported.");
492  continue;
493  }
494  auto property_a = std::static_pointer_cast<Atomic>(property);
495  auto l_reasoner = reasonerManager_->getReasonerForRelation(
496  PredicateIndicator(property_a->stringForm(), 2));
497  if (l_reasoner.empty()) {
498  edbOnlyLiterals.push_back(l);
499  auto l_p = l->propertyTerm();
500  if (l_p && l_p->termType() == TermType::ATOMIC &&
501  !isMaterializedInEDB(std::static_pointer_cast<Atomic>(l_p)->stringForm())) {
502  // generate a "don't know" message and return.
503  auto out = std::make_shared<TokenBuffer>();
504  auto channel = TokenStream::Channel::create(out);
505  auto dontKnow = std::make_shared<AnswerDontKnow>();
506  KB_WARN("Predicate {} is neither materialized in EDB nor defined by a reasoner.", *l->predicate());
507  channel->push(dontKnow);
508  channel->push(EndOfEvaluation::get());
509  return out;
510  }
511  } else {
512  computableLiterals.push_back(std::make_shared<RDFComputable>(*l, l_reasoner));
513  }
514  }
515 
516  // --------------------------------------
517  // sort positive literals.
518  // --------------------------------------
519  std::sort(edbOnlyLiterals.begin(), edbOnlyLiterals.end(), EDBComparator(vocabulary()));
520 
521  // --------------------------------------
522  // run EDB query with all edb-only literals.
523  // --------------------------------------
524  std::shared_ptr<TokenBuffer> edbOut;
525  if (edbOnlyLiterals.empty()) {
526  edbOut = std::make_shared<TokenBuffer>();
527  auto channel = TokenStream::Channel::create(edbOut);
528  channel->push(GenericYes());
529  channel->push(EndOfEvaluation::get());
530  } else {
531  auto edb = getBackendForQuery();
532  edbOut = edb_->getAnswerCursor(edb,
533  std::make_shared<GraphPathQuery>(edbOnlyLiterals, graphQuery->ctx()));
534  }
535  pipeline->addStage(edbOut);
536 
537  // --------------------------------------
538  // handle positive IDB literals.
539  // --------------------------------------
540  std::shared_ptr<TokenBroadcaster> idbOut;
541  if (computableLiterals.empty()) {
542  idbOut = edbOut;
543  } else {
544  idbOut = std::make_shared<TokenBroadcaster>();
545  pipeline->addStage(idbOut);
546  // --------------------------------------
547  // Compute dependency groups of computable literals.
548  // --------------------------------------
549  DependencyGraph dg;
550  dg.insert(computableLiterals.begin(), computableLiterals.end());
551 
552  // --------------------------------------
553  // Construct a pipeline for each dependency group.
554  // --------------------------------------
555  if (dg.numGroups() == 1) {
556  auto &literalGroup = *dg.begin();
557  createComputationPipeline(
558  pipeline,
559  createComputationSequence(literalGroup.member_),
560  edbOut,
561  idbOut,
562  graphQuery->ctx());
563  } else {
564  // there are multiple dependency groups. They can be evaluated in parallel.
565 
566  // combines sub-answers computed in different parallel steps
567  auto answerCombiner = std::make_shared<ConjunctiveBroadcaster>();
568  // create a parallel step for each dependency group
569  for (auto &literalGroup: dg) {
570  // --------------------------------------
571  // Construct a pipeline for each dependency group.
572  // --------------------------------------
573  createComputationPipeline(
574  pipeline,
575  createComputationSequence(literalGroup.member_),
576  edbOut,
577  answerCombiner,
578  graphQuery->ctx());
579  }
580  answerCombiner >> idbOut;
581  pipeline->addStage(answerCombiner);
582  }
583  }
584 
585  // --------------------------------------
586  // Evaluate all negative literals in parallel.
587  // --------------------------------------
588  std::shared_ptr<TokenBroadcaster> lastStage;
589  if (!negativeLiterals.empty()) {
590  // run a dedicated stage where negated literals can be evaluated in parallel
591  auto negStage = std::make_shared<PredicateNegationStage>(
592  this, graphQuery->ctx(), negativeLiterals);
593  idbOut >> negStage;
594  lastStage = negStage;
595  } else {
596  lastStage = idbOut;
597  }
598 
599  auto out = std::make_shared<AnswerBuffer_WithReference>(pipeline);
600  lastStage >> out;
601  edbOut->stopBuffering();
602  return out;
603 }

613 TokenBufferPtr KnowledgeBase::submitQuery(const FormulaPtr &phi, const QueryContextPtr &ctx) {
614  auto outStream = std::make_shared<TokenBuffer>();
615 
616  auto pipeline = std::make_shared<QueryPipeline>();
617  pipeline->addStage(outStream);
618 
619  // decompose input formula into parts that are considered in disjunction,
620  // and thus can be evaluated in parallel.
621  QueryTree qt(phi);
622  for (auto &path: qt) {
623  // each node in a path is either a predicate, a negated predicate,
624  // a modal formula, or the negation of a modal formula.
625 
626  std::vector<FramedTriplePatternPtr> posLiterals, negLiterals;
627  std::vector<std::shared_ptr<ModalFormula>> posModals, negModals;
628 
629  for (auto &node: path.nodes()) {
630  switch (node->type()) {
631  case FormulaType::PREDICATE: {
632  auto pat = std::make_shared<FramedTriplePattern>(
633  std::static_pointer_cast<Predicate>(node), false);
634  pat->setTripleFrame(ctx->selector);
635  posLiterals.push_back(pat);
636  break;
637  }
638 
639  case FormulaType::MODAL:
640  posModals.push_back(std::static_pointer_cast<ModalFormula>(node));
641  break;
642 
643  case FormulaType::NEGATION: {
644  auto negation = (Negation *) node.get();
645  auto negated = negation->negatedFormula();
646  switch (negated->type()) {
647  case FormulaType::PREDICATE: {
648  auto pat = std::make_shared<FramedTriplePattern>(
649  std::static_pointer_cast<Predicate>(negated), true);
650  pat->setTripleFrame(ctx->selector);
651  negLiterals.push_back(pat);
652  break;
653  }
654  case FormulaType::MODAL:
655  negModals.push_back(std::static_pointer_cast<ModalFormula>(negated));
656  break;
657  default:
658  throw QueryError("Unexpected negated formula type {} in QueryTree.", (int) negated->type());
659  }
660  break;
661  }
662  default:
663  throw QueryError("Unexpected formula type {} in QueryTree.", (int) node->type());
664  }
665  }
666 
667  std::shared_ptr<TokenBroadcaster> lastStage;
668  std::shared_ptr<TokenBuffer> firstBuffer;
669 
670  // first evaluate positive literals if any
671  // note that the first stage is buffered, so that the next stage can be added to the pipeline
672  // and only after stopping the buffering messages will be forwarded to the next stage.
673  if (posLiterals.empty()) {
674  // if there are none, we still need to indicate begin and end of stream for the rest of the pipeline.
675  // so we just push `bos` (an empty substitution) followed by `eos` and feed these messages to the next stage.
676  firstBuffer = std::make_shared<TokenBuffer>();
677  lastStage = firstBuffer;
678  auto channel = TokenStream::Channel::create(lastStage);
679  channel->push(GenericYes());
680  channel->push(EndOfEvaluation::get());
681  pipeline->addStage(lastStage);
682  } else {
683  auto pathQuery = std::make_shared<GraphPathQuery>(posLiterals, ctx);
684  firstBuffer = submitQuery(pathQuery);
685  lastStage = firstBuffer;
686  pipeline->addStage(lastStage);
687  }
688 
689  // --------------------------------------
690  // Evaluate all positive modals in sequence.
691  // TODO: compute dependency between modals, evaluate independent modals in parallel.
692  // they could also be independent in evaluation context only, we could check which variables receive
693  // grounding already before in posLiteral query!
694  // this is effectively what is done in submitQuery(pathQuery)
695  // --------------------------------------
696  for (auto &posModal: posModals) {
697  // FIXME: use of this pointer below. only ok if destructor makes sure to destroy all pipelines.
698  // maybe a better way would be having another class generating pipelines with the ability
699  // to create reference pointer on a KnowledgeBase, or use a weak ptr here
700  auto modalStage = std::make_shared<ModalStage>(this, posModal, ctx);
701  modalStage->selfWeakRef_ = modalStage;
702  lastStage >> modalStage;
703  lastStage = modalStage;
704  pipeline->addStage(lastStage);
705  }
706 
707  // --------------------------------------
708  // Evaluate all negative literals in parallel.
709  // --------------------------------------
710  if (!negLiterals.empty()) {
711  // run a dedicated stage where negated literals can be evaluated in parallel
712  auto negLiteralStage = std::make_shared<PredicateNegationStage>(
713  this, ctx, negLiterals);
714  lastStage >> negLiteralStage;
715  lastStage = negLiteralStage;
716  pipeline->addStage(lastStage);
717  }
718 
719  // --------------------------------------
720  // Evaluate all negative modals in parallel.
721  // --------------------------------------
722  if (!negModals.empty()) {
723  // run a dedicated stage where negated modals can be evaluated in parallel
724  auto negModalStage = std::make_shared<ModalNegationStage>(
725  this, ctx, negModals);
726  lastStage >> negModalStage;
727  lastStage = negModalStage;
728  pipeline->addStage(lastStage);
729  }
730 
731  lastStage >> outStream;
732  firstBuffer->stopBuffering();
733  pipeline->addStage(lastStage);
734  }
735  // At this point outStream could already contain solutions, but these are buffered
736  // such that they won't be lost during pipeline creation.
737 
738  // if there were multiple paths, consolidate answers from them.
739  // e.g. if one yields no and the other true, the no should be ignored.
740  std::shared_ptr<TokenBroadcaster> finalStage;
741  if (qt.numPaths() > 1) {
742  auto consolidator = std::make_shared<DisjunctiveBroadcaster>();
743  outStream >> consolidator;
744  finalStage = consolidator;
745  } else {
746  finalStage = outStream;
747  }
748 
749  // Finally, wrap output into AnswerBuffer_WithReference object.
750  // Note that the AnswerBuffer_WithReference object is used such that the caller can
751  // destroy the whole pipeline by de-referencing the returned AnswerBufferPtr.
752  auto out = std::make_shared<AnswerBuffer_WithReference>(pipeline);
753  finalStage >> out;
754  outStream->stopBuffering();
755 
756  return out;
757 }

916 bool KnowledgeBase::loadOntologyFile(const std::shared_ptr<OntologyFile> &source, bool followImports) {
917  std::queue<std::string> ontologyURIs;
918  ontologyURIs.push(URI::resolve(source->uri()));
919 
920  while (!ontologyURIs.empty()) {
921  auto resolved = ontologyURIs.front();
922  ontologyURIs.pop();
923  auto origin = DataSource::getNameFromURI(resolved);
924  auto newVersion = DataSource::getVersionFromURI(resolved);
925 
926  // check which backends need to load the ontology.
927  auto backendsToLoad = prepareLoad(origin, newVersion);
928  if (backendsToLoad.empty()) continue;
929 
930  // some OWL files are downloaded compile-time via CMake,
931  // they are downloaded into owl/external e.g. there are SOMA.owl and DUL.owl.
932  // TODO: rework handling of cmake-downloaded ontologies, e.g. should also work when installed
933  auto p = std::filesystem::path(KNOWROB_SOURCE_DIR) / "owl" / "external" /
934  std::filesystem::path(resolved).filename();
935  const std::string *importURI = (exists(p) ? &p.native() : &resolved);
936 
937  KB_INFO("Loading ontology at '{}' with version "
938  "\"{}\" and origin \"{}\".", *importURI, newVersion, origin);
939 
940  OntologyParser parser(*importURI, source->tripleFormat());
941  parser.setOrigin(origin);
942  parser.setFrame(source->frame());
943  // filter is called for each triple, if it returns false, the triple is skipped
944  parser.setFilter([this](const FramedTriple &triple) {
945  return !vocabulary_->isAnnotationProperty(triple.predicate());
946  });
947  // define a prefix for naming blank nodes
948  parser.setBlankPrefix(std::string("_") + origin);
949  auto result = parser.run([this, &backendsToLoad](const TripleContainerPtr &triples) {
950  auto transaction = edb_->createTransaction(
951  getBackendForQuery(),
952  BackendInterface::Insert,
953  BackendInterface::Including,
954  backendsToLoad);
955  transaction->commit(triples);
956  });
957  if (!result) {
958  KB_WARN("Failed to parse ontology {} ({})", *importURI, source->uri());
959  return false;
960  }
961  finishLoad(source, origin, newVersion);
962 
963  // load imported ontologies
964  if (followImports) {
965  for (auto &imported: parser.imports()) {
966  ontologyURIs.push(URI::resolve(imported));
967  }
968  }
969  }
970 
971  return true;
972 }

974 bool KnowledgeBase::loadSPARQLDataSource(const std::shared_ptr<DataSource> &source) {
975  auto service = std::static_pointer_cast<SPARQLService>(source);
976  auto serviceURI = URI::resolve(source->uri());
977  auto origin = service->origin();
978  // SPARQL does not have versioning. Some endpoints may store the version as a triple,
979  // but this is not standardized. Some may encode version in the URI which we try to extract
980  // below. Otherwise, we just use the current day as version causing a re-load every day.
981  auto newVersion = DataSource::getVersionFromURI(serviceURI);
982 
983  // get all backends that do not have the data loaded yet
984  auto backendsToLoad = prepareLoad(origin, newVersion);
985  if (backendsToLoad.empty()) {
986  // data is already loaded
987  return true;
988  }
989  KB_INFO("Loading data from SPARQL endpoint at '{}' with version "
990  "\"{}\" and origin \"{}\".", serviceURI, newVersion, origin);
991 
992  auto result = service->load([this, &backendsToLoad](const TripleContainerPtr &triples) {
993  auto transaction = edb_->createTransaction(
994  getBackendForQuery(),
995  BackendInterface::Insert,
996  BackendInterface::Including,
997  backendsToLoad);
998  transaction->commit(triples);
999  });
1000  if (!result) {
1001  KB_WARN("Failed to load data from SPARQL service at {}.", serviceURI);
1002  return false;
1003  }
1004  finishLoad(service, origin, newVersion);
1005 
1006  return true;
1007 }
