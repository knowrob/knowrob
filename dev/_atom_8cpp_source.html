38 std::shared_ptr<knowrob::Atom> Atom::Tabled(std::string_view name) {
39  auto it = table().find(name);
40  if (it != table().end()) {
41  if (auto atomPtr = it->second.value().lock()) {
42  // Atom still exists, return it
43  return atomPtr;
44  }
45  table().erase(it);
46  }
47  // Atom does not exist or was destroyed, create a new one
48  auto inserted = table().emplace(name, std::nullopt);
49  auto &jt = inserted.first;
50  auto atom = std::make_shared<knowrob::Atom>(jt->first);
51  jt->second = atom;
52  auto locked = jt->second.value().lock();
53  if (!locked) {
54  throw std::runtime_error("Failed to lock Atom");
55  }
56  return locked;
57 }
