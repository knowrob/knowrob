41 void DataDrivenReasoner::doUpdate() {
42  // do the update
43  update();
44  auto now = std::chrono::high_resolution_clock::now();
45  isUpdateQueued_ = false;
46  isInvalidated_ = false;
47 
48  if (!hasFeature(UpdatesItself) && !hasFeature(InvalidatesItself)) {
49  // if the reasoner does not update itself and does not invalidate itself, then
50  // periodically update the reasoner.
51  auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(now - lastUpdate_);
52  if (duration > updateInterval_) {
53  // we cannot keep up with the update rate, so directly re-queue the update.
54  queueUpdate();
55  } else {
56  auto remaining = std::chrono::milliseconds(
57  static_cast<int>(updateInterval_.count()) - duration.count());
58  // Create a timer that triggers the update after the remaining time.
59  // It is not optimal to create a new thread for each update though,
60  // but it would be good to stick to the worker thread pool for performing the
61  // update. However, the worker thread pool does not support delayed execution yet.
62  // Note: this is done so to avoid calling sleep for the worker thread
63  // as it would block it from being used for other tasks.
64  std::thread timeout([&remaining, this]() {
65  std::this_thread::sleep_for(remaining);
66  queueUpdate();
67  });
68  timeout.detach();
69  }
70  }
71 
72  lastUpdate_ = now;
73 }

85 void DataDrivenReasoner::stop() {
86  if (!isRunning_) return;
87  isRunning_ = false;
88  // stop the periodic updating
89  if (!hasFeature(UpdatesItself)) {
90  if (updateRunner_ && !updateRunner_->isTerminated()) {

140 void DataDrivenReasoner::processReplacement(const std::vector<FramedTriplePtr> &triples) {
141  setReasonerOrigin(triples);
142 
143  if (inferredTriples_.empty()) {
144  inferredTriples_.insert(triples.begin(), triples.end());
145  reasonerManager().kb()->insertAll(triples);
146  } else {
147  auto &oldTriples = inferredTriples_;
148  // ensure that input triples are sorted which is required for set_difference
149  std::set<FramedTriplePtr> newTriples(triples.begin(), triples.end());
150 
151  std::vector<FramedTriplePtr> triplesToRemove, triplesToAdd;
152  // old inferences without new inferences are the ones that do not hold anymore.
153  std::set_difference(oldTriples.begin(), oldTriples.end(),
154  newTriples.begin(), newTriples.end(),
155  std::inserter(triplesToRemove, triplesToRemove.begin()));
156  // new inferences without old inferences are the ones that are really new.
157  std::set_difference(newTriples.begin(), newTriples.end(),
158  oldTriples.begin(), oldTriples.end(),
159  std::inserter(triplesToAdd, triplesToAdd.begin()));
160  // update the set of inferred triples.
161  for (auto &triple: triplesToRemove) {
162  inferredTriples_.erase(triple);
163  }
164  inferredTriples_.insert(triplesToAdd.begin(), triplesToAdd.end());
165  // update the knowledge base
166  if (!triplesToAdd.empty()) {
167  reasonerManager().kb()->insertAll(triplesToAdd);
168  }
169  if (!triplesToRemove.empty()) {
170  reasonerManager().kb()->removeAll(triplesToRemove);
171  }
172  }
173 }
