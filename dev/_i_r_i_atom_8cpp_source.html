19 std::shared_ptr<knowrob::IRIAtom> IRIAtom::Tabled(std::string_view name) {
20  auto it = Atom::table().find(name);
21  if (it != table().end()) {
22  if (auto atomPtr = it->second.value().lock()) {
23  if (atomPtr->isIRI()) {
24  return std::static_pointer_cast<IRIAtom>(atomPtr);
25  }
26  }
27  Atom::table().erase(it);
28  }
29  // Atom does not exist or was destroyed, create a new one
30  auto inserted = table().emplace(name, std::nullopt);
31  auto &jt = inserted.first;
32  auto atom = std::make_shared<knowrob::IRIAtom>(jt->first);
33  jt->second = atom;
34  auto locked = jt->second.value().lock();
35  if (!locked) {
36  throw std::runtime_error("Failed to lock IRIAtom");
37  }
38  return std::static_pointer_cast<IRIAtom>(locked);
39 }
