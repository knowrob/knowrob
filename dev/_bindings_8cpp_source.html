102 bool Bindings::unifyWith(const Bindings &other) {
103  for (const auto &pair: other.mapping_) {
104  auto it = mapping_.find(pair.first);
105  if (it == mapping_.end()) {
106  // new variable instantiation
107  auto jt = mapping_.insert(pair).first;
108  } else {
109  // variable has already an instantiation, need to unify
110  TermPtr t0 = it->second.second;
111  TermPtr t1 = pair.second.second;
112 
113  // t0 and t1 are not syntactically equal -> compute a unifier
114  Unifier sigma(t0, t1);
115  if (sigma.exists()) {
116  // a unifier exists
117  it->second.second = sigma.apply();
118  } else {
119  // no unifier exists
120  return false;
121  }
122  }
123  }
124 
125  return true;
126 }

149  FormulaPtr applyBindings(const FormulaPtr &phi, const Bindings &bindings) { //NOLINT
150  if (phi->isGround()) {
151  return phi;
152  }
153  switch (phi->type()) {
154  case FormulaType::MODAL: {
155  auto modal = std::static_pointer_cast<ModalFormula>(phi);
156  auto inner = applyBindings(modal->modalFormula(), bindings);
157  if (inner == modal->modalFormula()) {
158  return modal;
159  } else {
160  return std::make_shared<ModalFormula>(modal->modalOperator(), inner);
161  }
162  }
163  case FormulaType::IMPLICATION: {
164  auto implication = std::static_pointer_cast<Implication>(phi);
165  auto antecedent = applyBindings(implication->antecedent(), bindings);
166  auto consequent = applyBindings(implication->consequent(), bindings);
167  if (antecedent == implication->antecedent() && consequent == implication->consequent()) {
168  return implication;
169  } else {
170  return std::make_shared<Implication>(antecedent, consequent);
171  }
172  }
173  case FormulaType::NEGATION: {
174  auto negation = std::static_pointer_cast<Negation>(phi);
175  auto negated = applyBindings(negation->negatedFormula(), bindings);
176  if (negated == negation->negatedFormula()) {
177  return negation;
178  } else {
179  return std::make_shared<Negation>(negated);
180  }
181  }
182  case FormulaType::CONJUNCTION:
183  return applyCompoundBindings<Conjunction>(std::static_pointer_cast<Conjunction>(phi), bindings);
184  case FormulaType::DISJUNCTION:
185  return applyCompoundBindings<Disjunction>(std::static_pointer_cast<Disjunction>(phi), bindings);
186  case FormulaType::PREDICATE: {
187  auto predicate = std::static_pointer_cast<Predicate>(phi);
188  auto args = predicate->arguments();
189  auto newArgs = std::vector<TermPtr>(args.size());
190  auto hasNewArg = false;
191  for (uint32_t i = 0; i < args.size(); i++) {
192  auto arg = applyBindings(args[i], bindings);
193  if (arg != args[i]) {
194  hasNewArg = true;
195  }
196  newArgs[i] = arg;
197  }
198  if (!hasNewArg) {
199  return predicate;
200  } else {
201  return std::make_shared<Predicate>(predicate->functor(), newArgs);
202  }
203  }
204  }
205  return phi;
206  }

208  TermPtr applyBindings(const TermPtr &t, const Bindings &bindings) { //NOLINT
209  if (t->isGround()) {
210  return t;
211  }
212  switch (t->termType()) {
213  case TermType::ATOMIC:
214  return t;
215  case TermType::VARIABLE: {
216  auto var = std::static_pointer_cast<Variable>(t);
217  auto term = bindings.get(var->name());
218  if (term) {
219  return term;
220  } else {
221  return var;
222  }
223  }
224  case TermType::FUNCTION: {
225  auto function = std::static_pointer_cast<Function>(t);
226  auto args = function->arguments();
227  auto newArgs = std::vector<TermPtr>(args.size());
228  auto hasNewArg = false;
229  for (uint32_t i = 0; i < args.size(); i++) {
230  auto arg = applyBindings(args[i], bindings);
231  if (arg != args[i]) {
232  hasNewArg = true;
233  }
234  newArgs[i] = arg;
235  }
236  if (!hasNewArg) {
237  return function;
238  } else {
239  return std::make_shared<Function>(function->functor(), newArgs);
240  }
241  }
242  }
243  return t;
244  }
245 }
