15 static inline bool has_intersection(const std::set<std::string_view> &a, const std::set<std::string_view> &b)
16 {
17  const std::set<std::string_view> &smaller = (a.size()<b.size() ? a : b);
18  const std::set<std::string_view> &larger = ((&smaller == &a) ? b : a);

20  [larger](auto &v){ return larger.count(v)>0; });

39 void DependencyGraph::insert(const DependencyNodePtr &newNode)
40 {
41  nodes_.push_back(newNode);
42 
43  // find the modal node groups that have a shared variable with newNode.
44  std::list<std::list<DependencyGroup>::iterator > dependencies;
45  for(auto it=groups_.begin(); it!=groups_.end(); ++it) {
46  if(has_intersection(it->variables_, newNode->variables())) {
47  dependencies.push_back(it);
48  }
49  }
50 
51  DependencyGroup *newNodeGroup = nullptr;
52  if(dependencies.empty()) {
53  // the literal does not have a shared variable with an existing group.
54  // create a new one with just the literal as a member.
55  auto &newGroup = groups_.emplace_back();
56  newNodeGroup = &newGroup;
57  }
58  else if(dependencies.size()==1) {
59  // add to group
60  auto &group = *dependencies.front();
61  newNodeGroup = &group;
62  }
63  else {
64  // merge multiple groups into one
65  auto &newGroup = groups_.emplace_back();
66  for(auto groupIterator : dependencies) {
67  newGroup += *groupIterator;
68  groups_.erase(groupIterator);
69  }
70  newNodeGroup = &newGroup;
71  }
72  // update neighbor relation based on shared variables with other nodes
73  for(auto &x : newNodeGroup->member_) {
74  if(has_intersection(x->variables(), newNode->variables())) {
75  x->neighbors_.push_back(newNode);
76  newNode->neighbors_.push_back(x);
77  }
78  }
79  // finally add the new node to the group
80  newNodeGroup->member_.push_back(newNode);
81  newNodeGroup->variables_.insert(newNode->variables().begin(), newNode->variables().end());
82 }
