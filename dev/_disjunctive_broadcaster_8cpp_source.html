15 void DisjunctiveBroadcaster::pushDeferredMessages() {
16  if (isCertainlyPositive_) {
17  for (auto &x: deferredPositiveAnswers_) {
18  TokenBroadcaster::push(x);
19  }
20  } else {
21  // neither a certain positive nor a certain negative answer has been produced.
22  if (deferredPositiveAnswers_.empty()) {

24  TokenBroadcaster::push(negativeAnswers_.back());
25  } else if (!negativeAnswers_.empty()) {

37  }
38  }

43 void DisjunctiveBroadcaster::pushAnswer(const AnswerPtr &answer) {
44  // note that this stage consolidates results of independent evaluations that
45  // are considered in disjunction. Thus, a positive answer from one evaluation
46  // is sufficient to consider the query as satisfiable.
47  // However, a negative answer from one evaluation is not sufficient to consider
48  // the query as unsatisfiable, as there might be other evaluations that
49  // consider the query as satisfiable.
50 
51  if (answer->isNegative()) {
52  negativeAnswers_.emplace_back(std::static_pointer_cast<const AnswerNo>(answer));
53  } else if (answer->isPositive()) {
54  // a positive answer indicates that a subsystem suggests the input query is satisfiable.
55  // however, the answer can be uncertain, so the "YES" answer is only
56  // pushed directly in case it is a certain answer.
57  // pushing of uncertain answers is delayed until a certain answer has been produced,
58  // or until the end of the stream is reached.
59  auto positiveAnswer = std::static_pointer_cast<const AnswerYes>(answer);
60 
61  if (positiveAnswer->isCertain()) {
62  isCertainlyPositive_ = true;
63  TokenBroadcaster::push(answer);
64  pushDeferredMessages();
65  } else if (isCertainlyPositive_) {
66  // push uncertain positive message if a certain positive answer has been produced
67  TokenBroadcaster::push(answer);
68  } else {
69  // else defer pushing uncertain positive answer until a certain answer has been produced, or eof reached
70  deferredPositiveAnswers_.emplace_back(positiveAnswer);
71  }
72  } else {
73  // neither a positive nor a negative answer, i.e. reasoning system has no clue
74  }
75 }
