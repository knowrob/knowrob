47 static FormulaPtr toFormula_recursive(GraphTerm *term) { //NOLINT
48  switch (term->termType()) {
49  case GraphTermType::Pattern: {
50  auto pattern = ((GraphPattern *) term)->value();
51  if (pattern->isNegated()) {
52  return std::make_shared<Negation>(pattern->predicate());
53  } else {
54  return pattern->predicate();
55  }
56  }
57  case GraphTermType::Union: {
58  auto &terms = ((GraphUnion *) term)->terms();
59  std::vector<FormulaPtr> formulae;
60  for (const auto &t: terms) {
61  auto phi = toFormula_recursive(t.get());
62  if (phi) {
63  formulae.push_back(phi);
64  }
65  }
66  return std::make_shared<Disjunction>(formulae);
67  }
68  case GraphTermType::Sequence: {
69  auto &terms = ((GraphSequence *) term)->terms();
70  std::vector<FormulaPtr> formulae;
71  for (const auto &t: terms) {
72  auto phi = toFormula_recursive(t.get());
73  if (phi) {
74  formulae.push_back(phi);
75  }
76  }
77  return std::make_shared<Conjunction>(formulae);
78  }
79  case GraphTermType::Builtin:
80  break;
81  }
82  return nullptr;
83 }
