108 void MongoKnowledgeGraph::initializeMongo() {
109  // make sure s/p/o index is defined
110  tripleCollection_->createTripleIndex();
111  // a collection with just a single document used for querying
112  oneCollection_ = std::make_shared<Collection>(
113  tripleCollection_->connection(),
114  tripleCollection_->dbName().c_str(),
115  MONGO_KG_ONE_COLLECTION);
116  // Make sure there is one document in the "one" collection.
117  // The collection is used to initiate a pipeline for a single input document.
118  if (oneCollection_->empty()) {
119  Document oneDoc(bson_new());
120  bson_t scopeDoc, timeDoc;
121  bson_decimal128_t infinity, zero;
122  bson_decimal128_from_string(BSON_DECIMAL128_INF, &infinity);
123  bson_decimal128_from_string("0", &zero);
124  // TODO: time interval is optional now, right? so we do not need to add the fields here? test it!
125  // but maybe mongolog still needs it
126  BSON_APPEND_DOCUMENT_BEGIN(oneDoc.bson(), "v_scope", &scopeDoc);
127  BSON_APPEND_DOCUMENT_BEGIN(&scopeDoc, "time", &timeDoc);
128  BSON_APPEND_DECIMAL128(&timeDoc, "since", &zero);
129  BSON_APPEND_DECIMAL128(&timeDoc, "until", &infinity);
130  bson_append_document_end(&scopeDoc, &timeDoc);
131  bson_append_document_end(oneDoc.bson(), &scopeDoc);
132  oneCollection_->storeOne(oneDoc);
133  }
134  // Create an object used for taxonomy operations
135  taxonomy_ = std::make_shared<MongoTaxonomy>(tripleCollection_, oneCollection_);
136 }

192 bool MongoKnowledgeGraph::insertOne(const FramedTriple &tripleData) {
193  auto &fallbackOrigin = vocabulary_->importHierarchy()->defaultGraph();
194  bool isTaxonomic = vocabulary_->isTaxonomicProperty(tripleData.predicate());
195  MongoTriple mngTriple(vocabulary_, tripleData, fallbackOrigin, isTaxonomic);
196  tripleCollection_->storeOne(mngTriple.document());
197 
198  if (isSubClassOfIRI(tripleData.predicate())) {
199  taxonomy_->update({{tripleData.subject(), tripleData.valueAsString()}}, {});
200  } else if (isSubPropertyOfIRI(tripleData.predicate())) {
201  taxonomy_->update({}, {{tripleData.subject(), tripleData.valueAsString()}});
202  }
203 
204  return true;
205 }

207 bool MongoKnowledgeGraph::insertAll(const TripleContainerPtr &triples) {
208  // only used in case triples do not specify origin field
209  auto &fallbackOrigin = vocabulary_->importHierarchy()->defaultGraph();
210  auto bulk = tripleCollection_->createBulkOperation();
211  struct TaxonomyAssertions {
212  std::vector<MongoTaxonomy::StringPair> subClassAssertions;
213  std::vector<MongoTaxonomy::StringPair> subPropertyAssertions;
214  } tAssertions;
215 
216  std::for_each(triples->begin(), triples->end(),
217  [&](auto &data) {
218  MongoTriple mngTriple(vocabulary_, *data, fallbackOrigin,
219  vocabulary_->isTaxonomicProperty(data->predicate().data()));
220  bulk->pushInsert(mngTriple.document().bson());
221 
222  if (isSubClassOfIRI(data->predicate())) {
223  tAssertions.subClassAssertions.emplace_back(data->subject(), data->valueAsString());
224  } else if (isSubPropertyOfIRI(data->predicate())) {
225  tAssertions.subPropertyAssertions.emplace_back(data->subject(), data->valueAsString());
226  }
227  });
228  bulk->execute();
229 
230  taxonomy_->update(tAssertions.subClassAssertions, tAssertions.subPropertyAssertions);
231 
232  return true;
233 }

272 void MongoKnowledgeGraph::count(const ResourceCounter &callback) const {
273  for (auto &filename: {PIPELINE_RELATION_COUNTER, PIPELINE_CLASS_COUNTER}) {
274  const bson_t *result;
275  Cursor cursor(tripleCollection_);
276  Document document(Pipeline::loadFromJSON(
277  filename, {
278  {"COLLECTION", tripleCollection_->name()}
279  }));
280  cursor.aggregate(document.bson());
281  while (cursor.next(&result)) {
282  bson_iter_t iter;
283  if (!bson_iter_init(&iter, result)) break;
284  if (!bson_iter_find(&iter, "resource")) break;
285  auto property = bson_iter_utf8(&iter, nullptr);
286  if (!bson_iter_find(&iter, "count")) break;
287  auto count = bson_iter_as_int64(&iter);
288  callback(property, count);
289  }
290  }

293 void MongoKnowledgeGraph::iterate(TripleCursor &cursor, const TripleVisitor &visitor) {
294  FramedTripleView tripleData;
295  FramedTriplePtr triplePtr;
296  triplePtr.ptr = &tripleData;
297  // Mongo cursor own the allocation, and the memory of a document will be deallocated by the cursor during iteration.
298  // @see https://mongoc.org/libmongoc/current/mongoc_cursor_next.html
299  // So it cannot be allowed that the visitor takes over ownership, hence owned is set to false.
300  triplePtr.owned = false;
301 
302  // iterate over matching documents
303  while (cursor.nextTriple(*triplePtr.ptr)) {
304  visitor(triplePtr);
305  }
306 }

313 void MongoKnowledgeGraph::batch(const TripleHandler &callback) const {
314  TripleCursor cursor(tripleCollection_);
315  std::vector<FramedTriplePtr> batchData(GlobalSettings::batchSize());
316  uint32_t currentSize = 0;
317 
318  while (true) {
319  auto &current = batchData[currentSize];
320  if (current.ptr && current.owned) {
321  delete current.ptr;
322  }
323  current.ptr = new FramedTripleCopy();
324  current.owned = true;
325 
326  if (!cursor.nextTriple(*current.ptr)) {
327  break;
328  }
329  currentSize++;
330  if (currentSize == GlobalSettings::batchSize()) {
331  auto batch = std::make_shared<ProxyTripleContainer>(&batchData);
332  callback(batch);
333  currentSize = 0;
334  }
335  }
336  if (currentSize != 0) {
337  batchData.resize(currentSize);
338  auto batch = std::make_shared<ProxyTripleContainer>(&batchData);
339  callback(batch);
340  }
341 }

343 void MongoKnowledgeGraph::match(const FramedTriplePattern &query, const TripleVisitor &visitor) {
344  bool b_isTaxonomicProperty;
345  if (query.propertyTerm()->termType() == TermType::ATOMIC) {
346  b_isTaxonomicProperty = vocabulary_->isTaxonomicProperty(((Atomic *) query.propertyTerm().get())->stringForm());
347  } else {
348  b_isTaxonomicProperty = false;
349  }
350  TripleCursor cursor(tripleCollection_);
351  // filter documents by triple pattern
352  MongoTriplePattern mngQuery(query, b_isTaxonomicProperty, vocabulary_->importHierarchy());
353  cursor.filter(mngQuery.bson());
354  iterate(cursor, visitor);
355 }

358 static inline BindingsCursorPtr doLookup(const T &query, const TripleStore &store) {
359  bson_t pipelineDoc = BSON_INITIALIZER;

361  BSON_APPEND_ARRAY_BEGIN(&pipelineDoc, "pipeline", &pipelineArray);

364  bson_append_array_end(&pipelineDoc, &pipelineArray);
365
